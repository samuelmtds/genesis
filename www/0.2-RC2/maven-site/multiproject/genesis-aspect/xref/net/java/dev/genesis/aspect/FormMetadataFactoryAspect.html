68                 AnnotationHandler {
69              public void processFormAnnotation(final FormMetadata formMetadata,
70                    final Annotation annotation) {
71  
72                 final String value = ((UntypedAnnotationProxy) annotation)
73                       .getValue();
74                 final int equalIndex = value.indexOf('=');
75                 if (equalIndex > 0) {
76                    formMetadata.addNamedCondition(value.substring(0, equalIndex)
77                          .trim(), JXPathContext.compile(value
78                          .substring(equalIndex + 1)));
79                 }
80              }
81  
82              public void processFieldAnnotation(final FormMetadata formMetadata,
83                    final FieldMetadata fieldMetadata, final Annotation annotation) {
84                 processFormAnnotation(formMetadata, annotation);
85              }
86  
87              public void processMethodAnnotation(
88                    final FormMetadata formMetadata,
89                    final MethodMetadata methodMetadata,
90                    final Annotation annotation) {
91                 processFormAnnotation(formMetadata, annotation);
92              }

234             public void processMethodAnnotation(
235                   final FormMetadata formMetadata,
236                   final MethodMetadata methodMetadata,
237                   final Annotation annotation) {
238 
239                final Map attributesMap = GenesisUtils.getAttributesMap(
240                      ((UntypedAnnotationProxy) annotation).getValue());
241                final String widgetName = (String)attributesMap
242                      .get("widgetName");
243                final String objectFieldName = (String)attributesMap
244                      .get("objectField");
245                final String indexFieldName = (String)attributesMap
246                      .get("indexField");
247                final String callOnInit = (String)attributesMap
248                      .get("callOnInit");
249 
250                final DataProviderMetadata dataProviderMetadata = methodMetadata
251                      .getDataProviderMetadata();
252 
253                if (widgetName == null && objectFieldName == null && indexFieldName == null) {
254                   throw new RuntimeException("At least one of widgetName, objectField or " + 
255                         "indexField must be specified for @DataProvider in " +
256                         methodMetadata.getMethodEntry().getMethodName());
257                }
258 
259                dataProviderMetadata.setCallOnInit(!"false".equals(callOnInit));
260 
261                PropertyDescriptor[] descriptors = PropertyUtils
262                      .getPropertyDescriptors(formMetadata.getFormClass());
263 
264                Map descriptorsPerPropertyName = new HashMap();
265 
266                for (int i = 0; i < descriptors.length; i++) {
267                   descriptorsPerPropertyName.put(descriptors[i].getName(), 
268                         descriptors[i]);
269                }
270 
271                if (objectFieldName != null) {
272                   PropertyDescriptor descriptor = (PropertyDescriptor)
273                         descriptorsPerPropertyName.get(objectFieldName);
274 
275                   if (descriptor == null) {
276                      throw new RuntimeException("The object "
277                            + formMetadata.getFormClass()
278                            + " doesn´t have a field called " + objectFieldName);
279                   }
280 
281                   final Class fieldType = descriptor.getPropertyType();
282 
283                   if (fieldType.isPrimitive()
284                         || (fieldType.isArray() &&
285                               fieldType.getComponentType().isPrimitive())) {
286                      throw new RuntimeException("The object " + objectFieldName
287                            + " has an object field called " + objectFieldName
288                            + " that cannot be primitive or array of primitives");
289                   }
290 
291                   dataProviderMetadata.setObjectField(new FieldEntry(
292                         descriptor.getName(), descriptor.getPropertyType()));
293                }
294 
295                if (indexFieldName != null) {
296                   PropertyDescriptor descriptor = (PropertyDescriptor)
297                         descriptorsPerPropertyName.get(indexFieldName);
298 
299                   if (descriptor == null) {
300                      throw new RuntimeException("The object "
301                            + formMetadata.getFormClass()
302                            + " doesn´t have a field called " + indexFieldName);
303                   }
304 
305                   final Class fieldType = descriptor.getPropertyType()
306                         .isArray() ? descriptor.getPropertyType()
307                         .getComponentType() : descriptor.getPropertyType();
308 
309                   if (!Collection.class.isAssignableFrom(descriptor
310                         .getPropertyType())
311                         && !Integer.TYPE.isAssignableFrom(fieldType)
312                         && !Integer.class.isAssignableFrom(fieldType)) {
313                      throw new RuntimeException(
314                            "The object "
315                                  + objectFieldName
316                                  + " has an index field called "
317                                  + indexFieldName
318                                  + " that's not an Integer, int, Collection of them, or array of them.");
319                   }
320 
321                   dataProviderMetadata.setIndexField(new FieldEntry(descriptor
322                         .getName(), descriptor.getPropertyType()));
323                }
324 
325                dataProviderMetadata
326                      .setWidgetName(widgetName != null ? widgetName
327                            : objectFieldName != null ? objectFieldName
328                                  : indexFieldName);
329             }
330          }

506       public FormMetadata getFormMetadata(final Class formClass) {
507          FormMetadata formMetadata = (FormMetadata) cache.get(formClass);
508          if (formMetadata == null) {
509             formMetadata = new FormMetadata(formClass);
510             processAnnotations(formMetadata);
511             cache.put(formClass, formMetadata);
512          }
513          return formMetadata;

516       private void processAnnotations(final FormMetadata formMetadata) {
517          processFormAnnotations(formMetadata);
518          processFieldsAnnotations(formMetadata);

522       private void processFormAnnotations(final FormMetadata formMetadata) {
523          final List annotations = Annotations.getAnnotationInfos(formMetadata
524                .getFormClass());
525          AnnotationInfo info;
526          MetadataAttribute attr;
527          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
528             info = (AnnotationInfo) iter.next();
529             attr = MetadataAttribute.get(info.getName());
530 
531             if (attr == null) {
532                continue;
533             }
534 
535             attr.getHandler().processFormAnnotation(formMetadata,
536                   info.getAnnotation());
537          }
538       }

540       private void processFieldsAnnotations(final FormMetadata formMetadata) {
541          final Class clazz = formMetadata.getFormClass();
542          final PropertyDescriptor[] propertyDescriptors = PropertyUtils
543                .getPropertyDescriptors(clazz);
544          PropertyDescriptor propDesc;
545          FieldMetadata fieldMetadata;
546          for (int i = 0; i < propertyDescriptors.length; i++) {
547             propDesc = propertyDescriptors[i];
548 
549             // Ignoring java.lang.Object.getClass()
550             if (propDesc.getName().equals("class")
551                   || propDesc.getReadMethod() == null
552                   || Annotations.getAnnotation("NotBound", propDesc
553                         .getReadMethod()) != null) {
554                continue;
555             }
556 
557             fieldMetadata = new FieldMetadata(propDesc.getName(), propDesc
558                   .getPropertyType(), propDesc.getWriteMethod() != null);
559             formMetadata.addFieldMetadata(propDesc.getName(), fieldMetadata);
560             processFieldAnnotations(formMetadata, fieldMetadata, getReadMethod(
561                   clazz, propDesc));
562 
563          }
564       }

585       private void processMethodsAnnotations(final FormMetadata formMetadata) {
586          final Method[] methods = formMetadata.getFormClass().getMethods();
587          boolean isAction;
588          boolean isProvider;
589          MethodMetadata methodMetadata;
590 
591          for (int i = 0; i < methods.length; i++) {
592             isAction = Annotations.getAnnotation("Action", methods[i]) != null;
593             isProvider = Annotations.getAnnotation("DataProvider", methods[i]) != null;
594 
595             if (!isAction && !isProvider) {
596                continue;
597             }
598 
599             methodMetadata = new MethodMetadata(methods[i], isAction, isProvider);
600             formMetadata.addMethodMetadata(methods[i], methodMetadata);
601             processMethodAnnotations(formMetadata, methodMetadata, methods[i]);
602          }
603       }

605       private void processMethodAnnotations(final FormMetadata formMetadata,
606             final MethodMetadata methodMetadata, final Method actionMethod) {
607          final List annotations = Annotations.getAnnotationInfos(actionMethod);
608          AnnotationInfo info;
609          MetadataAttribute attr;
610          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
611             info = (AnnotationInfo) iter.next();
612             attr = MetadataAttribute.get(info.getName());
613 
614             if (attr == null) {
615                continue;
616             }
617 
618             attr.getHandler().processMethodAnnotation(formMetadata,
619                   methodMetadata, info.getAnnotation());
620          }

623       // Work around for a bug in JDK 1.4.2
624       private Method getReadMethod(Class clazz, PropertyDescriptor pd) {
625          Method method = pd.getReadMethod();
626 
627          if (clazz.equals(method.getDeclaringClass())) {
628             return method;
629          }
630 
631          try {
632             method = clazz.getMethod(method.getName(), 
633                   method.getParameterTypes());
634             return (method == null) ? pd.getReadMethod() : method;
635          } catch (NoSuchMethodException nsme) {
636             return method;
637          }
