53     public static boolean validateBigDecimalRange(Object bean, Field field) {
54        String value = ValidatorUtils.getValueAsString(bean, field.getProperty());
55        
56        if (!validateBigDecimal(bean, field)) {
57           return false;
58        }
59        
60        BigDecimal inputValue = normalize(ConvertUtils.convert(value,
61              BigDecimal.class));
62        BigDecimal min = normalize(ConvertUtils.convert(field.getVarValue("min"),
63              BigDecimal.class));
64        BigDecimal max = normalize(ConvertUtils.convert(field.getVarValue("max"),
65              BigDecimal.class));
66        
67        return min.compareTo(inputValue) <= 0 && max.compareTo(inputValue) >= 0;
68     }

116    public static boolean validateRequiredIf(Object bean, Field field,
117    Validator validator) {
118       final Object form = validator.getParameterValue(Validator.BEAN_PARAM);
119       String value = null;
120       boolean required = false;
121       
122       if (isString(bean)) {
123          value = (String) bean;
124       } else {
125          value = ValidatorUtils.getValueAsString(bean, field.getProperty());
126       }
127       
128       String fieldJoin = "AND";
129       
130       if (!GenericValidator.isBlankOrNull(field.getVarValue("fieldJoin"))) {
131          fieldJoin = field.getVarValue("fieldJoin");
132       }
133       
134       if (fieldJoin.equalsIgnoreCase("AND")) {
135          required = true;
136       }
137       
138       for (int i = 0 ; !GenericValidator.isBlankOrNull(field.getVarValue(
139       "field[" + i + "]")); i++) {
140          String dependProp = field.getVarValue("field[" + i + "]");
141          String dependTest = field.getVarValue("fieldTest[" + i + "]");
142          String dependTestValue = field.getVarValue("fieldValue[" + i + "]");
143          String dependIndexed = field.getVarValue("fieldIndexed[" + i + "]");
144          
145          if (dependIndexed == null) {
146             dependIndexed = "false";
147          }
148          
149          String dependVal = null;
150          boolean thisRequired = false;
151          
152          if (field.isIndexed() && dependIndexed.equalsIgnoreCase("true")) {
153             String key = field.getKey();
154             
155             if ((key.indexOf("[") > -1) && (key.indexOf("]") > -1)) {
156                String ind = key.substring(0, key.indexOf(".") + 1);
157                dependProp = ind + dependProp;
158             }
159          }
160          
161          dependVal = ValidatorUtils.getValueAsString(form, dependProp);
162          
163          if (dependTest.equals(FIELD_TEST_NULL) ||
164          dependTest.equals(FIELD_TEST_NOTNULL)) {
165             thisRequired = GenericValidator.isBlankOrNull(dependVal);
166             
167             if (dependTest.equals(FIELD_TEST_NOTNULL)) {
168                thisRequired = !thisRequired;
169             }
170          }
171          
172          if (dependTest.equals(FIELD_TEST_EQUAL)) {
173             thisRequired =  dependTestValue.equalsIgnoreCase(dependVal);
174          }
175          
176          if (fieldJoin.equalsIgnoreCase("AND")) {
177             required = required && thisRequired;
178          } else {
179             required = required || thisRequired;
180          }
181       }
182       
183       if (required) {
184          return ((value != null) && (value.length() > 0));
185       }
186       
187       return true;
188    }
