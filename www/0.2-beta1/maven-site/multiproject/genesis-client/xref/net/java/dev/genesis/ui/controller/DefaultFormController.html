113    public void setup() throws Exception {
114       ctx = createJXPathContext();
115       ctx.getVariables().declareVariable(FORM_METADATA_KEY, formMetadata);
116       ctx.getVariables().declareVariable(CHANGED_MAP_KEY, changedMap);
117       if (PropertyUtils.isWriteable(form, "context")
118             && Map.class.isAssignableFrom(PropertyUtils.getPropertyType(form,
119                   "context"))) {
120          PropertyUtils.setProperty(form, "context", getVariablesMap());
121       }
122 
123       evaluate(false);
124 
125       last.putAll(PropertyUtils.describe(form));
126       lastSaved.clear();
127       lastSaved.putAll(last);
128 
129       ctx.getVariables().declareVariable(LAST_MAP_KEY, last);
130       ctx.getVariables().declareVariable(LAST_SAVED_MAP_KEY, lastSaved);
131    }

137    protected void populate(Map properties, boolean stringMap) throws Exception {
138       updateChangedMap(properties, stringMap, true);
139 
140       if (changedMap.isEmpty()) {
141          log.debug("Nothing changed.");
142          return;
143       }
144 
145       PropertyUtils.copyProperties(form, changedMap);
146       evaluate(true);
147    }

149    private void updateChangedMap(Map newData, boolean stringMap,
150          boolean logMessages) {
151       Object value;
152       FieldMetadata fieldMeta;
153       Map.Entry entry;
154 
155       changedMap.clear();
156       changedMap.putAll(GenesisUtils.normalizeMap(newData));
157 
158       if (log.isDebugEnabled()) {
159          log.debug("changedMap started as: " + changedMap);
160       }
161 
162       for (final Iterator i = changedMap.entrySet().iterator(); i.hasNext();) {
163          entry = (Map.Entry) i.next();
164          fieldMeta = formMetadata.getFieldMetadata(entry.getKey().toString());
165 
166          if (fieldMeta.isDisplayOnly()) {
167             i.remove();
168 
169             if (log.isDebugEnabled()) {
170                log.debug(entry.getKey()
171                      + " removed from changedMap for being a "
172                      + " displayOnly field");
173             }
174 
175             continue;
176          }
177 
178          value = (stringMap) ? fieldMeta.getConverter().convert(
179                fieldMeta.getFieldClass(), entry.getValue()) : entry.getValue();
180 
181          if (fieldMeta.getEqualityComparator().equals(value,
182                last.get(entry.getKey()))) {
183             i.remove();
184 
185             if (log.isDebugEnabled()) {
186                log.debug(entry.getKey() + " removed from changedMap for being "
187                      + " equal to previousValue; is " + entry.getValue()
188                      + "; was:" + last.get(entry.getKey()));
189             }
190 
191             continue;
192          }
193 
194          if (/*logMessages && */log.isDebugEnabled()) {
195             log.debug("Field '" + entry.getKey() + "' changed to '"
196                   + entry.getValue() + "'; was " + last.get(entry.getKey()));
197          }
198 
199          entry.setValue(value);
200       }
201    }

208    protected void evaluate(boolean updateMaps) throws Exception {
209       evaluateClearOnConditions();
210       evaluateNamedConditions();
211       evaluateEnabledWhenConditions();
212       evaluateVisibleWhenConditions();
213       evaluateCallWhenConditions();
214       evaluateDataProviderTriggers();
215 
216       if (!updateMaps) {
217          return;
218       }
219 
220       final Map newData = PropertyUtils.describe(form);
221       updateChangedMap(newData, false, false);
222       last.clear();
223       last.putAll(newData);
224    }

226    protected void evaluateClearOnConditions() throws Exception {
227       final Map fieldMetadatas = formMetadata.getFieldMetadatas();
228       final Map toCopy = new HashMap();
229 
230       boolean changed;
231       FieldMetadata fieldMetadata;
232       Map.Entry entry;
233       final Map properties = PropertyUtils.describe(form);
234 
235       do {
236          changed = false;
237          properties.putAll(toCopy);
238 
239          for (final Iterator i = fieldMetadatas.entrySet().iterator(); i
240                .hasNext();) {
241             entry = (Map.Entry) i.next();
242             fieldMetadata = (FieldMetadata) entry.getValue();
243 
244             if (fieldMetadata.getClearOnCondition() == null
245                   || fieldMetadata.getEmptyResolver().isEmpty(
246                         properties.get(entry.getKey()))) {
247                i.remove();
248                continue;
249             }
250 
251             if (isConditionSatisfied(fieldMetadata.getClearOnCondition())) {
252 
253                if (log.isDebugEnabled()) {
254                   log.debug("ClearOn Condition for field '" + entry.getKey()
255                         + "' satisfied. Setting '"
256                         + fieldMetadata.getEmptyValue() + "'");
257                }
258                changed = true;
259                toCopy.put(entry.getKey(), fieldMetadata.getEmptyValue());
260                i.remove();
261             }
262          }
263       } while (changed);
264 
265       if (toCopy.isEmpty()) {
266          if (log.isDebugEnabled()) {
267             log.debug("Nothing changed in ClearOn conditions.");
268          }
269       } else {
270          PropertyUtils.copyProperties(form, toCopy);
271          changedMap.putAll(toCopy);
272       }
273    }

275    protected void evaluateNamedConditions() {
276       Map.Entry entry;
277 
278       for (final Iterator i = formMetadata.getNamedConditions().entrySet()
279             .iterator(); i.hasNext();) {
280          entry = (Map.Entry) i.next();
281 
282          ctx.getVariables().declareVariable(entry.getKey().toString(),
283                isSatisfied((CompiledExpression) entry.getValue()));
284 
285          if (log.isDebugEnabled()) {
286             log.debug("Named Condition '" + entry.getKey() + "' evaluated as '"
287                   + ctx.getVariables().getVariable(entry.getKey().toString())
288                   + "'");
289          }
290       }
291    }

293    protected void evaluateEnabledWhenConditions() {
294       Map.Entry entry;
295       MemberMetadata memberMetadata;
296 
297       final Map toEvaluate = new HashMap(formMetadata.getFieldMetadatas());
298       toEvaluate.putAll(formMetadata.getActionMetadatas());
299 
300       for (final Iterator i = toEvaluate.entrySet().iterator(); i.hasNext();) {
301          entry = (Map.Entry) i.next();
302          memberMetadata = (MemberMetadata) entry.getValue();
303 
304          if (memberMetadata.getEnabledCondition() == null) {
305             continue;
306          }
307 
308          enabledMap.put(memberMetadata.getName(), isSatisfied(memberMetadata
309                .getEnabledCondition()));
310 
311          if (log.isDebugEnabled()) {
312             log.debug("EnabledWhen Condition for member '"
313                   + memberMetadata.getName() + "' evaluated as '"
314                   + enabledMap.get(memberMetadata.getName()) + "'");
315          }
316       }
317    }

345    protected void evaluateCallWhenConditions() {
346       Map.Entry entry;
347       ActionMetadata actionMetadata;
348 
349       for (final Iterator i = formMetadata.getActionMetadatas().entrySet()
350             .iterator(); i.hasNext();) {
351          entry = (Map.Entry) i.next();
352          actionMetadata = (ActionMetadata) entry.getValue();
353 
354          if (actionMetadata.getCallCondition() == null) {
355             continue;
356          }
357 
358          final boolean satisfied = isSatisfied(
359                actionMetadata.getCallCondition()).booleanValue();
360 
361          if (satisfied) {
362             callActions.add(actionMetadata);
363          }
364 
365          if (log.isDebugEnabled()) {
366             log.debug("CallWhen Condition for method '" + entry.getKey()
367                   + "' evaluated as '" + satisfied + "'");
368          }
369       }
370    }

372    protected void evaluateDataProviderTriggers() {
373       Map.Entry entry;
374       DataProviderMetadata dataProviderMetadata;
375 
376       for (final Iterator i = formMetadata.getDataProviderMetadatas()
377             .entrySet().iterator(); i.hasNext();) {
378          entry = (Map.Entry) i.next();
379          dataProviderMetadata = (DataProviderMetadata) entry.getValue();
380 
381          if (dataProviderMetadata.getCallCondition() == null) {
382             continue;
383          }
384 
385          final boolean satisfied = isSatisfied(
386                dataProviderMetadata.getCallCondition()).booleanValue();
387 
388          if (satisfied) {
389             dataProviderActions.add(dataProviderMetadata);
390          }
391 
392          if (log.isDebugEnabled()) {
393             log.debug("CallWhen Condition for method '" + entry.getKey()
394                   + "' evaluated as '" + satisfied + "'");
395          }
396       }

427    public void reset() throws Exception {
428       if (log.isDebugEnabled()) {
429          log.debug("reseting form '" + form + "'");
430          Map.Entry entry;
431          for (Iterator iter = lastSaved.entrySet().iterator(); iter.hasNext();) {
432             entry = (Map.Entry) iter.next();
433             log.debug("Last saved value for '" + entry.getKey() + "' is '"
434                   + entry.getValue() + "'");
435          }
436       }
437 
438       populate(lastSaved, false);
439 
440       cleanUp();
441    }
