89     public void bind() throws Exception {
90        final FormMetadata formMetadata = getFormMetadata(form);
91        controller.setForm(form);
92        controller.setFormMetadata(formMetadata);
93        controller.setup();
94  
95        final Collection dataProviders = new ArrayList(formMetadata.getDataProviderMetadatas().values());
96        // Some actions could be data providers
97        dataProviders.addAll(formMetadata.getActionMetadatas().values());
98        
99        bindFieldMetadatas(formMetadata);
100       bindActionMetadatas(formMetadata);
101       bindDataProviders(dataProviders);
102 
103       updateAndSave(dataProviders, true);
104    }

106    protected void updateAndSave(Collection dataProviders, boolean displayAll) 
107          throws Exception {
108       final Map changedMap = controller.getChangedMap();
109       
110       invokeCallWhenActions();
111       populateDataProviders(controller.getFormMetadata(), dataProviders);
112       controller.update();
113       updateEnabledState();
114       updateVisibleState();
115 
116       controller.save();
117 
118       Map toDisplay = null;
119       if (displayAll) {
120          toDisplay = PropertyUtils.describe(form);
121          toDisplay.keySet().retainAll(boundFieldNames);
122       } else {
123          changedMap.putAll(controller.getChangedMap());
124          toDisplay = changedMap;
125       }
126 
127       if (log.isDebugEnabled()) {
128          log.debug("Displaying: " + toDisplay);
129       }
130 
131       thinlet.displayBean(toDisplay, root);
132    }

134    protected FormMetadata getFormMetadata(final Object form) {
135       if (!(form instanceof FormMetadataFactory)) {
136          throw new IllegalArgumentException(form + " should implement " + 
137                "FormMetadataFactory; probably it should have been annotated " +
138                "with @Form or your aop.xml/weaving process should be " +
139                "properly configured.");
140       }
141 
142       return ((FormMetadataFactory)form).getFormMetadata(form.getClass());
143    }

145    protected void bindFieldMetadatas(final FormMetadata formMetadata) {
146       for (final Iterator i = formMetadata.getFieldMetadatas().entrySet()
147             .iterator(); i.hasNext(); ) {
148          final Map.Entry entry = (Map.Entry)i.next();
149          final String name = entry.getKey().toString();
150          final List components = findComponents(name);
151 
152          if (components.isEmpty()) {
153             log.warn(name + " could not be found while binding " + form.getClass());
154             continue;
155          }
156 
157          final String className = Thinlet.getClass(components.get(0));
158          final FieldMetadata fieldMetadata = (FieldMetadata)entry.getValue();
159 
160          if (Arrays.binarySearch(supportedFieldWidgets, className) < 0) {
161             if (!fieldMetadata.isDisplayOnly()) {
162                log.warn(name + " is not represented by an editable widget (" + 
163                      className + ")");
164             }
165 
166             continue;
167          }
168 
169          if (fieldMetadata.isDisplayOnly()) {
170             if (log.isDebugEnabled()) {
171                log.debug(name + " is a display only field, skipping...");
172             }
173             
174             continue;
175          }
176 
177          if (log.isTraceEnabled()) {
178             log.trace("Binding field " + name + " in " + 
179                   form.getClass().getName() + " to " + className);
180          }
181 
182          boundFieldNames.add(name);
183 
184          for (final Iterator it = components.iterator(); it.hasNext(); ) {
185             final Object component = it.next();
186 
187             if (className.equals(BaseThinlet.CHECKBOX)) {
188                thinlet.setMethod(component, "action", "setValue(" + 
189                      thinlet.getName(component) + "," + 
190                      thinlet.getName(component) + ".group)", root, this);
191             } else if (Arrays.binarySearch(fieldsChangedByAction, className) > -1) {
192                thinlet.setMethod(component, "action", "setValue(" + name + "," +
193                      name + ".name)", root, this);
194             } else {
195                thinlet.setMethod(component, "focuslost", "setValue(" + name + 
196                      "," + name + ".name)", root, this);
197             }
198          }
199       }
200    }

202    private List findComponents(final String name) {
203       final Object widget = thinlet.find(root, name);
204 
205       if (widget != null) {
206          return Collections.singletonList(widget);
207       }
208 
209       final List groupComponents = thinlet.getAllOfClass(root, 
210             BaseThinlet.CHECKBOX);
211 
212       for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
213          final Object component = i.next();
214 
215          if (!name.equals(thinlet.getGroup(component))) {
216             i.remove();
217          }
218       }
219 
220       return groupComponents;
221    }

223    public void setValue(Object component, String name) throws Exception {
224       final String type = Thinlet.getClass(component);
225 
226       String value = null;
227 
228       if (type.equals(BaseThinlet.CHECKBOX)) {
229          if (!thinlet.isSelected(component)) {
230             return;
231          }
232 
233          value = name;
234       } else if (type.equals(BaseThinlet.COMBOBOX) || type.equals(BaseThinlet.LIST)) {
235          component = thinlet.getSelectedItem(component);
236          
237          value = thinlet.getName(component != null ? component : null);
238       } else if (type.equals(BaseThinlet.PROGRESS_BAR) || type.equals(BaseThinlet.SLIDER)) {
239          value = thinlet.getValue(component);
240       } else {
241          value = thinlet.getText(component);
242       }
243 
244       final Map properties = new HashMap();
245       properties.put(name, value);
246 
247       controller.populate(properties);
248 
249       try {
250          updateAndSave(controller.getDataProviderActions(), false);
251       } catch (Exception e) {
252          controller.reset();
253          updateAndSave(controller.getDataProviderActions(), false);
254          throw e;
255       }
256    }

258    protected void bindActionMetadatas(final FormMetadata formMetadata) {
259       for (final Iterator i = formMetadata.getActionMetadatas().entrySet()
260             .iterator(); i.hasNext(); ) {
261          final Map.Entry entry = (Map.Entry)i.next();
262          final String name = ((MethodEntry)entry.getKey()).getMethodName();
263          final Object component = thinlet.find(root, name);
264 
265          if (component == null) {
266             log.warn(name + " could not be found while binding " + form.getClass());
267             continue;
268          }
269 
270          final String className = Thinlet.getClass(component);
271 
272          if (!BaseThinlet.BUTTON.equals(className)) {
273             log.warn(name + " is not represented by a button; it is a " + 
274                   className);
275 
276             continue;
277          }
278 
279          if (log.isTraceEnabled()) {
280             log.trace("Binding action " + name + " in " + form.getClass().getName());
281          }
282 
283          thinlet.setMethod(component, "action", "invokeAction(" + 
284                name + ".name)", root, this);
285       }
286    }

288    public void invokeAction(String name) throws Exception {
289       if (log.isDebugEnabled()) {
290          log.debug("Invoking action: " + name);
291       }
292 
293       final ActionMetadata actionMetadata = ((ActionMetadata)controller
294             .getFormMetadata().getActionMetadatas().get(new MethodEntry(name, 
295             new String[0])));
296 
297       if (actionMetadata.isValidateBefore()) {
298          final String formName = form.getClass().getName();
299 
300          final Map validationErrors = ValidationUtils.getInstance().getMessages(
301                   ValidationUtils.getInstance().validate(form, formName), 
302                   formName);
303 
304          if (!validationErrors.isEmpty()) {
305             throw new ValidationException(validationErrors);
306          }
307       }
308       
309       invokeDataProviderMethod(actionMetadata, true);
310       controller.update();
311       updateAndSave(controller.getDataProviderActions(), false);
312    }

314    protected void bindDataProviders(final Collection dataProviders) {
315       for (final Iterator i = dataProviders.iterator(); i.hasNext();) {
316          final DataProviderMetadata dataProviderMetadata = 
317                (DataProviderMetadata)i.next();
318          if (!dataProviderMetadata.isProvider()) {
319             continue;
320          }
321          
322          final String name = getFieldName(dataProviderMetadata);
323          final Object component = thinlet.find(root, name);
324 
325          if (component == null) {
326             log.warn(name + " could not be found while binding data provider in " + 
327                   form.getClass());
328             continue;
329          }
330 
331          final String className = Thinlet.getClass(component);
332 
333          if (!className.equals(BaseThinlet.TABLE)) {
334             continue;
335          }
336 
337          if (log.isDebugEnabled()) {
338             log.debug("Binding an action method for " + name);
339          }
340 
341          thinlet.setMethod(component, "action", "updateSelection(" + name
342                + ".name," + name + ")", root, this);
343          dataProvidedListByFieldName.put(name, dataProviderMetadata);
344       }
345    }

349             dataProviderMetadata.getObjectField().getFieldName() : 
350             dataProviderMetadata.getIndexField().getFieldName();

353    public void updateSelection(String name, Object table) throws Exception {
354       final int[] selected = thinlet.getSelectedIndexes(table);
355       final List list = (List)dataProvided.get(name);
356       final DataProviderMetadata dataMeta = (DataProviderMetadata)dataProvidedListByFieldName
357             .get(name);
358 
359       dataMeta.populateSelectedFields(form, list, selected);
360       controller.update();
361       updateAndSave(controller.getDataProviderActions(), false);
362    }

364    protected void populateDataProviders(final FormMetadata formMetadata, 
365          final Collection dataProviders) throws Exception {
366       for (final Iterator i = dataProviders.iterator();

375    protected void invokeDataProviderMethod(
376          final DataProviderMetadata dataProviderMetadata, boolean invokeActions) throws Exception {
377       
378       if (!dataProviderMetadata.isProvider()) {
379          if (invokeActions) {
380             final String actionName = dataProviderMetadata.getName();
381             final ThinletMetadata thinletMeta = getThinletMetadata(thinlet);
382             if(thinletMeta.invokePreAction(thinlet, actionName)){
383                dataProviderMetadata.invoke(form);
384                thinletMeta.invokePosAction(thinlet, actionName);
385             }
386          }
387          return;
388       }
389 
390       final String name = getFieldName(dataProviderMetadata);
391       final Object component = thinlet.find(root, name);
392 
393       if (component == null) {
394          log.warn(name + " could not be found while populating using "
395                + form.getClass());
396          return;
397       }
398 
399       if (log.isDebugEnabled()) {
400          log.debug("Populating " + name + " with "
401                + dataProviderMetadata.getMethodEntry().getMethodName());
402       }
403 
404       final Object ret = dataProviderMetadata.invoke(form);
405       final List items = (dataProviderMetadata.getObjectField().isArray()) ? Arrays
406             .asList((Object[]) ret)
407             : (List) ret;
408       dataProvided.put(name, items);
409       final String className = Thinlet.getClass(component);
410 
411       if (className.equals(BaseThinlet.TABLE)) {
412          dataProviderMetadata.resetSelectedFields(form);
413          thinlet.populateFromCollection(component, items);
414       } else if (className.equals(BaseThinlet.COMBOBOX)) {
415          final String key = (String) thinlet.getProperty(component, "key");
416 
417          if (key == null) {
418             throw new PropertyMisconfigurationException("Property 'key' "
419                   + "must be configured for the widget named " + name);
420          }
421 
422          final String value = (String) thinlet.getProperty(component, "value");
423          final Object blankObject = thinlet.getProperty(component, "blank");
424          boolean blank;
425 
426          if (blankObject == null) {
427             blank = false;
428          } else if (blankObject instanceof String) {
429             blank = Boolean.valueOf(blankObject.toString()).booleanValue();
430          } else if (blankObject instanceof Boolean) {
431             blank = ((Boolean) blankObject).booleanValue();
432          } else {
433             throw new PropertyMisconfigurationException("Property 'blank' "
434                   + "for the widget named " + name + " must either be left "
435                   + "empty or contain a boolean value");
436          }
437 
438          thinlet.populateFromCollection(component, items, key, value, blank);
439       } else {
440          // TODO: only table and combobox are implemented
441          throw new UnsupportedOperationException(className + " is not "
442                + "supported for data providing");
443       }
444    }

446    protected void invokeCallWhenActions() throws Exception {
447       for (final Iterator i = controller.getCallActions().iterator(); 
448             i.hasNext(); ) {
449          final ActionMetadata actionMetadata = (ActionMetadata)i.next();
450          
451          if (log.isDebugEnabled()) {
452             log.debug("Calling " + actionMetadata.getMethodEntry().getMethodName() +
453                   " automatically");
454          }
455          
456          invokeDataProviderMethod(actionMetadata, true);
457       }
458    }

460    protected void updateEnabledState() {
461       final Collection enabledStateEntries = enabledStateMap.entrySet();
462 
463       for (final Iterator i = controller.getEnabledMap().entrySet().iterator();
464             i.hasNext(); ) {
465          final Map.Entry entry = (Map.Entry)i.next();
466 
467          if (!enabledStateEntries.contains(entry)) {
468             if (log.isDebugEnabled()) {
469                log.debug("Changing " + entry.getKey() + " enabled state to " + 
470                      entry.getValue());
471             }
472 
473             thinlet.setEnabled(thinlet.find(root, entry.getKey().toString()), 
474                   ((Boolean)entry.getValue()).booleanValue());
475          }
476       }
477 
478       enabledStateMap.clear();
479       enabledStateMap.putAll(controller.getEnabledMap());
480    }

485       for (final Iterator i = controller.getVisibleMap().entrySet().iterator();
486             i.hasNext(); ) {
487          final Map.Entry entry = (Map.Entry)i.next();

489          if (!visibleStateEntries.contains(entry)) {
490             if (log.isDebugEnabled()) {

492                      entry.getValue());
493             }
494

496                   ((Boolean)entry.getValue()).booleanValue());
497          }
498       }
499 
500       visibleStateMap.clear();
501       visibleStateMap.putAll(controller.getVisibleMap());
502    }
