115    public void bind() throws Exception {
116       final FormMetadata formMetadata = getFormMetadata(form);
117       final Collection dataProviders = new ArrayList(formMetadata
118             .getDataProviderMetadatas().values());
119 
120       bindFieldMetadatas(formMetadata);
121       bindActionMetadatas(formMetadata);
122       bindDataProviders(dataProviders);
123 
124       setupController();
125    }

127    protected void setupController() throws Exception {
128       if (!controller.isSetup()) {
129          controller.addFormControllerListener(this);
130          controller.setup();
131       } else {
132          controller.fireAllEvents(this);
133          controller.addFormControllerListener(this);
134       }
135    }

146    protected void bindFieldMetadata(String name, FieldMetadata fieldMetadata) {
147       final List components = findComponents(name);
148 
149       if (components.isEmpty()) {
150          log.warn(name + " could not be found while binding " + form.getClass());
151          return;
152       }
153 
154       final Object mainComponent = components.get(0);
155       final String className = Thinlet.getClass(mainComponent);
156 
157       if (Arrays.binarySearch(supportedFieldWidgets, className) < 0) {
158          if (fieldMetadata.isWriteable()) {
159             log.warn(name + " is not represented by an editable widget (" + 
160                   className + ")");
161          }
162 
163          return;
164       }
165 
166       createWidgetGroup(mainComponent, name);
167 
168       if (!fieldMetadata.isWriteable()) {
169          if (log.isDebugEnabled()) {
170             log.debug(name + " is a display only field, skipping...");
171          }
172          
173          return;
174       }
175 
176       if (log.isTraceEnabled()) {
177          log.trace("Binding field " + name + " in " + 
178                form.getClass().getName() + " to " + className);
179       }
180 
181       boundFieldNames.add(name);
182 
183       for (final Iterator it = components.iterator(); it.hasNext(); ) {
184          bindComponent(name, className, it.next());
185       }
186    }

188    protected void bindComponent(String name, String className, Object component) {
189       if (className.equals(BaseThinlet.CHECKBOX) ||
190             className.equals(BaseThinlet.TOGGLE_BUTTON)) {
191          thinlet.setMethod(component, "action", "setValue(" + 
192                thinlet.getName(component) + "," + 
193                thinlet.getName(component) + 
194                (thinlet.getGroup(component) == null ? ".name)" : ".group)"),
195                root, this);
196       } else if (Arrays.binarySearch(fieldsChangedByAction, className) > -1) {
197          thinlet.setMethod(component, "action", "setValue(" + name + "," +
198                name + ".name)", root, this);
199       } else {
200          thinlet.setMethod(component, "focuslost", "setValue(" + name + 
201                "," + name + ".name)", root, this);
202       }
203    }

205    protected List findComponents(final String name) {
206       final Object widget = thinlet.find(root, name);
207 
208       if (widget != null) {
209          return Collections.singletonList(widget);
210       }
211 
212       if (groupComponents == null) {
213          groupComponents = thinlet.getAllOfClass(root, new String[] {
214                BaseThinlet.CHECKBOX, BaseThinlet.TOGGLE_BUTTON }, 
215                componentSearchDepth);
216       }
217 
218       final List components = new ArrayList();
219 
220       for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
221          final Object component = i.next();
222 
223          if (name.equals(thinlet.getGroup(component))) {
224             components.add(component);
225          }
226       }
227 
228       return components;
229    }

239    protected void createWidgetGroup(Object component, String name) {
240       final Collection enabledWidgetGroupCollection = new ArrayList();
241       final Collection visibleWidgetGroupCollection = new ArrayList();
242 
243       enabledWidgetGroupCollection.add(name);
244       visibleWidgetGroupCollection.add(name);
245 
246       final String widgetGroup = (String)thinlet.getProperty(component,
247             "widgetGroup");
248       final String enabledWidgetGroup = (String)thinlet.getProperty(component,
249             "enabledWidgetGroup");
250       final String visibleWidgetGroup = (String)thinlet.getProperty(component,
251             "visibleWidgetGroup");
252 
253       if (widgetGroup != null) {
254          final List widgets = Arrays.asList(widgetGroup.split(","));
255          enabledWidgetGroupCollection.addAll(widgets);
256          visibleWidgetGroupCollection.addAll(widgets);
257 
258          if (log.isDebugEnabled()) {
259             log.debug("Adding " + widgetGroup + " as a widget group for " +
260                   name + " in class " + form.getClass());
261          }
262       }
263       if (enabledWidgetGroup != null) {
264          enabledWidgetGroupCollection.addAll(Arrays.asList(enabledWidgetGroup.split(",")));
265 
266          if (log.isDebugEnabled()) {
267             log.debug("Adding " + enabledWidgetGroup + " as a enabled widget group for " +
268                   name + " in class " + form.getClass());
269          }
270       }
271       if (visibleWidgetGroup != null) {
272          visibleWidgetGroupCollection.addAll(Arrays.asList(visibleWidgetGroup.split(",")));
273 
274          if (log.isDebugEnabled()) {
275             log.debug("Adding " + visibleWidgetGroup + " as a visible widget group for " +
276                   name + " in class " + form.getClass());
277          }
278       }
279 
280       enabledWidgetGroupMap.put(name, enabledWidgetGroupCollection);
281       visibleWidgetGroupMap.put(name, visibleWidgetGroupCollection);
282    }

284    public void setValue(Object component, String name) throws Exception {
285       final String type = Thinlet.getClass(component);
286 
287       String value = null;
288 
289       if (type.equals(BaseThinlet.CHECKBOX) || type.equals(BaseThinlet.TOGGLE_BUTTON)) {
290          if (thinlet.getGroup(component) != null) {
291             if (!thinlet.isSelected(component)) {
292                return;
293             }
294          
295             value = name;
296          } else {
297             value = String.valueOf(thinlet.isSelected(component));
298          }
299       } else if (type.equals(BaseThinlet.COMBOBOX) || type.equals(BaseThinlet.LIST)) {
300          component = thinlet.getSelectedItem(component);
301          
302          value = component != null ? thinlet.getName(component) : null;
303       } else if (type.equals(BaseThinlet.PROGRESS_BAR) || type.equals(BaseThinlet.SLIDER)) {
304          value = String.valueOf(thinlet.getValue(component));
305       } else {
306          value = thinlet.getText(component);
307       }
308 
309       final Map properties = new HashMap();
310       properties.put(name, value);
311 
312       controller.populate(properties, converters);
313    }

323    protected void bindActionMetadata(String name) {
324       final Object component = thinlet.find(root, name);
325 
326       if (component == null) {
327          log.warn(name + " could not be found while binding " + form.getClass());
328          return;
329       }
330 
331       final String className = Thinlet.getClass(component);
332 
333       if (!BaseThinlet.BUTTON.equals(className)) {
334          log.warn(name + " is not represented by a button; it is a " + 
335                className);
336 
337          return;
338       }
339 
340       createWidgetGroup(component, name);
341       
342       if (log.isTraceEnabled()) {
343          log.trace("Binding action " + name + " in " + form.getClass().getName());
344       }
345 
346       thinlet.setMethod(component, "action", "invokeAction(" + 
347             name + ".name)", root, this);
348    }

356    protected void bindDataProvider(DataProviderMetadata dataProviderMetadata) {
357       final String name = dataProviderMetadata.getWidgetName();
358       final Object component = thinlet.find(root, name);
359 
360       if (component == null) {
361          log.warn(name + " could not be found while binding data provider in " + 
362                form.getClass());
363          return;
364       }
365 
366       final String className = Thinlet.getClass(component);
367 
368       if (!(className.equals(BaseThinlet.TABLE) || 
369             className.equals(BaseThinlet.COMBOBOX) ||
370             className.equals(BaseThinlet.LIST))) {
371          log.warn(className + " is not a supported widget for binding a " +
372                "data provider");
373          return;
374       }
375 
376       createWidgetGroup(component, name);
377 
378       if (log.isDebugEnabled()) {
379          log.debug("Binding an action method for " + name);
380       }
381 
382       thinlet.setMethod(component, "action", "updateSelection(" + name
383             + ".name," + name + ")", root, this);
384       dataProvided.put(name, dataProviderMetadata);
385    }

387    public void updateSelection(String name, Object widget) throws Exception {
388       int[] selected = thinlet.getSelectedIndexes(widget);
389       
390       final String widgetClass = BaseThinlet.getClass(widget);
391 
392       if (selected.length >= 1 && (BaseThinlet.COMBOBOX.equals(widgetClass) ||
393             BaseThinlet.LIST.equals(widgetClass)) && isBlank(widget, name)) {
394          boolean remove = false;
395 
396          for (int i = 0; i < selected.length; i++) {
397             selected[i] = selected[i] - 1;
398 
399             if (selected[i] == -1) {
400                remove = true;
401             }
402          }
403 
404          if (remove) {
405             if (selected.length == 1) {
406                selected = new int[0];
407             } else {
408                final int[] aux = selected;
409                selected = new int[aux.length - 1];
410                System.arraycopy(aux, 1, selected, 0, aux.length - 1);
411             }
412          }
413       }
414 
415       final DataProviderMetadata dataMeta = (DataProviderMetadata)dataProvided
416             .get(name);
417 
418       controller.updateSelection(dataMeta, selected);
419    }

421    public void invokeAction(String name) throws Exception {
422       Map stringProperties = null;
423 
424       if (name != null) {
425          final MethodMetadata methodMetadata = getMethodMetadata(name);
426 
427          if (methodMetadata != null && 
428                methodMetadata.getActionMetadata() != null && 
429                methodMetadata.getActionMetadata().isValidateBefore()) {
430             stringProperties = getStringProperties();
431             thinlet.populate(null, root, stringProperties, false);
432          }
433       }
434 
435       controller.invokeAction(name, stringProperties);
436    }

475    public void dataProvidedListChanged(DataProviderMetadata metadata, List items)
476          throws Exception {
477       final String name = metadata.getWidgetName();
478       final Object component = thinlet.find(root, name);
479 
480       if (component == null) {
481          log.warn(name + " could not be found while populating using "
482                + form.getClass());
483          return;
484       }
485 
486       if (log.isDebugEnabled()) {
487          log.debug("Populating " + name + " with " + metadata.getName());
488       }
489 
490       final String className = Thinlet.getClass(component);
491 
492       if (className.equals(BaseThinlet.TABLE)) {
493          if (!controller.isResetOnDataProviderChange()) {
494             resetSelectedFields(metadata);
495          }
496 
497          thinlet.populateFromCollection(component, items, formatters);
498       } else if (className.equals(BaseThinlet.COMBOBOX) || 
499                className.equals(BaseThinlet.LIST)) {
500          //TODO: This parsing shouldn't occur every time
501          final String key = (String) thinlet.getProperty(component, "key");
502 
503          if (key == null) {
504             throw new PropertyMisconfigurationException("Property 'key' "
505                   + "must be configured for the widget named " + name);
506          }
507 
508          final String value = (String)thinlet.getProperty(component, "value");
509          final boolean virtual = isVirtual(component, name);
510          final boolean blank = isBlank(component, name);
511          final String blankLabel = (String)thinlet.getProperty(component, 
512                "blankLabel");
513 
514          thinlet.populateFromCollection(component, items, key, value, virtual, 
515                blank, blankLabel, formatters);
516       } else {
517          throw new UnsupportedOperationException(className + " is not "
518                + "supported for data providing");
519       }
520    }

539    private boolean isBoolean(Object component, String name, String propertyName) {
540       final Object booleanObject = thinlet.getProperty(component, propertyName);
541 
542       boolean ret;
543 
544       if (booleanObject == null) {
545          ret = false;
546       } else if (booleanObject instanceof String) {
547          ret = Boolean.valueOf(booleanObject.toString()).booleanValue();
548       } else if (booleanObject instanceof Boolean) {
549          ret = ((Boolean) booleanObject).booleanValue();
550       } else {
551          throw new PropertyMisconfigurationException("Property '" + 
552                propertyName + "' " + "for the widget named " + name + " must " +
553                "either be left empty or contain a boolean value");
554       }
555 
556       return ret;
557    }

567    protected void conditionsChanged(Map updatedConditions, boolean enabled) {
568       for (final Iterator i = updatedConditions.entrySet().iterator();
569             i.hasNext(); ) {
570          final Map.Entry entry = (Map.Entry)i.next();
571 
572          final Collection widgetGroup = getWidgetGroupCollection(
573                entry.getKey().toString(), enabled);
574 
575          if (widgetGroup == null) {
576             log.warn(entry.getKey() + (enabled ? " enabled" : " visible")
577                   + " state should be changed, "
578                   + "but could not be found in the view component for "
579                   + form.getClass());
580             continue;
581          }
582 
583          if (log.isDebugEnabled()) {
584             log.debug("Changing " + entry.getKey()
585                   + (enabled ? " enabled" : " visible") + " state to "
586                   + entry.getValue());
587          }
588 
589          for (Iterator it = widgetGroup.iterator(); it.hasNext(); ) {
590             final Object widget = thinlet.find(root, it.next().toString());
591 
592             if (widget == null) {
593                log.warn(entry.getKey() + (enabled ? " enabled" : " visible")
594                      + " state should be changed, "
595                      + "but could not be found in the view component for "
596                      + form.getClass());
597                continue;
598             }
599 
600             if (enabled) {
601                thinlet.setEnabled(widget, ((Boolean)entry.getValue())
602                      .booleanValue());
603             } else {
604                thinlet.setVisible(widget, ((Boolean)entry.getValue())
605                      .booleanValue());
606             }
607          }
608       }
609    }
