165                                                 throws ScreenNotFoundException {
166       try {
167          InputStream is = getResourceAsStream(fileName, 
168                                              handler.getClass().getClassLoader());
169 
170          if (is == null) {
171             throw new ScreenNotFoundException(fileName);
172          }

274    protected void getAllOfClass(List all, Object component, String[] classNames, 
275          int currentDepthLevel, int maximumDepthLevel) {
276       Object[] components = getItems(component);
277       boolean match;
278 
279       for (int i = 0; i < components.length; i++) {
280          match = false;
281 
282          for (int j = 0; !match && j < classNames.length; j++) {
283             if (getClass(components[i]).equals(classNames[j])) {
284                all.add(components[i]);
285                match = true;
286             }
287          }
288 
289          if (!match && currentDepthLevel < maximumDepthLevel) {
290             getAllOfClass(all, components[i], classNames, currentDepthLevel + 1,
291                   maximumDepthLevel);
292          }
293       }
294    }

451    protected void displayBean(Object bean, Object root, Map formatters) 
452          throws IllegalAccessException, InvocationTargetException, 
453                 NoSuchMethodException {
454       final Map properties = bean instanceof Map ? (Map)bean : 
455             PropertyUtils.describe(bean);
456 
457       if (root instanceof Thinlet) {
458          root = ((Thinlet)root).getDesktop();
459       }
460 
461       String propertyName;
462       String propertyValue = null;
463       String type;
464       Object component;
465       Collection groupComponents = null;
466 
467       for (final Iterator propertyNames = properties.keySet().iterator();
468                                                    propertyNames.hasNext(); ) {
469          propertyName = propertyNames.next().toString();
470          propertyValue = null;
471          component = find(root, propertyName);
472 
473          if (component == null) {
474             if (groupComponents == null) {
475                groupComponents = getAllOfClass(root, CHECKBOX);
476             }
477 
478             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
479                component = i.next();
480 
481                if (propertyName.equals(getGroup(component)) && 
482                       getName(component) != null ) {
483                   if (propertyValue == null) {
484                      propertyValue = getPropertyValue(properties, propertyName, 
485                            true, formatters);
486                   }
487 
488                   if (getName(component).equals(propertyValue)) {
489                      setSelected(component, true);
490                      break;
491                   }
492                }
493             }
494 
495             continue;
496          }
497 
498          type = getClass(component);
499 
500          if (type.equals(COMBOBOX) || (type.equals(LIST) && getSelection(component).equals(SINGLE))) {
501             Object selectedComponent = getSelectedItem(component);
502 
503             if (selectedComponent != null && type.equals(LIST)) {
504                setSelected(selectedComponent, false);
505             }
506 
507             if (type.equals(COMBOBOX)) {
508                setSelected(component, -1);
509                setText(component, "");
510             }
511 
512             String keyPropertyName = (String)getProperty(component, "key");
513 
514             if (keyPropertyName != null) {
515                Object o = properties.get(propertyName);
516                o = o == null ? null : PropertyUtils.getProperty(
517                      properties.get(propertyName), keyPropertyName);
518 
519                propertyValue = format(formatters, keyPropertyName, o);
520             } else {
521                propertyValue = getPropertyValue(properties, propertyName, true, 
522                      formatters);
523             }
524 
525             selectedComponent = find(component, propertyValue);
526 
527             if (selectedComponent != null) {
528                if (type.equals(LIST)) {
529                   setSelected(selectedComponent, true);
530                } else {
531                   setSelected(component, getIndexOf(component, selectedComponent));
532                }
533             }
534          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
535             setValue(component, Integer.parseInt(getPropertyValue(properties, propertyName, 
536                   false, formatters)));
537          } else if (type.equals(PANEL)) {
538             displayBean(properties.get(propertyName), component);
539          } else if (type.equals(CHECKBOX)) {
540             setSelected(component, Boolean.TRUE.equals(properties.get(propertyName)));
541          } else if (!type.equals(TABLE) && !type.equals(LIST)){
542             setText(component, getPropertyValue(properties, propertyName, 
543                   false, formatters));
544          }
545       }
546    }

548    protected String getPropertyValue(Map properties, String propertyName,
549          boolean enumKey, Map formatters) {
550       Object o = properties.get(propertyName);
551 
552       if (enumKey && o instanceof Enum) {
553         return o.toString();
554       }
555 
556       return format(formatters, propertyName, o);
557    }

594    protected void populate(Object bean, Object root, final Map properties, 
595          final boolean changeBean) throws IllegalAccessException, 
596          InvocationTargetException, NoSuchMethodException {
597       if (changeBean) {
598          properties.putAll(BeanUtils.describe(bean));
599       }
600 
601       if (root instanceof Thinlet) {
602          root = ((Thinlet)root).getDesktop();
603       }
604 
605       String propertyName;
606       String type;
607       Object component;
608       Collection groupComponents = null;
609 
610       for (final Iterator propertyNames = properties.keySet().iterator();
611                                                    propertyNames.hasNext(); ) {
612          propertyName = propertyNames.next().toString();
613          component = find(root, propertyName);
614 
615          if (component == null) {
616             if (groupComponents == null) {
617                groupComponents = getAllOfClass(root, CHECKBOX);
618             }
619 
620             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
621                component = i.next();
622 
623                if (propertyName.equals(getGroup(component)) && 
624                      isSelected(component)) {
625                   properties.put(propertyName, getName(component));
626                   break;
627                }
628             }
629 
630             continue;
631          }
632 
633          type = getClass(component);
634 
635          if (type.equals(COMBOBOX) || type.equals(LIST)) {
636             component = getSelectedItem(component);
637 
638             if (component != null) {
639                properties.put(propertyName, getName(component));
640             } else {
641                properties.put(propertyName, null);
642             }
643          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
644             properties.put(propertyName, String.valueOf(getValue(component)));
645          } else if (type.equals(TABLE)) {
646             // skip it intentionally
647          } else {
648             properties.put(propertyName, getText(component));
649          }
650       }
651 
652       if (changeBean) {
653          BeanUtils.populate(bean, properties);
654 
655          if (bean instanceof Form) {
656             ((Form)bean).afterPopulate();
657          }
658       }
659    }

711    protected void populateFromEnum(Object component, Class clazz, boolean blank) {
712       final boolean combobox = getClass(component).equals(COMBOBOX);
713 
714       if (!combobox && !getClass(component).equals(LIST)) {
715          throw new UnsupportedOperationException();
716       }
717 
718       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
719 
720       String key;
721       Object enumInstance;
722 
723       removeAll(component);
724 
725       if (blank) {
726          add(component, createItemOfType("", "", null, type));
727       }
728 
729       for (final Iterator i = Enum.getInstances(clazz).iterator(); i.hasNext(); ) {
730          enumInstance = i.next();
731          key = enumInstance.toString();
732 
733          add(component, createItemOfType(key, FormatterRegistry.getInstance()
734                .format(enumInstance), enumInstance, type));
735       }
736    }

866    protected void populateFromCollection(Object component, Collection c,
867          Map formatters, Map widgetFactories) throws IllegalAccessException,
868          InvocationTargetException, NoSuchMethodException {
869       if (!getClass(component).equals(TABLE)) {
870          throw new UnsupportedOperationException();
871       }
872 
873       removeAll(component);
874 
875       final String virtualPrefix = VIRTUAL + ':';
876       final Collection propertyNames = new ArrayList();
877       final Collection virtualPropertyNames = new ArrayList();
878 
879       final String componentName = getName(component);
880       final Object[] columns = getTableColumns(component);
881       
882       for (int i = 0; i < columns.length; i++) {
883          String name = getName(columns[i]);
884 
885          if (name == null) {
886             throw new IllegalArgumentException("column (index " + i + "; " +
887                   "text: " + getText(columns[i]) +") in " + 
888                   "table " + componentName + " does not have a name");
889          }
890 
891          if (name.startsWith(virtualPrefix)) {
892             name = name.substring(virtualPrefix.length());
893             virtualPropertyNames.add(name);
894          }
895 
896          propertyNames.add(name);
897       }
898 
899 
900       String propertyName;
901       Object row;
902       Object bean;
903       int indexOfDot;
904       boolean skip;
905 
906       for (final Iterator i = c.iterator(); i.hasNext(); ) {
907          bean = i.next();
908          row = createRow();
909 
910          for (final Iterator it = propertyNames.iterator(); it.hasNext(); ) {
911             propertyName = it.next().toString();
912 
913             if (virtualPropertyNames.contains(propertyName)) {
914                Object cell = createItemOfType(propertyName,
915                      getVirtualFormatter(formatters, componentName,
916                      propertyName).format(bean), bean, ItemType.CELL,
917                      getWidgetFactory(widgetFactories, componentName + '.' + 
918                      propertyName));
919                add(row, cell);
920                continue;
921             }
922 
923             indexOfDot = 0;
924             skip = false;
925 
926             while ((indexOfDot = propertyName.indexOf('.', indexOfDot)) != -1) {
927                if (PropertyUtils.getProperty(bean, 
928                      propertyName.substring(0, indexOfDot)) == null) {
929                   Object cell = createItemOfType(propertyName, format(
930                         formatters, componentName + '.' + propertyName, null),
931                         null, ItemType.CELL, getWidgetFactory(widgetFactories, 
932                         componentName + '.' + propertyName));
933 
934                   add(row, cell);
935                   skip = true;
936                   break;
937                }
938 
939                indexOfDot += 1;
940             }
941 
942             if (skip) {
943                continue;
944             }
945 
946             Object value = PropertyUtils.getProperty(bean, propertyName);
947             Object cell = createItemOfType(propertyName, format(formatters,
948                   componentName + '.' + propertyName, value), value,
949                   ItemType.CELL, getWidgetFactory(widgetFactories, 
950                   componentName + '.' + propertyName));

952             add(row, cell);
