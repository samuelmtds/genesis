101       public class DocumentBindingStrategy extends JTextComponentBindingStrategy {
102          private DocumentListener listener;
103          private boolean processing;
104 
105          public void addListener(JTextComponent component) {
106             component.getDocument().addDocumentListener(
107                   listener = createDocumentListener());
108          }
109 
110          protected DocumentListener createDocumentListener() {
111             return new DocumentListener() {
112                public void changedUpdate(DocumentEvent e) {
113                   update();
114                }
115 
116                public void insertUpdate(DocumentEvent e) {
117                   update();
118                }
119 
120                public void removeUpdate(DocumentEvent e) {
121                   update();
122                }
123 
124                private void update() {
125                   processing = true;
126 
127                   try {
128                      updateForm();
129                   } finally {
130                      processing = false;
131                   }
132                }
133             };
134          }
135 
136          public void setValue(final Object value) {
137             final Runnable updateValue = new Runnable() {
138                public void run() {
139                   updateComponentValue(value);
140                }
141             };
142 
143             if (!processing) {
144                updateValue.run();
145             } else {
146                if (!fieldMetadata.getEqualityComparator().equals(
147                      formatValue(value), getValue())) {
148                   EventQueue.invokeLater(updateValue);
149                }
150             }
151          }
152 
153          public void removeListener(JTextComponent component) {
154             if (listener != null) {
155                component.getDocument().removeDocumentListener(listener);
156             }
157          }
158       }

183          return AbstractBinder.BINDING_STRATEGY_AS_YOU_TYPE.equals(
184                getBinder().getBindingStrategy((String)component
