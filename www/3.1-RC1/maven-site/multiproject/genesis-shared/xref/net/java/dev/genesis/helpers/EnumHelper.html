19  package net.java.dev.genesis.helpers;

21  import java.lang.reflect.InvocationTargetException;
22  import java.util.Arrays;
23  import java.util.Collection;
24  
25  import net.java.dev.genesis.reflection.ReflectionInvoker;
26  
27  
28  public final class EnumHelper {
29     private static final EnumHelper instance = new EnumHelper();
30  
31     private Class enumClass;
32  
33     private EnumHelper() {
34        try {
35           enumClass = Class.forName("java.lang.Enum");
36        } catch (Exception e) {
37           // not JDK 5 or later
38        }
39     }
40  
41     private Object invoke(Object target, String methodName) {
42        try {
43           return ReflectionInvoker.getInstance().invoke(target, methodName);
44        } catch (InvocationTargetException e) {
45           throw new RuntimeException(e.getTargetException());
46        } catch (Exception e) {
47           throw new RuntimeException(e);
48        }
49     }
50  
51     public static EnumHelper getInstance() {
52        return instance;
53     }
54  
55     public Class getEnumClass() {
56        return enumClass;
57     }
58  
59     public boolean supportsEnum() {
60        return enumClass != null;
61     }
62     
63     public boolean isReusableComponentEnum(Class clazz) {
64        return net.java.dev.reusablecomponents.lang.Enum.class.isAssignableFrom(clazz);
65     }
66     
67     public boolean isJava5Enum(Class clazz) {
68        return supportsEnum() && Boolean.TRUE.equals(invoke(clazz, "isEnum"));
69     }
70     
71     public boolean isReusableComponentEnum(Object object) {
72        return object instanceof net.java.dev.reusablecomponents.lang.Enum;
73     }
74     
75     public boolean isJava5Enum(Object object) {
76        return object != null && isJava5Enum(object.getClass());
77     }
78     
79     public boolean isEnum(Object object) {
80        return isReusableComponentEnum(object) || isJava5Enum(object);
81     }
82     
83     public Collection values(Class clazz) {
84        if (isReusableComponentEnum(clazz)) {
85           return net.java.dev.reusablecomponents.lang.Enum.getInstances(clazz);
86        } else if (isJava5Enum(clazz)) {
87           return Arrays.asList((Object[])invoke(clazz, "getEnumConstants"));
88        }
89  
90        throw new IllegalArgumentException(clazz.getName() + " is not an enum");
91     }
92     
93     public Object valueOf(Class clazz, String name) {
94        if (isReusableComponentEnum(clazz)) {
95           return net.java.dev.reusablecomponents.lang.Enum.get(clazz, name);
96        } else if (isJava5Enum(clazz)) {
97           Object[] enums = (Object[]) invoke(clazz, "getEnumConstants");
98           for (int i = 0; i < enums.length; i++) {
99              String enumName = getName(enums[i]);
100             
101             if (name.equals(enumName)) {
102                return enums[i];
103             }
104          }
105 
106          return null;
107       }
108 
109       throw new IllegalArgumentException(clazz.getName() + " is not an enum");
110    }
111    
112    public Class getDeclaringClass(Object object) {
113       if (isReusableComponentEnum(object)) {
114          return ((net.java.dev.reusablecomponents.lang.Enum)object).getBaseClass();
115       } else if (isJava5Enum(object)) {
116          return (Class) invoke(object, "getDeclaringClass");
117       }
118 
119       throw new IllegalArgumentException(object + " is not an enum");
120    }
121    
122    public String getName(Object object) {
123       if (isReusableComponentEnum(object)) {
124          return ((net.java.dev.reusablecomponents.lang.Enum)object).getName();
125       } else if (isJava5Enum(object)) {
126          return object.toString();
127       }
128 
129       throw new IllegalArgumentException(object + " is not an enum");
130    }
