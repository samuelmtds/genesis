33  public class JLabelComponentBinderTest extends GenesisTestCase {
34     private JLabel label;
35     private MockSwingBinder binder;
36     private WidgetBinder componentBinder;
37     private BoundField boundField;
38     private MockForm form;
39     private FieldMetadata fieldMeta;
40  
41     public JLabelComponentBinderTest() {
42        super("JLabel Component Binder Unit Test");
43     }
44  
45     protected void setUp() throws Exception {
46        label = new JLabel();
47        binder = new MockSwingBinder(new JPanel(), form = new MockForm());
48        componentBinder = binder.getWidgetBinder(label);
49        fieldMeta = form.getFormMetadata().getFieldMetadata("stringField");
50     }
51  
52     public void testSetValue() throws Exception {
53        assertTrue(componentBinder instanceof JLabelComponentBinder);
54  
55        assertNull(componentBinder.bind(binder, label,
56              (DataProviderMetadata) null));
57        assertNull(componentBinder.bind(binder, label, (ActionMetadata) null));
58        assertNotNull(boundField = componentBinder.bind(binder, label, fieldMeta));
59  
60        boundField.setValue("someValue");
61        assertEquals("someValue", label.getText());
62  
63        boundField.setValue(null);
64        assertEquals("", label.getText());
65     }
