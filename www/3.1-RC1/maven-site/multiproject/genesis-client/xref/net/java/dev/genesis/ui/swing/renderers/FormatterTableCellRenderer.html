19  package net.java.dev.genesis.ui.swing.renderers;
20  
21  import java.awt.Component;
22  
23  import javax.swing.JTable;
24  import javax.swing.table.DefaultTableCellRenderer;
25  import javax.swing.table.TableColumn;
26  
27  import net.java.dev.genesis.text.FormatterRegistry;
28  import net.java.dev.genesis.ui.swing.SwingBinder;
29  
30  public class FormatterTableCellRenderer extends DefaultTableCellRenderer {
31     protected SwingBinder getBinder(JTable table) {
32        return (SwingBinder) table.getClientProperty(SwingBinder.BINDER_KEY);
33     }
34  
35     protected String getColumnIdentifier(JTable table, int index) {
36        TableColumn column = table.getColumnModel().getColumn(index);
37        String[] names = (String[]) table.getClientProperty(SwingBinder.COLUMN_NAMES);
38  
39        if (names != null && names.length > column.getModelIndex()) {
40           return names[column.getModelIndex()];
41        }
42  
43        return (String) column.getIdentifier();
44     }
45  
46     protected Object format(JTable table, Object value, int column) {
47        final SwingBinder binder = getBinder(table);
48        if (binder == null) {
49           return format(value);
50        }
51  
52        final String property = getColumnIdentifier(table, column);
53        boolean isVirtual = binder.isVirtual(table, property);
54  
55        return binder.format(binder.getName(table), property, value, isVirtual);
56     }
57  
58     protected Object format(Object value) {
59        return FormatterRegistry.getInstance().format(value);
60     }
61  
62     public Component getTableCellRendererComponent(JTable table, Object value,
63           boolean isSelected, boolean hasFocus, int row, int column) {
64        return super.getTableCellRendererComponent(table, format(table, value,
65              column), isSelected, hasFocus, row, column);
66     }
67  }
