35  public class WrapFactory extends org.mozilla.javascript.WrapFactory {
36     private static Method[] methods = ScriptFunctionsAdapter.class.getMethods();
37     private static Map javaMethods = new HashMap(methods.length);

43     public Scriptable wrapAsJavaObject(Context cx, Scriptable scope,
44           Object javaObject, Class staticType) {
45        if (javaObject instanceof ScriptFunctionsAdapter) {
46           return new NativeJavaObject();
47        }

49        return super.wrapAsJavaObject(cx, scope, javaObject, staticType);

52     public Object wrap(Context cx, Scriptable scope, Object obj, Class staticType) {
53        if ((obj == null) || (obj == Undefined.instance)
54              || obj instanceof Scriptable) {
55           return obj;
56        }
57  
58        if ((staticType != null) && staticType.isPrimitive()) {
59           if (staticType == Void.TYPE) {
60              return Undefined.instance;
61           }
62  
63           return obj;
64        }
65  
66        if (!isJavaPrimitiveWrap()) {
67           if (obj instanceof String || obj instanceof Number
68                 || obj instanceof Boolean || obj instanceof Character) {
69              return obj;
70           }
71        }
72  
73        Class cls = obj.getClass();
74  
75        if (cls.isArray()) {
76           return NativeJavaArray.wrap(scope, obj);
77        }
78  
79        return wrapAsJavaObject(cx, scope, obj, staticType);
80     }

88     protected class NativeJavaMethod extends
89           org.mozilla.javascript.NativeJavaMethod {
90        private final Method m;
91  
92        public NativeJavaMethod(Method method, String name) {
93           super(method, name);
94           this.m = method;
95        }
96  
97        /***
98         * <p>
99         * Alternative call method that do not coerce Java Objects (e.g.:
100        * java.lang.Integer to java.lang.Double).
101        * </p>
102        * 
103        * <p>
104        * Only Javascript Objects are coerced
105        * </p>
106        */
107       public Object call(Context cx, Scriptable scope, Scriptable thisObj,
108             Object[] args) {
109          try {
110             Class[] parameterTypes = m.getParameterTypes();
111 
112             if (parameterTypes.length != args.length) {
113                methodNofFound(args);
114             }
115 
116             for (int i = 0; i < args.length; i++) {
117                if (args[i] == null) {
118                   continue;
119                }
120 
121                if ((args[i] == Undefined.instance)
122                      || args[i] instanceof Scriptable
123                      || (parameterTypes[i] == Character.TYPE)) {
124                   if (!NativeJavaObject.canConvert(args[i], parameterTypes[i])) {
125                      methodNofFound(args);
126                   }
127 
128                   args[i] = Context.jsToJava(args[i], parameterTypes[i]);
129                }
130             }
131 
132             return m.invoke(null, args);
133          } catch (InvocationTargetException ite) {
134             throw Context.throwAsScriptRuntimeEx(ite.getTargetException());
135          } catch (Exception ex) {
136             throw Context.throwAsScriptRuntimeEx(ex);
137          }
138       }
139 
140       protected void methodNofFound(Object[] args) {
141          StringBuffer buffer = new StringBuffer();
142          buffer.append(m.getDeclaringClass());
143          buffer.append('.');
144          buffer.append(getFunctionName());
145          buffer.append('(');
146 
147          for (int i = 0; i < args.length; i++) {
148             if (i > 0) {
149                buffer.append(',');
150             }
151 
152             buffer.append((args[i] == null) ? "null" : args[i].getClass()
153                   .getName());
154          }
155 
156          buffer.append(')');
157 
158          throw Context.reportRuntimeError(ScriptRuntime.getMessage1(
159                "msg.java.no_such_method", buffer.toString()));
160       }
161    }

163    protected class NativeJavaObject extends
164          org.mozilla.javascript.NativeJavaObject {
165       public Object get(String name, Scriptable start) {
166          NativeJavaMethod method = (NativeJavaMethod) javaMethods.get(name);
167 
168          if (method != null) {
169             return method;
170          }
171 
172          for (int i = 0; i < methods.length; i++) {
173             final Method m = methods[i];
174 
175             if (m.getName().equals(name)) {
176                method = new NativeJavaMethod(m, name);
177                javaMethods.put(name, method);
178 
179                return method;
180             }
181          }
182 
183          return Scriptable.NOT_FOUND;
184       }
