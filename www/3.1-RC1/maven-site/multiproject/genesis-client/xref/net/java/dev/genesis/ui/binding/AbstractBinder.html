240    /***
241     * Method that does the bind. It will look in the form and associate
242     * fields, actions and dataproviders with components in the UI.
243     * This method should be the last one used by a binder. 
244     */
245    public void bind() {
246       try {
247          bindFieldMetadatas(formMetadata);
248          bindActionMetadatas(formMetadata);
249          bindDataProviders(formMetadata);
250 
251          setupController();
252          markBound();
253       } catch (Exception e) {
254          unbind();
255 
256          handleException(e);
257       }
258    }

263    protected void bindFieldMetadatas(final FormMetadata formMetadata) {
264       for (final Iterator i =
265             formMetadata.getFieldMetadatas().entrySet().iterator();
266             i.hasNext();) {
267          final Map.Entry entry = (Map.Entry) i.next();
268 
269          BoundField boundField =
270             bindFieldMetadata(entry.getKey().toString(), formMetadata,
271                (FieldMetadata) entry.getValue());
272 
273          if (boundField != null) {
274             boundFields.put(boundField.getName(), boundField);
275          }
276       }
277    }
278 
279    protected void bindActionMetadatas(final FormMetadata formMetadata) {
280       for (final Iterator i =
281             formMetadata.getActionMetadatas().entrySet().iterator();
282             i.hasNext();) {
283          final Map.Entry entry = (Map.Entry) i.next();
284 
285          BoundAction boundAction =
286             bindActionMetadata(((MethodEntry) entry.getKey()).getMethodName(),
287                formMetadata, (ActionMetadata) entry.getValue());
288 
289          if (boundAction != null) {
290             boundActions.put(boundAction.getName(), boundAction);
291          }
292       }
293    }
294 
295    protected void bindDataProviders(final FormMetadata formMetadata) {
296       for (final Iterator i =
297             formMetadata.getDataProviderMetadatas().values().iterator();
298             i.hasNext();) {
299          final DataProviderMetadata dataProviderMetadata =
300             (DataProviderMetadata) i.next();
301 
302          BoundDataProvider boundDataProvider =
303             bindDataProvider(dataProviderMetadata.getWidgetName(),
304                formMetadata, dataProviderMetadata);
305 
306          if (boundDataProvider != null) {
307             boundDataProviders.put(boundDataProvider.getName(),
308                boundDataProvider);
309 
310             bindDataProviderObjectField(dataProviderMetadata, boundDataProvider);
311          }
312       }
313    }

325    protected BoundField bindFieldMetadata(String name,
326          FormMetadata formMetadata, FieldMetadata fieldMetadata) {
327       final Object widget = lookup(name);
328 
329       if (widget == null) {
330          final Object group = lookupButtonGroup(name);
331 
332          if (group == null) {
333             log.warn(name + " could not be found while binding "
334                   + getForm().getClass());
335 
336             return null;
337          }
338 
339          GroupBinder binder = getGroupBinder(group);
340 
341          if (binder == null) {
342             log.warn("No Binder registered for " + group.getClass());
343 
344             return null;
345          }
346 
347          return binder.bind(this, group, fieldMetadata);
348       }
349 
350       WidgetBinder binder = getWidgetBinder(widget);
351 
352       if (binder == null) {
353          log.warn("No Binder registered for " + widget.getClass());
354 
355          return null;
356       }
357 
358       return binder.bind(this, widget, fieldMetadata);
359    }

488    protected void conditionsChanged(Map updatedConditions, boolean enabled) {
489       for (final Iterator i = updatedConditions.entrySet().iterator();
490             i.hasNext();) {
491          final Map.Entry entry = (Map.Entry) i.next();
492          BoundMember member = getBoundMember(entry.getKey().toString());
493 
494          if (member == null) {
495             return;
496          }
497 
498          if (log.isDebugEnabled()) {
499             log.debug("Changing " + entry.getKey() +
500                (enabled ? " enabled" : " visible") + " state to " +
501                entry.getValue());
502          }
503 
504          if (enabled) {
505             member.setEnabled(((Boolean) entry.getValue()).booleanValue());
506          } else {
507             member.setVisible(((Boolean) entry.getValue()).booleanValue());
508          }
509       }
510    }

573    public void invokeAction(String name) {
574       try {
575          Map stringProperties = null;
576 
577          if (name != null) {
578             final MethodMetadata methodMetadata = getMethodMetadata(name);
579 
580             if ((methodMetadata != null) &&
581                   (methodMetadata.getActionMetadata() != null) &&
582                   methodMetadata.getActionMetadata().isValidateBefore()) {
583                stringProperties = getStringProperties();
584             }
585          }
586 
587          controller.invokeAction(name, stringProperties);
588       } catch (Exception e) {
589          handleException(e);
590       }
591    }

597    protected Map getStringProperties() throws Exception {
598       Map stringProperties = ValidationUtils.getInstance().getPropertiesMap(form);
599 
600       for (Iterator iter = stringProperties.entrySet().iterator(); iter.hasNext();) {
601          Map.Entry entry = (Map.Entry) iter.next();
602          String fieldName = (String) entry.getKey();
603          Object value = entry.getValue();
604 
605          if (value == null || "".equals(value)) {
606             BoundField bound = (BoundField) boundFields.get(fieldName);
607 
608             if (bound == null) {
609                continue;
610             }
611 
612             entry.setValue(bound.getValue());
613          }
614       }
615 
616       return stringProperties;
617    }

629    protected Formatter getVirtualFormatter(String name, String propertyName) {
630       if (propertyName != null && propertyName.startsWith(VIRTUAL_PREFIX)) {
631          propertyName = propertyName.substring(VIRTUAL_PREFIX.length());
632       }
633 
634       String key = propertyName == null ? name : name + '.' + propertyName;
635       Formatter virtualFormatter = (Formatter) formatters.get(key);
636 
637       if (virtualFormatter == null) {
638          throw new IllegalArgumentException("There is no formatter "
639                + "registered for virtual property " + key);
640       }
641 
642       return virtualFormatter;
643    }

649    public String format(String name, String property, Object value, boolean isVirtual) {
650       if (isVirtual) {
651          return getVirtualFormatter(name, property).format(value);
652       }
653 
654       Formatter formatter = null;
655 
656       if (formatters != null) {
657          final String key = property == null ? name + '.' : name + '.'
658                + property;
659          formatter = (Formatter) formatters.get(key);
660       }
661 
662       return (formatter == null) ? FormatterRegistry.getInstance()
663             .format(value) : formatter.format(value);
664    }

831       if ((selected.length >= 1) && isBlank) {
832          boolean remove = false;
833 
834          for (int i = 0; i < selected.length; i++) {
835             selected[i] = selected[i] - 1;

838                remove = true;
839             }
840          }

844                selected = new int[0];
