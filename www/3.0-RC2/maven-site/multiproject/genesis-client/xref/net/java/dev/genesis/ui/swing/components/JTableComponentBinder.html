41  public class JTableComponentBinder extends AbstractComponentBinder {
42     public BoundDataProvider bind(SwingBinder binder, Component component,
43        DataProviderMetadata dataProviderMetadata) {
44        return new JTableComponentBoundDataProvider(binder, (JTable) component,
45           dataProviderMetadata);
46     }

48     public class JTableComponentBoundDataProvider extends AbstractBoundMember
49        implements BoundDataProvider {
50        private final JTable component;
51        private final DataProviderMetadata dataProviderMetadata;
52        private final ListSelectionListener listener;
53        private final TableCellRenderer renderer;
54  
55        public JTableComponentBoundDataProvider(SwingBinder binder,
56              JTable component, DataProviderMetadata dataProviderMetadata) {
57           super(binder, component);
58           this.component = component;
59           this.dataProviderMetadata = dataProviderMetadata;
60  
61           this.component.getSelectionModel().addListSelectionListener(
62                 listener = createListSelectionListener());
63  
64           this.renderer = createCellRenderer();
65  
66           configureTableCellRenderer();
67        }
68  
69        protected JTable getComponent() {
70           return component;
71        }
72  
73        protected DataProviderMetadata getDataProviderMetadata() {
74           return dataProviderMetadata;
75        }
76  
77        protected ListSelectionListener getListener() {
78           return listener;
79        }
80  
81        protected TableCellRenderer createCellRenderer() {
82           return new FormatterTableCellRenderer();
83        }
84  
85        protected void configureTableCellRenderer() {
86           Enumeration en = component.getColumnModel().getColumns();
87  
88           while (en.hasMoreElements()) {
89              TableColumn column = (TableColumn) en.nextElement();
90  
91              if (column.getCellRenderer() == null) {
92                 column.setCellRenderer(renderer);
93              }
94           }
95        }
96  
97        protected ListSelectionListener createListSelectionListener() {
98           return new ListSelectionListener() {
99              public void valueChanged(ListSelectionEvent event) {
100                if (event.getValueIsAdjusting()) {
101                   return;
102                }
103 
104                getBinder().updateFormSelection(getDataProviderMetadata(),
105                      getIndexes());
106             }
107          };
108       }
109 
110       protected int[] getIndexes() {
111          return component.getSelectedRows();
112       }
113 
114       public void updateIndexes(int[] indexes) {
115          if (Arrays.equals(getIndexes(), indexes)) {
116             return;
117          }
118 
119          deactivateSelectionListener();
120 
121          try {
122             ListSelectionModel sm = component.getSelectionModel();
123             sm.clearSelection();
124    
125             for (int i = 0; i < indexes.length; i++) {
126                sm.addSelectionInterval(indexes[i], indexes[i]);
127             }
128          } finally {
129             reactivateSelectionListener();
130          }
131       }
132 
133       public void updateList(List data) throws Exception {
134          int[] selected = dataProviderMetadata.isResetSelection() ? new int[0]
135                : component.getSelectedRows();
136 
137          createTableModelAdapter().setData(data);
138          setSelectedIndexes(data.size(), selected);
139       }
140 
141       protected void setSelectedIndexes(int listSize, int[] indexes) {
142          deactivateSelectionListener();
143          
144          try {
145             component.clearSelection();
146             for (int i = 0; i < indexes.length; i++) {
147                if (indexes[i] >= listSize) {
148                   continue;
149                }
150    
151                component.getSelectionModel().addSelectionInterval(indexes[i],
152                      indexes[i]);
153             }
154          } finally {
155             reactivateSelectionListener();
156          }
157       }
158 
159       protected void deactivateSelectionListener() {
160          if (listener != null) {
161             component.getSelectionModel().removeListSelectionListener(listener);
162          }
163       }
164 
165       protected void reactivateSelectionListener() {
166          if (listener != null) {
167             component.getSelectionModel().addListSelectionListener(listener);
168          }
169       }
170 
171       protected TableModelAdapter createTableModelAdapter() {
172          return new TableModelAdapter() {
173             public void setData(List data) throws Exception {
174                DefaultTableModel model = (DefaultTableModel) component
175                      .getModel();
176 
177                int dataSize = data.size();
178                if (model.getRowCount() != data.size()) {
179                   model.setRowCount(dataSize);
180                }
181 
182                int columnCount = component.getColumnModel().getColumnCount();
183                if (model.getColumnCount() != columnCount) {
184                   model.setColumnCount(columnCount);
185                }
186 
187                int i = 0;
188                TableColumn column;
189                String propertyName;
190                for (Iterator iter = data.iterator(); iter.hasNext(); i++) {
191                   Object bean = iter.next();
192 
193                   for (int j = 0; j < model.getColumnCount(); j++) {
194                      column = component.getColumnModel().getColumn(j);
195                      int modelIndex = column.getModelIndex();
196 
197                      Object value = getBinder().isVirtual(
198                            propertyName = getIdentifier(column)) ? bean
199                            : getProperty(bean, propertyName);
200                      model.setValueAt(value, i, modelIndex);
201                   }
202                }
203             }
204          };
205       }
206 
207       protected String getIdentifier(TableColumn column) {
208          String identifier = null;
209          String[] names = (String[]) component.getClientProperty(
210                SwingBinder.COLUMN_NAMES);
211 
212          if (names != null && names.length > column.getModelIndex()) {
213             identifier = names[column.getModelIndex()];
214 
215             if (identifier != null) {
216                return identifier;
217             }
218          }
219 
220          identifier = (String) column.getIdentifier();
221          if (identifier == null) {
222             throw new IllegalArgumentException("Column number "
223                   + column.getModelIndex() + " from Table "
224                   + getBinder().getName(getComponent())
225                   + " does not have an identifier");
226          }
227 
228          return identifier;
229       }
230 
231       public void unbind() {
232          if (listener != null) {
233             component.getSelectionModel().removeListSelectionListener(listener);
234          }
235 
236          if (renderer != null) {
237             Enumeration en = component.getColumnModel().getColumns();
238 
239             while (en.hasMoreElements()) {
240                TableColumn column = (TableColumn) en.nextElement();
241 
242                if (column.getCellRenderer() == renderer) {
243                   column.setCellRenderer(null);
244                }
245             }
246          }
247       }
248    }
