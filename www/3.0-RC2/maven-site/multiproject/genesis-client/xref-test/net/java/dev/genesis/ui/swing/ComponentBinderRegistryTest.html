95     private void assertComponentBinder(Class lookupClass, Class binderClass) {
96        Object binder = registry.get(lookupClass, true);
97        assertNotNull(binder);

101    private void assertNullComponentBinder(Class lookupClass) {

105    public void testRegister() {
106       final WidgetBinder binderInstance = new AbstractButtonComponentBinder();
107 
108       // Test for non-registered type
109       registry.register(Comparable.class, binderInstance);
110       assertSame(binderInstance, registry.get(Comparable.class, true));
111 
112       // Test for already registered type
113       registry.register(JButton.class, binderInstance);
114       assertSame(binderInstance, registry.get(JButton.class, true));
115 
116       final ButtonGroupBinder groupBinder = new ButtonGroupBinder();
117       Object binder = registry.registerButtonGroupBinder(groupBinder);
118       assertSame(groupBinder, binder);
119    }
120 
121    public void testDeregister() {
122       final WidgetBinder binderInstance = new AbstractButtonComponentBinder();
123 
124       assertNotNull(registry.get(JButton.class, true));
125       registry.deregister(AbstractButton.class);
126       assertNull(registry.get(JButton.class, true));
127 
128       // Test for already registered type
129       registry.register(AbstractButton.class, binderInstance);
130       assertSame(binderInstance, registry.get(JButton.class, true));
131 
132       registry.deregister(AbstractButton.class);
133       assertNull(registry.get(JButton.class, true));
134 
135       registry.deregisterButtonGroupBinder();
136       assertNull(registry.getButtonGroupBinder());
137    }
