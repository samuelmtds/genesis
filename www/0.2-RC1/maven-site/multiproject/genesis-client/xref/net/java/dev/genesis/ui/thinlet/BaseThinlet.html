147    protected Object parseAndSet(String fileName, ScreenHandler handler) 
148                                                 throws ScreenNotFoundException {
149       try {
150          InputStream is = getResourceAsStream(fileName, 
151                                              handler.getClass().getClassLoader());
152 
153          if (is == null) {
154             throw new ScreenNotFoundException(fileName);
155          }
156          
157          handler.setScreen(parse(is, handler));
158 
159          is.close();
160 
161          return handler.getScreen();
162       } catch (IOException ioe) {
163          throw new ScreenNotFoundException(fileName);
164       }
165    }

366    protected void displayBean(Object bean, Object root, Map formatters) 
367          throws IllegalAccessException, InvocationTargetException, 
368                 NoSuchMethodException {
369       final Map properties = bean instanceof Map ? (Map)bean : 
370             PropertyUtils.describe(bean);
371 
372       if (root instanceof Thinlet) {
373          root = ((Thinlet)root).getDesktop();
374       }
375 
376       String propertyName;
377       String propertyValue = null;
378       String type;
379       Object component;
380       Collection groupComponents = null;
381 
382       for (final Iterator propertyNames = properties.keySet().iterator();
383                                                    propertyNames.hasNext(); ) {
384          propertyName = propertyNames.next().toString();
385          propertyValue = null;
386          component = find(root, propertyName);
387 
388          if (component == null) {
389             if (groupComponents == null) {
390                groupComponents = getAllOfClass(root, CHECKBOX);
391             }
392 
393             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
394                component = i.next();
395 
396                if (propertyName.equals(getGroup(component)) && 
397                       getName(component) != null ) {
398                   if (propertyValue == null) {
399                      propertyValue = getPropertyValue(properties, propertyName, 
400                            true, formatters);
401                   }
402 
403                   if (getName(component).equals(propertyValue)) {
404                      setSelected(component, true);
405                      break;
406                   }
407                }
408             }
409 
410             continue;
411          }
412 
413          type = getClass(component);
414 
415          if (type.equals(COMBOBOX) || type.equals(LIST)) {
416             Object selectedComponent = getSelectedItem(component);
417 
418             if (selectedComponent != null && type.equals(LIST)) {
419                setSelected(selectedComponent, false);
420             }
421 
422             if (type.equals(COMBOBOX)) {
423                setSelected(component, -1);
424                setText(component, "");
425             }
426 
427             String valuePropertyName = (String)getProperty(component, "value");
428 
429             if (valuePropertyName != null) {
430                Object o = properties.get(propertyName);
431                o = o == null ? null : PropertyUtils.getProperty(
432                      properties.get(propertyName), valuePropertyName);
433 
434                propertyValue = format(formatters, valuePropertyName, o);
435             } else {
436                propertyValue = getPropertyValue(properties, propertyName, true, 
437                      formatters);
438             }
439 
440             selectedComponent = find(component, propertyValue);
441 
442             if (selectedComponent != null) {
443                if (type.equals(LIST)) {
444                   setSelected(selectedComponent, true);
445                } else {
446                   setSelected(component, getIndexOf(component, selectedComponent));
447                }
448             }
449          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
450             setValue(component, getPropertyValue(properties, propertyName, 
451                   false, formatters));
452          } else if (type.equals(PANEL)) {
453             displayBean(properties.get(propertyName), component);
454          } else if (type.equals(CHECKBOX)) {
455             setSelected(component, Boolean.TRUE.equals(properties.get(propertyName)));
456          } else if (!type.equals(TABLE)){
457             setText(component, getPropertyValue(properties, propertyName, 
458                   false, formatters));
459          }
460       }
461    }

500    protected void populate(Object bean, Object root, final Map properties, 
501          final boolean changeBean) throws IllegalAccessException, 
502          InvocationTargetException, NoSuchMethodException {
503       if (changeBean) {
504          properties.putAll(BeanUtils.describe(bean));
505       }
506 
507       if (root instanceof Thinlet) {
508          root = ((Thinlet)root).getDesktop();
509       }
510 
511       String propertyName;
512       String type;
513       Object component;
514       Collection groupComponents = null;
515 
516       for (final Iterator propertyNames = properties.keySet().iterator();
517                                                    propertyNames.hasNext(); ) {
518          propertyName = propertyNames.next().toString();
519          component = find(root, propertyName);
520 
521          if (component == null) {
522             if (groupComponents == null) {
523                groupComponents = getAllOfClass(root, CHECKBOX);
524             }
525 
526             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
527                component = i.next();
528 
529                if (propertyName.equals(getGroup(component)) && 
530                      isSelected(component)) {
531                   properties.put(propertyName, getName(component));
532                   break;
533                }
534             }
535 
536             continue;
537          }
538 
539          type = getClass(component);
540 
541          if (type.equals(COMBOBOX) || type.equals(LIST)) {
542             component = getSelectedItem(component);
543 
544             if (component != null) {
545                properties.put(propertyName, getName(component));
546             } else {
547                properties.put(propertyName, null);
548             }
549          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
550             properties.put(propertyName, getValue(component));
551          } else if (type.equals(TABLE)) {
552             // skip it intentionally
553          } else {
554             properties.put(propertyName, getText(component));
555          }
556       }
557 
558       if (changeBean) {
559          BeanUtils.populate(bean, properties);
560 
561          if (bean instanceof Form) {
562             ((Form)bean).afterPopulate();
563          }
564       }
565    }

596    protected void populateFromEnum(Object component, Class clazz, boolean blank) {
597       final boolean combobox = getClass(component).equals(COMBOBOX);
598 
599       if (!combobox && !getClass(component).equals(LIST)) {
600          throw new UnsupportedOperationException();
601       }
602 
603       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
604 
605       String key;
606       Object enumInstance;
607 
608       removeAll(component);
609 
610       if (blank) {
611          add(component, createItemOfType("", "", type));
612       }
613 
614       for (final Iterator i = Enum.getInstances(clazz).iterator(); i.hasNext(); ) {
615          enumInstance = i.next();
616          key = enumInstance.toString();
617 
618          add(component, createItemOfType(key, FormatterRegistry.getInstance()
619                .format(enumInstance), type));
620       }
621    }

654    protected void populateFromCollection(Object component, Collection c, 
655                         String keyProperty, String valueProperty, boolean blank,
656                         String blankLabel, Map formatters) 
657          throws IllegalAccessException, InvocationTargetException, 
658                 NoSuchMethodException {
659       final boolean combobox = getClass(component).equals(COMBOBOX);
660 
661       if (!combobox && !getClass(component).equals(LIST)) {
662          throw new UnsupportedOperationException();
663       }
664 
665       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
666 
667       String key;
668       String description;
669       Object o;
670 
671       removeAll(component);
672 
673       if (blank) {
674          add(component, createItemOfType("", blankLabel == null ? 
675                "" : blankLabel, type));
676       }
677 
678       final String componentName = getName(component);
679 
680       for (final Iterator i = c.iterator(); i.hasNext(); ) {
681          o = i.next();
682 
683          key = format(formatters, componentName + '.' + keyProperty, PropertyUtils
684                .getProperty(o, keyProperty));
685          description = valueProperty == null ? 
686                format(formatters, componentName + '.', o) : format(formatters, 
687                componentName + '.' + valueProperty, PropertyUtils.getProperty(o, 
688                valueProperty));
689 
690          add(component, createItemOfType(key, description, type));
691       }
692    }

706    protected void populateFromCollection(Object component, Collection c, 
707          Map formatters) throws IllegalAccessException, 
708          InvocationTargetException, NoSuchMethodException {
709       if (!getClass(component).equals(TABLE)) {
710          throw new UnsupportedOperationException();
711       }
712 
713       removeAll(component);
714 
715       final Collection propertyNames = new ArrayList();
716       final Object[] columns = getTableColumns(component); 
717       
718       for (int i = 0; i < columns.length; i++) {
719          final String name = getName(columns[i]);
720 
721          if (name == null) {
722             throw new IllegalArgumentException("column (index " + i + ") in " + 
723                   "table " + getName(component) + " does not have a name");
724          }
725 
726          propertyNames.add(name);
727       }
728 
729       final String componentName = getName(component);
730 
731       String propertyName;
732       Object row;
733       Object bean;
734       int indexOfDot;
735       boolean skip;
736 
737       for (final Iterator i = c.iterator(); i.hasNext(); ) {
738          bean = i.next();
739          row = createRow();
740 
741          for (final Iterator it = propertyNames.iterator(); it.hasNext(); ) {
742             propertyName = it.next().toString();
743 
744             indexOfDot = 0;
745             skip = false;
746 
747             while ((indexOfDot = propertyName.indexOf('.', indexOfDot)) != -1) {
748                if (PropertyUtils.getProperty(bean, 
749                      propertyName.substring(0, indexOfDot)) == null) {
750                   add(row, createCell(propertyName, 
751                         format(formatters, componentName + '.' + propertyName, 
752                         null)));
753                   skip = true;
754                   break;
755                }
756 
757                indexOfDot += 1;
758             }
759 
760             if (skip) {
761                continue;
762             }
763 
764             add(row, createCell(propertyName, format(formatters, 
765                   componentName + '.' + propertyName, 
766                   PropertyUtils.getProperty(bean, propertyName))));
767          }
768 
769          add(component, row);
770       }
771    }

785    protected ThinletBinder prepareBinder(Object widget, Object form) 
786          throws Exception {
787       Map formPerClass = (Map)formPerClassPerComponent.get(widget);
788 
789       if (formPerClass == null) {
790          formPerClass = new HashMap();

936       if (throwable instanceof ScreenNotFoundException) {
937          handleException("The error screen file could not be found", throwable);
