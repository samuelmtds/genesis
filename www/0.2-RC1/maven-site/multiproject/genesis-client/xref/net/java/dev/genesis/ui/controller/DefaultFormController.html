159    protected void populate(Map properties, boolean stringMap, Map converters) 
160          throws Exception {
161       if (previousState == null) {
162          previousState = createFormState(currentState);
163       }
164 
165       try {
166          if (properties == null) {
167             properties = PropertyUtils.describe(form);
168          }
169 
170          updateChangedMap(properties, stringMap, converters);
171 
172          if (currentState.getChangedMap().isEmpty()) {
173             log.debug("Nothing changed.");
174 
175             return;
176          }
177 
178          if (stringMap) {
179             PropertyUtils.copyProperties(form, currentState.getChangedMap());
180          }
181 
182          evaluate(false);
183       } catch (Exception e) {
184          reset(previousState);
185 
186          throw e;
187       } finally {
188          previousState = null;
189       }
190    }

192    protected void updateChangedMap(Map newData, boolean stringMap, 
193             Map converters) {
194       Object value;
195       FieldMetadata fieldMeta;
196       Map.Entry entry;
197 
198       final Map changedMap = currentState.getChangedMap();
199 
200       changedMap.clear();
201       changedMap.putAll(GenesisUtils.normalizeMap(newData));
202 
203       if (log.isDebugEnabled()) {
204          log.debug("changedMap started as: " + changedMap);
205       }
206 
207       for (final Iterator i = changedMap.entrySet().iterator(); i.hasNext();) {
208          entry = (Map.Entry) i.next();
209          fieldMeta = formMetadata.getFieldMetadata(entry.getKey().toString());
210 
211          if (fieldMeta == null) {
212             continue;
213          }
214 
215          value = (stringMap) ? getConverter(fieldMeta, converters).convert(
216                fieldMeta.getFieldClass(), entry.getValue()) : entry.getValue();
217 
218          if (previousState != null && fieldMeta.getEqualityComparator().equals(
219                value, previousState.getValuesMap().get(entry.getKey()))
220                && fieldMeta.getEqualityComparator().equals(value,
221                      currentState.getValuesMap().get(entry.getKey()))) {
222             i.remove();
223 
224             if (log.isDebugEnabled()) {
225                log.debug(entry.getKey() + " removed from changedMap for being "
226                      + " equal to previousValue; is " + entry.getValue()
227                      + "; was:" + previousState.getValuesMap().get(entry
228                      .getKey()));
229             }
230 
231             continue;
232          }
233 
234          if (log.isDebugEnabled()) {
235             log.debug("Field '" + entry.getKey() + "' changed to '"
236                   + entry.getValue() + "'; was " + (previousState == null ? 
237                   "undefined" : previousState.getValuesMap().get(entry
238                   .getKey())));
239          }
240 
241          entry.setValue(value);
242          currentState.getValuesMap().put(entry.getKey(), entry.getValue());
243       }
244    }

257    protected void evaluate(boolean firstCall) throws Exception {
258       evaluateNamedConditions();
259 
260       if (evaluateClearOnConditions()) {
261          evaluateNamedConditions();
262       }
263 
264       if (evaluateCallWhenConditions(firstCall)) {
265          evaluateNamedConditions();
266       }
267 
268       if (evaluateDataProvidedIndexes()) {
269          evaluateNamedConditions();
270       }
271 
272       final Map newData = PropertyUtils.describe(form);
273       updateChangedMap(newData, false, null);
274 
275       fireValuesChanged(currentState.getChangedMap());
276 
277       evaluateEnabledWhenConditions();
278       evaluateVisibleWhenConditions();
279    }

287    protected boolean evaluateClearOnConditions() throws Exception {
288       final Map fieldMetadatas = formMetadata.getFieldMetadatas();
289       final Map dataProviderMetadatas = formMetadata.getDataProviderMetadatas();
290       final Map toCopy = new HashMap();
291 
292       boolean changed;
293       FieldMetadata fieldMetadata;
294       DataProviderMetadata dataProviderMeta;
295       List items;
296       Map.Entry entry;
297       final Map properties = PropertyUtils.describe(form);
298 
299       do {
300          changed = false;
301          properties.putAll(toCopy);
302 
303          for (final Iterator i = fieldMetadatas.entrySet().iterator(); i
304                .hasNext();) {
305             entry = (Map.Entry) i.next();
306             fieldMetadata = (FieldMetadata) entry.getValue();
307 
308             if (fieldMetadata.getClearOnCondition() == null
309                   || fieldMetadata.getEmptyResolver().isEmpty(
310                         properties.get(entry.getKey()))) {
311                i.remove();
312                continue;
313             }
314 
315             if (isConditionSatisfied(fieldMetadata.getClearOnCondition())) {
316                if (log.isDebugEnabled()) {
317                   log.debug("ClearOn Condition for field '" + entry.getKey()
318                         + "' satisfied. Setting '"
319                         + fieldMetadata.getEmptyValue() + "'");
320                }
321                changed = true;
322                toCopy.put(entry.getKey(), fieldMetadata.getEmptyValue());
323                i.remove();
324             }
325          }
326          
327          for (Iterator i = dataProviderMetadatas.entrySet().iterator(); 
328                i.hasNext();) {
329             entry = (Map.Entry) i.next();
330             dataProviderMeta = (DataProviderMetadata) entry.getValue();
331             
332             if (dataProviderMeta.getClearOnCondition() == null) {
333                i.remove();
334                continue;
335             }
336 
337             if (isConditionSatisfied(dataProviderMeta.getClearOnCondition())) {
338                if (log.isDebugEnabled()) {
339                   log.debug("ClearOn Condition for data provider '" + 
340                         entry.getKey() + "' satisfied. Clearing data " +
341                         "provider list.");
342                }
343 
344                changed = true;
345                currentState.getDataProvidedMap().put(dataProviderMeta, 
346                      items = new ArrayList());
347                fireDataProvidedListMetadataChanged(dataProviderMeta, items);
348                i.remove();
349             }
350          }
351       } while (changed);
352 
353       if (toCopy.isEmpty()) {
354          if (log.isDebugEnabled()) {
355             log.debug("Nothing changed in ClearOn conditions.");
356          }
357 
358          return false;
359       }
360 
361       PropertyUtils.copyProperties(form, toCopy);
362       currentState.getChangedMap().putAll(toCopy);
363 
364       return true;
365    }

381          }

393    protected void evaluateConditions(boolean enabled) {
394       Map.Entry entry;
395       MemberMetadata memberMetadata;
396       Boolean newValue;
397       Map conditionMap;
398 
399       final Map updatedConditions = new HashMap();
400       final Map toEvaluate = new HashMap(formMetadata.getFieldMetadatas());
401       toEvaluate.putAll(formMetadata.getActionMetadatas());
402 
403       for (final Iterator i = toEvaluate.entrySet().iterator(); i.hasNext();) {
404          entry = (Map.Entry) i.next();
405          memberMetadata = (MemberMetadata) entry.getValue();
406 
407          if ((enabled && memberMetadata.getEnabledCondition() == null)
408                || (!enabled && memberMetadata.getVisibleCondition() == null)) {
409             continue;
410          }
411 
412          newValue = isSatisfied(enabled ? memberMetadata.getEnabledCondition()
413                : memberMetadata.getVisibleCondition());
414          
415          conditionMap = enabled ?
416                currentState.getEnabledMap() : currentState.getVisibleMap();
417 
418          if (!newValue.equals(conditionMap.get(memberMetadata
419                .getName()))) {
420             conditionMap.put(memberMetadata.getName(), newValue);
421             updatedConditions.put(memberMetadata.getName(), newValue);
422          }
423 
424          if (log.isDebugEnabled()) {
425             log.debug((enabled ? "EnabledWhen" : "VisibleWhen")
426                   + " Condition for member '" + memberMetadata.getName()
427                   + "' evaluated as '"
428                   + conditionMap.get(memberMetadata.getName()) + "'");
429          }
430       }
431 
432       if (enabled) {
433          fireEnabledConditionChanged(updatedConditions);
434       } else {
435          fireVisibleConditionChanged(updatedConditions);
436       }
437    }

464    protected boolean evaluateDataProvidedIndexes() throws Exception {
465       boolean changed = false;
466       DataProviderMetadata dataMeta;
467       Object indexes;
468       int[] selectedIndexes;
469       Object objectFieldValue;
470 
471       for (final Iterator i = formMetadata.getDataProviderIndexes().values()
472             .iterator(); i.hasNext(); ) {
473          dataMeta = (DataProviderMetadata) i.next();
474          indexes = currentState.getChangedMap().get(dataMeta.getIndexField().getFieldName());
475 
476          if (indexes == null) {
477             if (log.isDebugEnabled()) {
478                log.debug("Index field " + dataMeta.getIndexField()
479                      .getFieldName() + " haven't been changed.");
480             }
481 
482             continue;
483          }
484 
485          changed = true;
486 
487          selectedIndexes = dataMeta.getSelectedIndexes(indexes);
488 
489          objectFieldValue = dataMeta.populateSelectedFields(form,
490                (List)currentState.getDataProvidedMap().get(dataMeta),
491                selectedIndexes);
492          fireDataProvidedIndexesChanged(dataMeta, selectedIndexes);
493 
494          if (dataMeta.getObjectField() == null) {
495             continue;
496          }
497 
498          currentState.getChangedMap().put(dataMeta.getObjectField()
499                .getFieldName(), objectFieldValue);
500       }
501 
502       return changed;
503    }

505    protected boolean invokeAction(MethodMetadata methodMetadata, 
506          boolean firstCall, boolean conditionally) throws Exception {
507       List items = null;
508       DataProviderMetadata dataProviderMeta = methodMetadata
509             .getDataProviderMetadata();
510 
511       if ((methodMetadata.getCallCondition() == null || firstCall) 
512             && conditionally) {
513          if (dataProviderMeta == null || (conditionally && !firstCall)) {
514             return false;
515          }
516 
517          if (!dataProviderMeta.isCallOnInit()) {
518             currentState.getDataProvidedMap().put(dataProviderMeta, 
519                   items = new ArrayList());
520             fireDataProvidedListMetadataChanged(dataProviderMeta, items);
521 
522             return true;
523          }
524          conditionally = false;
525       }
526 
527       final boolean satisfied = !conditionally || isConditionSatisfied(
528             methodMetadata.getCallCondition());
529 
530       if (satisfied) {
531          if (dataProviderMeta == null && !beforeInvokingMethod(methodMetadata)) {
532             return false;
533          }
534 
535          final Object ret = methodMetadata.invoke(form);
536 
537          if (dataProviderMeta != null) {
538             items = (ret.getClass().isArray()) ? Arrays.asList((Object[]) ret) :
539                  (List)ret;
540             currentState.getDataProvidedMap().put(dataProviderMeta, items);
541             fireDataProvidedListMetadataChanged(dataProviderMeta, items);
542          } 
543 
544          if (methodMetadata.getActionMetadata() != null) {
545             afterInvokingMethod(methodMetadata);
546          }
547       }
548 
549       if (log.isDebugEnabled()) {
550          log.debug("CallWhen Condition for method '" + 
551                methodMetadata.getName() + "' evaluated as '" + 
552                satisfied + "'");
553       }
554 
555       return satisfied;
556    }

600    public void reset(FormState state) throws Exception {
601       if (log.isDebugEnabled()) {
602          log.debug("Reseting form '" + form + "'");
603       }
604 
605       currentState = state = createFormState(state);
606       ctx.getVariables().declareVariable(CURRENT_STATE_KEY, currentState);
607 
608       fireEnabledConditionChanged(state.getEnabledMap());
609       fireVisibleConditionChanged(state.getVisibleMap());
610 
611       for (final Iterator i = state.getDataProvidedMap().entrySet().iterator(); 
612             i.hasNext();) {
613          final Map.Entry entry = (Map.Entry)i.next();
614          fireDataProvidedListMetadataChanged((DataProviderMetadata)entry
615                .getKey(), (List)entry.getValue());
616       }
617 
618       final Map currentValues = PropertyUtils.describe(form);
619 
620       for (final Iterator i = formMetadata.getDataProviderIndexes().values()
621             .iterator(); i.hasNext(); ) {
622          DataProviderMetadata dataMeta = (DataProviderMetadata) i.next();
623          String fieldName = dataMeta.getIndexField().getFieldName();
624          Object indexes = currentValues.get(fieldName);
625 
626          if (getFormMetadata().getFieldMetadata(fieldName)
627                .getEqualityComparator().equals(indexes, state.getValuesMap()
628                .get(fieldName))) {
629             continue;
630          }
631 
632          fireDataProvidedIndexesChanged(dataMeta, 
633                dataMeta.getSelectedIndexes(indexes));
634       }
635 
636 
637       for (final Iterator i = currentValues.entrySet().iterator(); 
638             i.hasNext();) {
639          Map.Entry entry = (Map.Entry)i.next();
640          FieldMetadata fieldMeta = formMetadata.getFieldMetadata(entry.getKey()
641                .toString());
642 
643          if (fieldMeta == null || fieldMeta.getEqualityComparator().equals(
644                entry.getValue(), state.getValuesMap().get(entry.getKey()))) {
645             i.remove();
646 
647             continue;
648          }
649 
650          currentValues.put(entry.getKey(), state.getValuesMap().get(
651                entry.getKey()));
652       }
653 
654       PropertyUtils.copyProperties(form, currentValues);
655       state.getChangedMap().clear();
656       state.getChangedMap().putAll(currentValues);
657 
658       fireValuesChanged(state.getValuesMap());
659    }

665    public void invokeAction(String name, Map stringProperties) throws Exception {
666       if (name == null) {
667          throw new IllegalArgumentException("Cannot invoke null action");
668       }
669 
670       if (log.isDebugEnabled()) {
671          log.debug("Invoking action: " + name);
672       }
673 
674       final MethodEntry entry = new MethodEntry(name, new String[0]);
675       final MethodMetadata methodMetadata = getFormMetadata()
676             .getMethodMetadata(entry);
677 
678       if (methodMetadata == null) {
679          throw new IllegalArgumentException("Couldn't find action named " + 
680                name);
681       }
682 
683       if (methodMetadata.getActionMetadata() != null && methodMetadata
684             .getActionMetadata().isValidateBefore()) {
685          final String formName = form.getClass().getName();
686 
687          final Map validationErrors = ValidationUtils.getInstance()
688                .getMessages(ValidationUtils.getInstance().validate(
689                stringProperties == null || stringProperties.isEmpty() ?
690                form : stringProperties, formName), formName);
691 
692          if (!validationErrors.isEmpty()) {
693             throw new ValidationException(validationErrors);
694          }
695       } 
696 
697       invokeActionWithReset(methodMetadata, false, false);
698 
699       update();
700    }

708       } catch (Exception e) {

735    public void fireAllEvents(FormControllerListener listener) throws Exception {
736       listener.enabledConditionsChanged(currentState.getEnabledMap());
737       listener.visibleConditionsChanged(currentState.getVisibleMap());
738 
739       for (final Iterator i = currentState.getDataProvidedMap().entrySet()
740             .iterator(); i.hasNext();) {
741          final Map.Entry entry = (Map.Entry)i.next();
742          listener.dataProvidedListChanged((DataProviderMetadata)entry.getKey(), 
743                (List)entry.getValue());
744       }
745 
746       final Map currentValues = PropertyUtils.describe(form);
747 
748       for (final Iterator i = formMetadata.getDataProviderIndexes().values()
749             .iterator(); i.hasNext(); ) {
750          DataProviderMetadata dataMetadata = (DataProviderMetadata) i.next();
751          String fieldName = dataMetadata.getIndexField().getFieldName();
752          Object currentValue = currentValues.get(fieldName);
753 
754          listener.dataProvidedIndexesChanged(dataMetadata,
755                dataMetadata.getSelectedIndexes(currentValue));
756       }
757 
758       for (final Iterator i = currentValues.entrySet().iterator(); 
759             i.hasNext();) {
760          Map.Entry entry = (Map.Entry)i.next();
761          FieldMetadata fieldMeta = formMetadata.getFieldMetadata(entry.getKey()
762                .toString());
763 
764          if (fieldMeta == null) {
765             i.remove();
766 
767             continue;
768          }
769       }
770 
771       listener.valuesChanged(currentValues);
