62     public void testBind() throws Throwable {
63        final Map map = new HashMap();
64        
65        binder = new ThinletBinder(thinlet, thinlet.getDesktop(), form) {
66           protected void bindFieldMetadatas(final FormMetadata formMetadata) {
67              map.put("bindFieldMetadatas(formMetadata)", formMetadata);
68           }
69  
70           protected void bindActionMetadatas(final FormMetadata formMetadata) {
71              map.put("bindActionMetadatas(formMetadata)", formMetadata);
72           }
73  
74           protected void bindDataProviders(final Collection dataProviders) {
75              map.put("bindDataProviders(dataProviders)", dataProviders);
76           }
77  
78           protected void setupController() throws Exception {
79              map.put("setupController()", Boolean.TRUE);
80           }
81        };
82  
83        binder.bind();
84        
85        // Assert FormMetadata used on bindings and form.getFormMetadata() are the same
86        assertSame("FormMetadata used on bindFieldMetadatas(formMetadata) is wrong.",
87              form.getFormMetadata(), map.get("bindFieldMetadatas(formMetadata)"));
88        assertSame("FormMetadata used on bindActionMetadatas(formMetadata) is wrong.",
89              form.getFormMetadata(), map.get("bindActionMetadatas(formMetadata)"));
90        
91        // Assert dataProvider values are equals
92        assertEquals("DataProvider used on bindDataProviders(dataProviders) is wrong.",
93              new ArrayList(form.getFormMetadata().getDataProviderMetadatas().values()),
94              map.get("bindDataProviders(dataProviders)"));
95        
96        // Assert that setupController() was called
97        assertEquals("binder.setupController() was not called.",
98              Boolean.TRUE, map.get("setupController()"));
99     }

101    public void testSetupController() throws Exception {
102       
103       // Assert that controller has not been setup
104       assertNull(form.getController().get("setup()"));
105       assertTrue(!form.getController().isSetup());
106       assertNull(form.getController().get("addFormControllerListener(listener)"));
107       assertTrue(!form.getController().getFormControllerListeners().contains(binder));
108 
109       // controller.setup() (1st time)
110       binder.setupController();
111       assertEquals(Boolean.TRUE, form.getController().get("setup()"));
112       assertTrue(form.getController().isSetup());
113       // Assert binder was added to the controller's listeners
114       assertEquals(binder, form.getController().get("addFormControllerListener(listener)"));
115       assertTrue(form.getController().getFormControllerListeners().contains(binder));
116       
117       // controller.setup() (2nd time)
118       // Remove binder from controller's listeners
119       form.getController().getFormControllerListeners().remove(binder);
120       binder.setupController();
121       assertTrue(form.getController().isSetup());
122       // Assert binder was added to the controller's listeners
123       assertTrue(form.getController().getFormControllerListeners().contains(binder));
124       // Assert that controller.fireAllEvents(binder) was called
125       assertSame(binder, form.getController().get("fireAllEvents(listener)"));
126    }

265    public void testUpdateSelection() throws Exception {
266 
267       // New table: Selected indexes must be {}
268       binder.updateSelection("table", ThinletUtils.newTable());
269       assertSelection(new int[0]);
270 
271       // table with first and third rows selected: Selected indexes must be {0, 2}
272       binder.updateSelection("table", ThinletUtils.setSelectedIndexes(
273             ThinletUtils.newTable(), new int[] { 0, 2 }));
274       assertSelection(new int[] { 0, 2 });
275 
276       // New combobox: Selected indexes must be < 0
277       binder.updateSelection("combobox", ThinletUtils.newCombobox());
278       assertSelection(-1);
279 
280       // combobox with second choice selected: Selected indexes must be {1}
281       binder.updateSelection("combobox", ThinletUtils.setSelected(ThinletUtils
282             .newCombobox(), 1));
283       assertSelection(1);
284 
285       // New combobox with a blank label: Selected indexes must be < 0
286       binder.updateSelection("combobox", ThinletUtils.setBlank(ThinletUtils
287             .newCombobox()));
288       assertSelection(-1);
289 
290       // combobox with second choice selected and a blank label: Selected indexes must be {0}
291       binder.updateSelection("combobox", ThinletUtils.setBlank(ThinletUtils
292             .setSelected(ThinletUtils.newCombobox(), 1)));
293       assertSelection(0);
294 
295       // combobox with third choice selected and a blank label: Selected indexes must be {1}
296       binder.updateSelection("combobox", ThinletUtils.setBlank(ThinletUtils
297             .setSelected(ThinletUtils.newCombobox(), 2)));
298       assertSelection(1);
299 
300       // New list: Selected indexes must be {}
301       binder.updateSelection("list", ThinletUtils.newList());
302       assertSelection(new int[0]);
303 
304       // list with second and third choice selected: Selected indexes must be {1, 2}
305       binder.updateSelection("list", ThinletUtils.setSelectedIndexes(
306             ThinletUtils.newList(), new int[] { 1, 2 }));
307       assertSelection(new int[] { 1, 2 });
308 
309       // New list with a blank label: Selected indexes must be {}
310       binder.updateSelection("list", ThinletUtils.setBlank(ThinletUtils
311             .newList()));
312       assertSelection(new int[0]);
313       
314       // New list with a blank label and first element selected: Selected indexes must be {}
315       binder.updateSelection("list", ThinletUtils.setBlank(ThinletUtils
316             .setSelectedIndexes(ThinletUtils.newList(), new int[] { 0 })));
317       assertSelection(new int[0]);
318 
319       // list with first, second and third choice selected and a blank label: Selected indexes must be {0, 1}
320       binder.updateSelection("list", ThinletUtils.setBlank(ThinletUtils
321                   .setSelectedIndexes(ThinletUtils.newList(), new int[] { 0, 1,
322                         2 })));
323       assertSelection(new int[] { 0, 1 });
324 
325       // list with second and third choice selected and a blank label: Selected indexes must be {0, 1}
326       binder.updateSelection("list", ThinletUtils.setBlank(ThinletUtils
327             .setSelectedIndexes(ThinletUtils.newList(), new int[] { 1, 2 })));
328       assertSelection(new int[] { 0, 1 });
329    }

355       final Map componentsMap = new HashMap();

362       componentsMap.put("list", Collections.singletonList(ThinletUtils.newList()));
363       componentsMap.put("spinbox", Collections.singletonList(ThinletUtils.newSpinBox()));

369       
370       // Some unsupported field widgets
371       componentsMap.put("table", Collections.singletonList(ThinletUtils.newTable()));
372       
373       // Some supported field widgets, but not writeable fields
374       componentsMap.put("checkbox_not_writeable", Collections.singletonList(ThinletUtils.newCheckbox()));
375       
376       // empty widgets list
377       componentsMap.put("no_components", Collections.EMPTY_LIST);
378       
379       thinlet = new BaseThinlet() {
380          public void setMethod(Object component, String key, String value, Object root, Object handler) {
381             methods.put(component, "key=" + key + ";value=" + value
382                   + ";root=" + root + ";handler=" + handler);
383          }
384       };
385       binder = new ThinletBinder(thinlet, thinlet.getDesktop(), form) {    
386          protected List findComponents(final String name) {
387             return (List)componentsMap.get(name);
388          }
389          
390          protected void createWidgetGroup(Object component, String name) {
391             widgetGroup.put(name, component);
392          }     
393       };
394       
395       final FormMetadata formMetadata = form.getFormMetadata();
396       formMetadata.addFieldMetadata("checkbox", new FieldMetadata("checkbox", Object.class, true));
397       formMetadata.addFieldMetadata("checkbox_group", new FieldMetadata("checkbox_group", Object.class, true));
398       formMetadata.addFieldMetadata("togglebutton", new FieldMetadata("togglebutton", Object.class, true));
399       formMetadata.addFieldMetadata("togglebutton_group", new FieldMetadata("togglebutton_group", Object.class, true));
400       formMetadata.addFieldMetadata("combobox", new FieldMetadata("combobox", Object.class, true));
401       formMetadata.addFieldMetadata("list", new FieldMetadata("list", Object.class, true));
402       formMetadata.addFieldMetadata("spinbox", new FieldMetadata("spinbox", Object.class, true));
403       formMetadata.addFieldMetadata("label", new FieldMetadata("label", Object.class, true));
404       formMetadata.addFieldMetadata("slider", new FieldMetadata("slider", Object.class, true));
405       formMetadata.addFieldMetadata("textarea", new FieldMetadata("textarea", Object.class, true));
406       formMetadata.addFieldMetadata("textfield", new FieldMetadata("textfield", Object.class, true));
407       formMetadata.addFieldMetadata("table", new FieldMetadata("table", Object.class, true));
408       formMetadata.addFieldMetadata("checkbox_not_writeable", new FieldMetadata("checkbox_not_writeable", Object.class, false));
409       formMetadata.addFieldMetadata("no_components", new FieldMetadata("no_components", Object.class, false));
410 
411       binder.bindFieldMetadatas(formMetadata);
412 
413       // Assert that supported widgets are bound
414       assertEquals("checkbox", thinlet.getName(widgetGroup.get("checkbox")));

421       assertEquals("label", thinlet.getName(widgetGroup.get("label")));
422       assertEquals("slider", thinlet.getName(widgetGroup.get("slider")));

424       assertEquals("textfield", thinlet.getName(widgetGroup.get("textfield")));
