19  package net.java.dev.genesis.aspect;
20  
21  import net.java.dev.genesis.exception.TimeoutException;
22  
23  import org.apache.commons.logging.Log;
24  import org.apache.commons.logging.LogFactory;
25  import org.codehaus.aspectwerkz.AspectContext;
26  import org.codehaus.aspectwerkz.joinpoint.JoinPoint;
27  
28  /***
29   * @Aspect("perJVM")
30   */
31  public class TimeoutAspect {
32     private static final Log log = LogFactory.getLog(TimeoutAspect.class);
33  
34     private static class WorkerThread extends Thread {
35        private Throwable throwable;
36        private Object returnValue;
37        private JoinPoint jp;
38        private Object lock = new Object();
39        private boolean started;
40        private boolean running;
41        private boolean waiting;
42  
43        public WorkerThread() {
44           setDaemon(true);
45        }
46  
47        public void execute(JoinPoint jp, long timeout) {
48           this.jp = jp;
49           returnValue = null;
50           throwable = null;
51           running = true;
52           waiting = false;
53  
54           if (!started) {
55              started = true;
56              start();
57           } else {
58              synchronized (lock) {
59                 lock.notifyAll();
60              }
61           }
62  
63           synchronized (lock) {
64              try {
65                 waiting = true;
66                 lock.wait(timeout);
67              } catch (InterruptedException ie) {
68                 log.info(ie);
69              }
70           }
71        }
72  
73        public void run() {
74           do {
75              try {
76                 returnValue = jp.proceed();
77              } catch (final Throwable t) {
78                 throwable = t;
79              }
80  
81              running = false;
82              jp = null;
83  
84              while (!waiting) {
85                 try {
86                    Thread.sleep(10);
87                 } catch (InterruptedException ie) {
88                    if (log.isTraceEnabled()) {
89                       log.trace(ie);
90                    }
91                 }
92              }
93  
94              synchronized (lock) {
95                 lock.notifyAll();
96  
97                 try {
98                    lock.wait();
99                 } catch (InterruptedException ie) {
100                   if (log.isTraceEnabled()) {
101                      log.trace(ie);
102                   }
103                }
104             }
105          } while (!isInterrupted());
106       }
107 
108       public Object getReturnValue() {
109          return returnValue;
110       }
111 
112       public Throwable getThrowable() {
113          return throwable;
114       }
115 
116       public boolean isRunning() {
117          return running;
118       }
119 
120       public void cleanUp() {
121          returnValue = null;
122          throwable = null;
123       }
124    }
125 
126    private final long timeout;
127    private final boolean keepThreadInstance;
128    private final ThreadLocal threadLocal = new ThreadLocal();
129    
130    public TimeoutAspect(final AspectContext ctx) {
131       this.timeout = Long.parseLong(ctx.getParameter("timeout"));
132       this.keepThreadInstance = "true".equals(
133             ctx.getParameter("keepThreadInstance"));
134    }
135 
136    private WorkerThread getWorkerThread() {
137       return (WorkerThread)threadLocal.get();
138    }
139 
140    private void setWorkerThread(WorkerThread workerThread) {
141       threadLocal.set(workerThread);
142    }
143 
144    /***
145     * @Around("timeout")
146     */
147    public Object timeoutAdvice(final JoinPoint jp) throws Throwable {
148       WorkerThread thread = getWorkerThread();
149 
150       if (thread == null) {
151          thread = new WorkerThread();
152          setWorkerThread(thread);
153 
154          if (keepThreadInstance && log.isDebugEnabled()) {
155             log.debug("New thread created: " + thread);
156          }
157       } else {
158          if (keepThreadInstance && log.isDebugEnabled()) {
159             log.debug("Reusing thread: " + thread);
160          }
161       }
162 
163       thread.execute(jp, timeout);
164 
165       if (thread.isRunning()) {
166          try {
167             thread.interrupt();
168          } catch (Throwable t) {
169             log.error(t);
170          }
171 
172          setWorkerThread(null);
173 
174          throw new TimeoutException("Execution took more than " + timeout + 
175                " ms");
176       }
177  
178       final Throwable throwable = thread.getThrowable();
179       final Object returnValue = thread.getReturnValue();
180 
181       if (keepThreadInstance) {
182          thread.cleanUp();
183       } else {
184          try {
185             thread.interrupt();
186          } catch (Throwable t) {
187             log.error(t);
188          }
189 
190          setWorkerThread(null);
191       }
192 
193       if (throwable != null) {
194          throw throwable;
195       }
196 
197       return returnValue;
198    }
