97              properties.putAll(GenesisUtils

99           }
100 
101          if (!properties.isEmpty()) {
102             PropertyUtils.copyProperties(scriptFactory, properties);
103          }

285             public void processMethodAnnotation(
286                   final FormMetadata formMetadata,
287                   final MethodMetadata methodMetadata,
288                   final Annotation annotation) {
289 
290                final Map attributesMap = GenesisUtils.getAttributesMap(
291                      ((UntypedAnnotation) annotation).value().toString());
292                final String widgetName = (String)attributesMap
293                      .get("widgetName");
294                final String objectFieldName = (String)attributesMap
295                      .get("objectField");
296                final String indexFieldName = (String)attributesMap
297                      .get("indexField");
298                final String callOnInit = (String)attributesMap
299                      .get("callOnInit");
300 
301                final DataProviderMetadata dataProviderMetadata = methodMetadata
302                      .getDataProviderMetadata();
303 
304                if (widgetName == null && objectFieldName == null && indexFieldName == null) {
305                   throw new RuntimeException("At least one of widgetName, objectField or " + 
306                         "indexField must be specified for @DataProvider in " +
307                         methodMetadata.getMethodEntry().getMethodName());
308                }
309 
310                dataProviderMetadata.setCallOnInit(!"false".equals(callOnInit));
311 
312                PropertyDescriptor[] descriptors = PropertyUtils
313                      .getPropertyDescriptors(formMetadata.getFormClass());
314 
315                Map descriptorsPerPropertyName = new HashMap();
316 
317                for (int i = 0; i < descriptors.length; i++) {
318                   descriptorsPerPropertyName.put(descriptors[i].getName(), 
319                         descriptors[i]);
320                }
321 
322                if (objectFieldName != null) {
323                   PropertyDescriptor descriptor = (PropertyDescriptor)
324                         descriptorsPerPropertyName.get(objectFieldName);
325 
326                   if (descriptor == null) {
327                      throw new RuntimeException("The object "
328                            + formMetadata.getFormClass()
329                            + " doesn´t have a field called " + objectFieldName);
330                   }
331 
332                   final Class fieldType = descriptor.getPropertyType();
333 
334                   if (fieldType.isPrimitive()
335                         || (fieldType.isArray() &&
336                               fieldType.getComponentType().isPrimitive())) {
337                      throw new RuntimeException("The object " + objectFieldName
338                            + " has an object field called " + objectFieldName
339                            + " that cannot be primitive or array of primitives");
340                   }
341 
342                   dataProviderMetadata.setObjectField(new FieldEntry(
343                         descriptor.getName(), descriptor.getPropertyType()));
344                }
345 
346                if (indexFieldName != null) {
347                   PropertyDescriptor descriptor = (PropertyDescriptor)
348                         descriptorsPerPropertyName.get(indexFieldName);
349 
350                   if (descriptor == null) {
351                      throw new RuntimeException("The object "
352                            + formMetadata.getFormClass()
353                            + " doesn´t have a field called " + indexFieldName);
354                   }
355 
356                   final Class fieldType = descriptor.getPropertyType()
357                         .isArray() ? descriptor.getPropertyType()
358                         .getComponentType() : descriptor.getPropertyType();
359 
360                   if (!Collection.class.isAssignableFrom(descriptor
361                         .getPropertyType())
362                         && !Integer.TYPE.isAssignableFrom(fieldType)
363                         && !Integer.class.isAssignableFrom(fieldType)) {
364                      throw new RuntimeException(
365                            "The object "
366                                  + objectFieldName
367                                  + " has an index field called "
368                                  + indexFieldName
369                                  + " that's not an Integer, int, Collection of them, or array of them.");
370                   }
371 
372                   dataProviderMetadata.setIndexField(new FieldEntry(descriptor
373                         .getName(), descriptor.getPropertyType()));
374                }
375 
376                dataProviderMetadata
377                      .setWidgetName(widgetName != null ? widgetName
378                            : objectFieldName != null ? objectFieldName
379                                  : indexFieldName);
380             }
381          }

579       private void processFormAnnotations(final FormMetadata formMetadata) {
580          final List annotations = Annotations.getAnnotationInfos(formMetadata
581                .getFormClass());
582          AnnotationInfo info;
583          MetadataAttribute attr;
584          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
585             info = (AnnotationInfo) iter.next();
586             attr = MetadataAttribute.get(info.getName());
587 
588             if (attr == null) {
589                continue;
590             }
591 
592             attr.getHandler().processFormAnnotation(formMetadata,
593                   info.getAnnotation());
594          }
595       }

597       private void processFieldsAnnotations(final FormMetadata formMetadata) {
598          final Class clazz = formMetadata.getFormClass();
599          final PropertyDescriptor[] propertyDescriptors = PropertyUtils
600                .getPropertyDescriptors(clazz);
601          PropertyDescriptor propDesc;
602          FieldMetadata fieldMetadata;
603          for (int i = 0; i < propertyDescriptors.length; i++) {
604             propDesc = propertyDescriptors[i];
605 
606             // Ignoring java.lang.Object.getClass()
607             if (propDesc.getName().equals("class")
608                   || propDesc.getReadMethod() == null
609                   || Annotations.getAnnotation("NotBound", propDesc
610                         .getReadMethod()) != null) {
611                continue;
612             }
613 
614             fieldMetadata = new FieldMetadata(propDesc.getName(), propDesc
615                   .getPropertyType(), propDesc.getWriteMethod() != null);
616             formMetadata.addFieldMetadata(propDesc.getName(), fieldMetadata);
617             processFieldAnnotations(formMetadata, fieldMetadata, getReadMethod(
618                   clazz, propDesc));
619 
620          }
621       }

623       private void processFieldAnnotations(final FormMetadata formMetadata,
624             final FieldMetadata fieldMetadata, final Method fieldGetterMethod) {
625          final List annotations = Annotations

642       private void processMethodsAnnotations(final FormMetadata formMetadata) {
643          final Method[] methods = formMetadata.getFormClass().getMethods();
644          boolean isAction;
645          boolean isProvider;
646          MethodMetadata methodMetadata;
647 
648          for (int i = 0; i < methods.length; i++) {
649             isAction = Annotations.getAnnotation("Action", methods[i]) != null;
650             isProvider = Annotations.getAnnotation("DataProvider", methods[i]) != null;
651 
652             if (!isAction && !isProvider) {
653                continue;
654             }
655 
656             methodMetadata = new MethodMetadata(methods[i], isAction, isProvider);
657             formMetadata.addMethodMetadata(methods[i], methodMetadata);
658             processMethodAnnotations(formMetadata, methodMetadata, methods[i]);
659          }
660       }

662       private void processMethodAnnotations(final FormMetadata formMetadata,
663             final MethodMetadata methodMetadata, final Method actionMethod) {
664          final List annotations = Annotations.getAnnotationInfos(actionMethod);
665          AnnotationInfo info;
666          MetadataAttribute attr;
667          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
668             info = (AnnotationInfo) iter.next();
669             attr = MetadataAttribute.get(info.getName());
670 
671             if (attr == null) {
672                continue;
673             }
674 
675             attr.getHandler().processMethodAnnotation(formMetadata,
676                   methodMetadata, info.getAnnotation());
677          }

680       // Work around for a bug in JDK 1.4.2
681       private Method getReadMethod(Class clazz, PropertyDescriptor pd) {
682          Method method = pd.getReadMethod();
683 
684          if (clazz.equals(method.getDeclaringClass())) {
685             return method;
686          }
687 
688          try {
689             method = clazz.getMethod(method.getName(), 
690                   method.getParameterTypes());
691             return (method == null) ? pd.getReadMethod() : method;
692          } catch (NoSuchMethodException nsme) {
693             return method;
694          }
