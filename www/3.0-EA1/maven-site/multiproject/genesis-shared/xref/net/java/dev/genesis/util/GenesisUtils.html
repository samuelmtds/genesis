19  package net.java.dev.genesis.util;

21  import java.util.HashMap;
22  import java.util.Map;
23  import java.util.StringTokenizer;
24  
25  public final class GenesisUtils {
26  
27     public static Map normalizeMap(final Map map) {
28        map.remove("class");
29        map.remove("context");
30        return map;
31     }
32  
33     /***
34      * Reads a string line containing <code>key1=value1 key2=value2</code>
35      * and makes a Map.
36      * 
37      * @param attributesLine
38      * @return
39      */
40     public static Map getAttributesMap(final String attributesLine) {
41        return getAttributesMap(attributesLine, " \t\n\r\f", "=");
42     }
43     
44     public static Map getAttributesMap(final String attributesLine, String pairsDelim, String keyValueDelim) {
45        final StringTokenizer pairsTok = new StringTokenizer(attributesLine, pairsDelim);
46        final Map attributesMap = new HashMap();
47        while (pairsTok.hasMoreTokens()) {
48           final StringTokenizer keyValueTok = new StringTokenizer(
49                 pairsTok.nextToken(), keyValueDelim);
50           if (keyValueTok.countTokens() == 2) {
51              attributesMap.put(keyValueTok.nextToken().trim(), keyValueTok
52                    .nextToken().trim());
53           }
54        }
55        return attributesMap;
56     }
