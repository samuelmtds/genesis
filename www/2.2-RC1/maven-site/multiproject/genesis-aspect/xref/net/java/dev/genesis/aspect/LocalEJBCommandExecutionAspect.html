19  package net.java.dev.genesis.aspect;
20  
21  import java.lang.reflect.InvocationTargetException;
22  
23  import javax.naming.InitialContext;
24  
25  import net.java.dev.genesis.ejb.CommandExecutorLocal;
26  import net.java.dev.genesis.ejb.CommandExecutorLocalHome;
27  
28  import org.apache.commons.logging.Log;
29  import org.apache.commons.logging.LogFactory;
30  import org.codehaus.aspectwerkz.CrossCuttingInfo;
31  import org.codehaus.aspectwerkz.joinpoint.JoinPoint;
32  import org.codehaus.aspectwerkz.joinpoint.MethodRtti;
33  
34  /***
35   * @Aspect perJVM
36   */
37  public class LocalEJBCommandExecutionAspect extends CommandInvocationAspect {
38     private static Log log = LogFactory.getLog(LocalEJBCommandExecutionAspect.class);
39  
40     private CommandExecutorLocalHome home;
41     private CommandExecutorLocal session;
42  
43     public LocalEJBCommandExecutionAspect(CrossCuttingInfo ccInfo) {
44        super(ccInfo);
45     }
46  
47     private CommandExecutorLocalHome getHome() throws Exception {
48        if (home == null) {
49           home = (CommandExecutorLocalHome)new InitialContext().lookup(
50               ccInfo.getParameter("jndiName"));
51        }
52  
53        return home;
54     }
55  
56     private CommandExecutorLocal getCommandExecutor() {
57        if (session == null) {
58           try {
59              session = getHome().create();
60           } catch (final Exception e) {
61              home = null;
62              log.error(e.getMessage(), e);
63  
64              throw new RuntimeException(e);
65           }
66        }
67  
68        return session;
69     }
70  
71     /***
72      * @Around localEjbCommandExecution
73      */
74     public Object commandExecution(final JoinPoint jp) throws Throwable {
75        final MethodRtti rtti = (MethodRtti)jp.getRtti();
76        final CommandResolver obj = (CommandResolver)jp.getTarget();
77        final Class[] classes = rtti.getParameterTypes();
78        final String[] classNames = new String[classes.length];
79        final String methodName = rtti.getMethod().getName();
80        final Object[] parameterValues = rtti.getParameterValues();
81  
82        for (int i = 0; i < classes.length; i++) {
83           classNames[i] = classes[i].getName();
84        }
85  
86        CommandExecutorLocal commandExecutorLocal = null;
87  
88        try {
89           commandExecutorLocal = getCommandExecutor();
90  
91           if (obj.isTransactional(rtti.getMethod())) {
92              return commandExecutorLocal.executeTransaction(obj, methodName,
93                    classNames, parameterValues);
94           } else if (obj.isRemotable(rtti.getMethod())) {
95              return commandExecutorLocal.executeQuery(obj, methodName, 
96                    classNames, parameterValues);
97           } else {
98              return jp.proceed();
99           }
100       } catch (final InvocationTargetException ite) {
101          throw ite.getTargetException();
102       }
103    }
