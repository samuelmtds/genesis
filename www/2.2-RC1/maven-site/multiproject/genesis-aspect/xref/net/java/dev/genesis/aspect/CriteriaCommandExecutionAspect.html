33  /***
34   * @Aspect perJVM
35   */
36  public class CriteriaCommandExecutionAspect extends CommandInvocationAspect {
37     private static final String CRITERIA_ATTRIBUTE = "Criteria";
38     
39     private boolean useOriginalMethod;
40  
41     public CriteriaCommandExecutionAspect(CrossCuttingInfo ccInfo) {
42        super(ccInfo);
43        
44        useOriginalMethod = "true".equals(ccInfo.getParameter("useOriginalMethod"));
45     }
46  
47     /***
48      * @Around criteriaCommandExecution
49      */
50     public Object commandExecution(JoinPoint jp) throws Throwable {
51        final CriteriaResolver obj = (CriteriaResolver)jp.getTarget();
52        final MethodRtti rtti = (MethodRtti) jp.getRtti();
53        final Class[] classes = rtti.getParameterTypes();
54        final String[] classNames = new String[classes.length];
55  
56        for (int i = 0; i < classes.length; i++) {
57           classNames[i] = classes[i].getName();
58        }
59  
60        final MethodTuple methodTuple = ((MethodRttiImpl) rtti).getMethodTuple();
61        final String methodName = useOriginalMethod ? methodTuple
62              .getOriginalMethod().getName() : methodTuple.getWrapperMethod()
63              .getName();
64        final Object[] parameterValues = rtti.getParameterValues();
65  
66        final UntypedAnnotationProxy annon = (UntypedAnnotationProxy) Annotations
67              .getAnnotation(CRITERIA_ATTRIBUTE, rtti.getMethod());
68        final CriteriaAnnotationParser parser = new CriteriaAnnotationParser(annon.getValue());
69        
70        return new CriteriaCommandExecutor(obj, methodName, classNames,
71              parameterValues, parser.getPersisterClassName(), parser
72                    .getOrderBy(), parser.isAsc(), obj.getPropertiesMap())
73              .execute();
74     }
75  
76     public static class CriteriaAnnotationParser {
77        private String persisterClassName;
78        private String[] orderBy;
79        private boolean[] isAsc;
80  
81        public CriteriaAnnotationParser(String annotation) {
82           String[] values = annotation.trim().split("(order-by//s*=)");
83           
84           if (values.length > 0) {
85              persisterClassName = values[0].trim();
86  
87              if (values.length > 1) {
88                 values = values[1].trim().split("(//s*,//s*)");
89                 orderBy = new String[values.length];
90                 isAsc = new boolean[values.length];
91  
92                 for (int i = 0; i < values.length; i++) {
93                    String[] props = values[i].split("//s+");
94  
95                    if (props.length == 0 || props.length > 2) {
96                       throw new IllegalArgumentException(
97                             "Malformed Criteria annotation");
98                    }
99                    orderBy[i] = props[0];
100 
101                   if (props.length < 2) {
102                      isAsc[i] = true;
103                      continue;
104                   }
105 
106                   if (props[1].equalsIgnoreCase("asc")) {
107                      isAsc[i] = true;
108                   } else if (props[1].equalsIgnoreCase("desc")) {
109                      isAsc[i] = false;
110                   } else {
111                      throw new IllegalArgumentException(
112                            "Malformed Criteria"
113                                  + "annotation: order-by clause must be 'asc' or 'desc'");
114                   }
115                }
116             }
117          }
118       }
119 
120       public String getPersisterClassName() {
121          return persisterClassName;
122       }
123 
124       public boolean[] isAsc() {
125          return isAsc;
126       }
127 
128       public String[] getOrderBy() {
129          return orderBy;
130       }
131    }
