34  public class ScriptUtils {
35     private static final ThreadLocal current = new ThreadLocal();

37     public static ScriptContext getCurrentContext() {
38        return (ScriptContext)current.get();
39     }
40  
41     public static void enter(ScriptContext ctx) {
42        current.set(ctx);
43     }
44  
45     public static void exit() {
46        current.set(null);
47     }

49     public static Object getPropertyValue(String objKey, String propertyName) {
50        final Object bean = getCurrentContext().lookup(objKey);
51        if (bean == null) {
52           throw new IllegalArgumentException("Bean called '" + objKey
53                 + "' not found");
54        }
55        try {
56           return PropertyUtils.getProperty(bean, propertyName);
57        } catch (Exception e) {
58           throw new IllegalArgumentException("Bean doesn't have the property '"
59                 + propertyName + "'");
60        }
61     }
62  
63     public static ScriptValue getScriptValue(Object obj) {
64        if (obj instanceof String) {
65           String s = obj.toString();
66  
67           String[] values = s.split(":");
68  
69           if (values.length == 1) {
70              return new ScriptValue(null, obj); 
71           }
72  
73           if (values.length != 2) {
74              throw new IllegalArgumentException("Malformed expression '" + obj + 
75                    "'");       
76           }
77  
78           return new ScriptValue(values[1], getPropertyValue(values[0],
79                 values[1]));       
80        }
81  
82        return new ScriptValue(null, obj); 
83     }

94     public static EmptyResolver getEmptyResolver(ScriptContext ctx,
95           ScriptValue scriptValue) {
96        if (scriptValue.getFieldName() == null) {
97           return EmptyResolverRegistry.getInstance().getDefaultEmptyResolverFor(
98                 scriptValue.getValue() == null ? Object.class : scriptValue
99                       .getValue().getClass());
100       }
101 
102       final FormMetadata meta = getFormMetadata(ctx);
103       final FieldMetadata fieldMetadata = meta.getFieldMetadata(scriptValue.getFieldName());
104 
105       if (fieldMetadata != null) {
106          return fieldMetadata.getEmptyResolver();
107       }
108 
109       return (EmptyResolver) EmptyResolverRegistry.getInstance().get(
110             scriptValue.getValue());
111    }
112 
113    public static EqualityComparator getEqualityComparator(ScriptContext ctx,
114          ScriptValue scriptValue) {
115       if (scriptValue.getFieldName() == null) {
116          return EqualityComparatorRegistry.getInstance()
117                .getDefaultEqualityComparatorFor(
118                      scriptValue.getValue() == null ? Object.class
119                            : scriptValue.getValue().getClass());
120       }
121 
122       final FormMetadata meta = getFormMetadata(ctx);
123       final FieldMetadata fieldMetadata = meta.getFieldMetadata(scriptValue.getFieldName());
124 
125       if (fieldMetadata != null) {
126          return fieldMetadata.getEqualityComparator();
127       }
128 
129       return EqualityComparatorRegistry.getInstance()
130             .getDefaultEqualityComparatorFor(scriptValue.getValue().getClass());
131    }
