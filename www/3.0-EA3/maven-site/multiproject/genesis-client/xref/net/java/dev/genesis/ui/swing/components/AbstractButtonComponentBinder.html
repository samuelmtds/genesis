19  package net.java.dev.genesis.ui.swing.components;

21  import net.java.dev.genesis.ui.binding.BoundAction;
22  import net.java.dev.genesis.ui.metadata.ActionMetadata;
23  import net.java.dev.genesis.ui.swing.SwingBinder;
24  
25  import java.awt.Component;
26  import java.awt.event.ActionEvent;
27  import java.awt.event.ActionListener;
28  
29  import javax.swing.AbstractButton;
30  
31  public class AbstractButtonComponentBinder extends AbstractComponentBinder {
32     public BoundAction bind(SwingBinder binder, Component component,
33        ActionMetadata actionMetadata) {
34        return new AbstractButtonComponentBoundAction(binder,
35           (AbstractButton) component, actionMetadata);
36     }
37  
38     public class AbstractButtonComponentBoundAction extends AbstractBoundMember
39           implements BoundAction {
40        private final AbstractButton component;
41        private final ActionMetadata actionMetadata;
42        private final ActionListener listener;
43  
44        public AbstractButtonComponentBoundAction(SwingBinder binder,
45           AbstractButton component, ActionMetadata actionMetadata) {
46           super(binder, component);
47           this.component = component;
48           this.actionMetadata = actionMetadata;
49           this.component.addActionListener(listener = createActionListener());
50        }
51  
52        protected ActionMetadata getActionMetadata() {
53           return actionMetadata;
54        }
55  
56        protected AbstractButton getComponent() {
57           return component;
58        }
59  
60        protected ActionListener getListener() {
61           return listener;
62        }
63  
64        protected ActionListener createActionListener() {
65           return new ActionListener() {
66              public void actionPerformed(ActionEvent event) {
67                 getBinder().invokeFormAction(getActionMetadata());
68              }
69           };
70        }
71  
72        public void unbind() {
73           if (listener != null) {
74              component.removeActionListener(listener);
75           }
76        }
77     }
78  }
