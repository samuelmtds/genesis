32     public Object wrap(Context cx, Scriptable scope, Object obj, Class staticType) {
33        if ((obj == null) || (obj == Undefined.instance)
34              || obj instanceof Scriptable) {
35           return obj;
36        }
37  
38        if ((staticType != null) && staticType.isPrimitive()) {
39           if (staticType == Void.TYPE) {
40              return Undefined.instance;
41           }
42  
43           return obj;
44        }
45  
46        if (!isJavaPrimitiveWrap()) {
47           if (obj instanceof String || obj instanceof Number
48                 || obj instanceof Boolean || obj instanceof Character) {
49              return obj;
50           }
51        }
52  
53        Class cls = obj.getClass();
54  
55        if (cls.isArray()) {
56           return NativeJavaArray.wrap(scope, obj);
57        }
58  
59        return wrapAsJavaObject(cx, scope, obj, staticType);
60     }

88              Object[] args) {
89           try {
90              Class[] parameterTypes = m.getParameterTypes();
91  
92              if (parameterTypes.length != args.length) {
93                 methodNofFound(args);
94              }
95  
96              for (int i = 0; i < args.length; i++) {
97                 if (args[i] == null) {
98                    continue;
99                 }
100 
101                if ((args[i] == Undefined.instance)
102                      || args[i] instanceof Scriptable
103                      || (parameterTypes[i] == Character.TYPE)) {
104                   if (!NativeJavaObject.canConvert(args[i], parameterTypes[i])) {
105                      methodNofFound(args);
106                   }
107 
108                   args[i] = Context.jsToJava(args[i], parameterTypes[i]);
109                }
110             }
111 
112             return m.invoke(null, args);
113          } catch (Exception ex) {
114             throw Context.throwAsScriptRuntimeEx(ex);
115          }
116       }

123          buffer.append('(');
