94     public void bind() {
95        try {
96           final FormMetadata formMetadata = getFormMetadata(form);
97  
98           bindFieldMetadatas(formMetadata);
99           bindActionMetadatas(formMetadata);
100          bindDataProviders(formMetadata);
101 
102          setupController();
103       } catch (Exception e) {
104          unbind();
105 
106          handleException(e);
107       }
108    }

110    protected void bindFieldMetadatas(final FormMetadata formMetadata) {
111       for (final Iterator i =
112             formMetadata.getFieldMetadatas().entrySet().iterator();
113             i.hasNext();) {
114          final Map.Entry entry = (Map.Entry) i.next();
115 
116          BoundField boundField =
117             bindFieldMetadata(entry.getKey().toString(), formMetadata,
118                (FieldMetadata) entry.getValue());
119 
120          if (boundField != null) {
121             boundFields.put(boundField.getName(), boundField);
122          }
123       }
124    }
125 
126    protected void bindActionMetadatas(final FormMetadata formMetadata) {
127       for (final Iterator i =
128             formMetadata.getActionMetadatas().entrySet().iterator();
129             i.hasNext();) {
130          final Map.Entry entry = (Map.Entry) i.next();
131 
132          BoundAction boundAction =
133             bindActionMetadata(((MethodEntry) entry.getKey()).getMethodName(),
134                formMetadata, (ActionMetadata) entry.getValue());
135 
136          if (boundAction != null) {
137             boundActions.put(boundAction.getName(), boundAction);
138          }
139       }
140    }

142    protected void bindDataProviders(final FormMetadata formMetadata) {
143       for (final Iterator i =
144             formMetadata.getDataProviderMetadatas().values().iterator();
145             i.hasNext();) {
146          final DataProviderMetadata dataProviderMetadata =
147             (DataProviderMetadata) i.next();
148 
149          BoundDataProvider boundDataProvider =
150             bindDataProvider(dataProviderMetadata.getWidgetName(),
151                formMetadata, dataProviderMetadata);
152 
153          if (boundDataProvider != null) {
154             boundDataProviders.put(boundDataProvider.getName(),
155                boundDataProvider);
156 
157             bindDataProviderObjectField(dataProviderMetadata, boundDataProvider);
158          }
159       }
160    }

267    protected void conditionsChanged(Map updatedConditions, boolean enabled) {
268       for (final Iterator i = updatedConditions.entrySet().iterator();
269             i.hasNext();) {
270          final Map.Entry entry = (Map.Entry) i.next();
271          BoundMember member = getBoundMember(entry.getKey().toString());
272 
273          if (member == null) {
274             return;
275          }
276 
277          if (log.isDebugEnabled()) {
278             log.debug("Changing " + entry.getKey() +
279                (enabled ? " enabled" : " visible") + " state to " +
280                entry.getValue());
281          }
282 
283          if (enabled) {
284             member.setEnabled(((Boolean) entry.getValue()).booleanValue());
285          } else {
286             member.setVisible(((Boolean) entry.getValue()).booleanValue());
287          }
288       }
289    }

349    public void invokeAction(String name) {
350       try {
351          Map stringProperties = null;
352 
353          if (name != null) {
354             final MethodMetadata methodMetadata = getMethodMetadata(name);
355 
356             if ((methodMetadata != null) &&
357                   (methodMetadata.getActionMetadata() != null) &&
358                   methodMetadata.getActionMetadata().isValidateBefore()) {
359                stringProperties = getStringProperties();

388    protected Formatter getVirtualFormatter(String name, String propertyName) {
389       if (propertyName.startsWith(VIRTUAL_PREFIX)) {
390          propertyName = propertyName.substring(VIRTUAL_PREFIX.length());
391       }
392 
393       Formatter virtualFormatter = (Formatter) formatters.get(name + '.'
394             + propertyName);
395 
396       if (virtualFormatter == null) {
397          throw new IllegalArgumentException("There is no formatter "
398                + "registered for virtual property " + name + '.' + propertyName);
399       }
400 
401       return virtualFormatter;
402    }

410    public String format(String name, String property, Object value, boolean isVirtual) {
411       if (isVirtual) {
412          return getVirtualFormatter(name, property).format(value);
413       }
414 
415       Formatter formatter = null;
416 
417       if (formatters != null) {
418          final String key = property == null ? name + '.' : name + '.'
419                + property;
420          formatter = (Formatter) formatters.get(key);
421       }
422 
423       return (formatter == null) ? FormatterRegistry.getInstance()
424             .format(value) : formatter.format(value);
425    }

512    public int[] getIndexesFromUI(int[] selected, boolean isBlank) {
513       if ((selected.length >= 1) && isBlank) {
514          boolean remove = false;
515 
516          for (int i = 0; i < selected.length; i++) {
517             selected[i] = selected[i] - 1;
518 
519             if (selected[i] < 0) {
520                remove = true;
521             }
522          }
523 
524          if (remove) {
525             if (selected.length == 1) {
526                selected = new int[0];
527             } else {
528                final int[] aux = selected;
529                selected = new int[aux.length - 1];
530                System.arraycopy(aux, 1, selected, 0, aux.length - 1);
531             }
532          }
533       }
534 
535       return selected;
536    }

538    public int[] getIndexesFromController(int[] selected, boolean isBlank) {
539       if (isBlank) {
540          if (selected.length == 0) {
541             return new int[] { -1 };
542          }
543 
544          for (int i = 0; i < selected.length; i++) {
545             if (selected[i] < 0) {
546                selected[i] = -1;
547                continue;
548             }
549 
550             selected[i] = selected[i] + 1;
551          }
552       }
553 
554       return selected;
