19  package net.java.dev.genesis.ui.swing.lookup;
20  
21  import javax.swing.JPanel;
22  
23  import net.java.dev.genesis.GenesisTestCase;
24  
25  public class DepthFirstComponentLookupStrategyTest extends GenesisTestCase {
26     private DepthFirstComponentLookupStrategy strategy;
27     private JPanel root;
28     private JPanel rootLevel1Position0;
29     private JPanel rootLevel1Position1;
30     private JPanel rootLevel2Position0;
31     private JPanel panelLevel1Position1;
32     private JPanel panelLevel2Position0;
33  
34     public DepthFirstComponentLookupStrategyTest() {
35        super("Depth First Component LookupStrategy Unit Test");
36     }
37  
38     protected void setUp() {
39        strategy = new DepthFirstComponentLookupStrategy();
40        panelLevel1Position1 = new JPanel();
41        panelLevel2Position0 = new JPanel();
42        root = new JPanel();
43        rootLevel1Position0 = new JPanel();
44        rootLevel1Position1 = new JPanel();
45        rootLevel2Position0 = new JPanel();
46  
47        root.add(rootLevel1Position0);
48        root.add(rootLevel1Position1);
49        rootLevel1Position0.add(rootLevel2Position0);
50  
51        rootLevel1Position1.add(panelLevel1Position1);
52        rootLevel2Position0.add(panelLevel2Position0);
53     }
54  
55     public void testSimpleLookup() {
56        panelLevel1Position1.setName("someName");
57  
58        assertSame(panelLevel1Position1, strategy.lookup(root, "someName"));
59  
60        panelLevel1Position1.setName("anotherName");
61        assertSame(panelLevel1Position1, strategy.lookup(root, "anotherName"));
62  
63        panelLevel2Position0.setName("theName");
64        assertSame(panelLevel2Position0, strategy.lookup(root, "theName"));
65     }
66     
67     public void testBreadthFirstLookup() {
68        panelLevel1Position1.setName("someName");
69        panelLevel2Position0.setName("someName");
70        assertSame(panelLevel2Position0, strategy.lookup(root, "someName"));
71  
72        panelLevel2Position0.setName("changeName");
73        assertSame(panelLevel1Position1, strategy.lookup(root, "someName"));
74     }
75  }
