19  package net.java.dev.genesis.reflection;
20  
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  public final class ClassesCache {
25  
26     private static Map cache = new HashMap();
27  
28     static {
29        cache.put(Byte.TYPE.getName(), Byte.TYPE);
30        cache.put(Character.TYPE.getName(), Character.TYPE);
31        cache.put(Short.TYPE.getName(), Short.TYPE);
32        cache.put(Integer.TYPE.getName(), Integer.TYPE);
33        cache.put(Long.TYPE.getName(), Long.TYPE);
34        cache.put(Float.TYPE.getName(), Float.TYPE);
35        cache.put(Double.TYPE.getName(), Double.TYPE);
36        cache.put(Boolean.TYPE.getName(), Boolean.TYPE);
37        cache.put(Void.TYPE.getName(), Void.TYPE);
38     }
39  
40     public static Class getClass(String className) throws ClassNotFoundException {
41        Class clazz = (Class) cache.get(className);
42  
43        if (clazz == null) {
44           clazz = Class.forName(className);
45           cache.put(className, clazz);
46        }
47  
48        return clazz;
49     }
50  
51  }
