34     /***
35      * Reads a string line containing <code>key1=value1 key2=value2</code>
36      * and makes a Map.
37      * 
38      * @param attributesLine
39      * @return
40      */
41     public static Map getAttributesMap(final String attributesLine) {
42        return getAttributesMap(attributesLine, " \t\n\r\f", "=");
43     }
44     
45     public static Map getAttributesMap(final String attributesLine, String pairsDelim, String keyValueDelim) {
46        final StringTokenizer pairsTok = new StringTokenizer(attributesLine, pairsDelim);
47        final Map attributesMap = new HashMap();
48        while (pairsTok.hasMoreTokens()) {
49           final StringTokenizer keyValueTok = new StringTokenizer(
50                 pairsTok.nextToken(), keyValueDelim);
51           if (keyValueTok.countTokens() == 2) {
52              attributesMap.put(keyValueTok.nextToken().trim(), keyValueTok
53                    .nextToken().trim());
54           }
55        }
56        return attributesMap;
57     }
58     
59     public static Map getAttributesMap(String[] values) {
60        final Map attributesMap = new HashMap(values.length);
61        for (int i = 0; i < values.length; i++) {
62           String[] value = values[i].split("//s*=//s*");
63          
64           if (value.length != 2) {
65              throw new IllegalArgumentException("Invalid argument: " + values[i]);
66           }
67           attributesMap.put(value[0].trim(), value[1].trim());
68        }
69        return attributesMap;
70     }
71     
72     public static boolean isBlank(String value) {
73        return value == null || value.trim().length() == 0;
74     }
