148                                                 throws ScreenNotFoundException {
149       try {
150          InputStream is = getResourceAsStream(fileName, 
151                                              handler.getClass().getClassLoader());
152 
153          if (is == null) {
154             throw new ScreenNotFoundException(fileName);
155          }
156          
157          handler.setScreen(parse(is, handler));
158 
159          is.close();
160 
161          return handler.getScreen();
162       } catch (IOException ioe) {
163          throw new ScreenNotFoundException(fileName);
164       }
165    }

366    protected void displayBean(Object bean, Object root, Map formatters) 
367          throws IllegalAccessException, InvocationTargetException, 
368                 NoSuchMethodException {
369       final Map properties = bean instanceof Map ? (Map)bean : 
370             PropertyUtils.describe(bean);
371 
372       if (root instanceof Thinlet) {
373          root = ((Thinlet)root).getDesktop();
374       }
375 
376       String propertyName;
377       String propertyValue = null;
378       String type;
379       Object component;
380       Collection groupComponents = null;
381 
382       for (final Iterator propertyNames = properties.keySet().iterator();
383                                                    propertyNames.hasNext(); ) {
384          propertyName = propertyNames.next().toString();
385          propertyValue = null;
386          component = find(root, propertyName);
387 
388          if (component == null) {
389             if (groupComponents == null) {
390                groupComponents = getAllOfClass(root, CHECKBOX);
391             }
392 
393             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
394                component = i.next();
395 
396                if (propertyName.equals(getGroup(component)) && 
397                       getName(component) != null ) {
398                   if (propertyValue == null) {
399                      propertyValue = getPropertyValue(properties, propertyName, 
400                            true, formatters);
401                   }
402 
403                   if (getName(component).equals(propertyValue)) {
404                      setSelected(component, true);
405                      break;
406                   }
407                }
408             }
409 
410             continue;
411          }
412 
413          type = getClass(component);
414 
415          if (type.equals(COMBOBOX) || type.equals(LIST)) {
416             Object selectedComponent = getSelectedItem(component);
417 
418             if (selectedComponent != null && type.equals(LIST)) {
419                setSelected(selectedComponent, false);
420             }
421 
422             if (type.equals(COMBOBOX)) {
423                setSelected(component, -1);
424                setText(component, "");
425             }
426 
427             String valuePropertyName = (String)getProperty(component, "value");
428 
429             if (valuePropertyName != null) {
430                Object o = properties.get(propertyName);
431                o = o == null ? null : PropertyUtils.getProperty(
432                      properties.get(propertyName), valuePropertyName);
433 
434                propertyValue = format(formatters, valuePropertyName, o);
435             } else {
436                propertyValue = getPropertyValue(properties, propertyName, true, 
437                      formatters);
438             }
439 
440             selectedComponent = find(component, propertyValue);
441 
442             if (selectedComponent != null) {
443                if (type.equals(LIST)) {
444                   setSelected(selectedComponent, true);
445                } else {
446                   setSelected(component, getIndexOf(component, selectedComponent));
447                }
448             }
449          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
450             setValue(component, getPropertyValue(properties, propertyName, 
451                   false, formatters));
452          } else if (type.equals(PANEL)) {
453             displayBean(properties.get(propertyName), component);
454          } else if (type.equals(CHECKBOX)) {
455             setSelected(component, Boolean.TRUE.equals(properties.get(propertyName)));
456          } else if (!type.equals(TABLE)){
457             setText(component, getPropertyValue(properties, propertyName, 
458                   false, formatters));
459          }
460       }
461    }

466 
467       if (enumKey && o instanceof Enum) {

469       }
470 
471       return format(formatters, propertyName, o);
472    }

509    protected void populate(Object bean, Object root, final Map properties, 
510          final boolean changeBean) throws IllegalAccessException, 
511          InvocationTargetException, NoSuchMethodException {
512       if (changeBean) {
513          properties.putAll(BeanUtils.describe(bean));
514       }
515 
516       if (root instanceof Thinlet) {
517          root = ((Thinlet)root).getDesktop();
518       }
519 
520       String propertyName;
521       String type;
522       Object component;
523       Collection groupComponents = null;
524 
525       for (final Iterator propertyNames = properties.keySet().iterator();
526                                                    propertyNames.hasNext(); ) {
527          propertyName = propertyNames.next().toString();
528          component = find(root, propertyName);
529 
530          if (component == null) {
531             if (groupComponents == null) {
532                groupComponents = getAllOfClass(root, CHECKBOX);
533             }
534 
535             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
536                component = i.next();
537 
538                if (propertyName.equals(getGroup(component)) && 
539                      isSelected(component)) {
540                   properties.put(propertyName, getName(component));
541                   break;
542                }
543             }
544 
545             continue;
546          }
547 
548          type = getClass(component);
549 
550          if (type.equals(COMBOBOX) || type.equals(LIST)) {
551             component = getSelectedItem(component);
552 
553             if (component != null) {
554                properties.put(propertyName, getName(component));
555             } else {
556                properties.put(propertyName, null);
557             }
558          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
559             properties.put(propertyName, getValue(component));
560          } else if (type.equals(TABLE)) {
561             // skip it intentionally
562          } else {
563             properties.put(propertyName, getText(component));
564          }
565       }
566 
567       if (changeBean) {
568          BeanUtils.populate(bean, properties);
569 
570          if (bean instanceof Form) {
571             ((Form)bean).afterPopulate();
572          }
573       }
574    }

611    protected void populateFromEnum(Object component, Class clazz, boolean blank) {
612       final boolean combobox = getClass(component).equals(COMBOBOX);
613 
614       if (!combobox && !getClass(component).equals(LIST)) {
615          throw new UnsupportedOperationException();
616       }
617 
618       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
619 
620       String key;
621       Object enumInstance;
622 
623       removeAll(component);
624 
625       if (blank) {
626          add(component, createItemOfType("", "", type));
627       }
628 
629       for (final Iterator i = Enum.getInstances(clazz).iterator(); i.hasNext(); ) {
630          enumInstance = i.next();
631          key = enumInstance.toString();
632 
633          add(component, createItemOfType(key, FormatterRegistry.getInstance()
634                .format(enumInstance), type));
635       }
636    }

669    protected void populateFromCollection(Object component, Collection c, 
670                         String keyProperty, String valueProperty, boolean blank,
671                         String blankLabel, Map formatters) 
672          throws IllegalAccessException, InvocationTargetException, 
673                 NoSuchMethodException {
674       final boolean combobox = getClass(component).equals(COMBOBOX);
675 
676       if (!combobox && !getClass(component).equals(LIST)) {
677          throw new UnsupportedOperationException();
678       }
679 
680       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
681 
682       String key;
683       String description;
684       Object o;
685 
686       removeAll(component);
687 
688       if (blank) {
689          add(component, createItemOfType("", blankLabel == null ? 
690                "" : blankLabel, type));
691       }
692 
693       final String componentName = getName(component);
694 
695       for (final Iterator i = c.iterator(); i.hasNext(); ) {
696          o = i.next();
697 
698          key = format(formatters, componentName + '.' + keyProperty, PropertyUtils
699                .getProperty(o, keyProperty));
700          description = valueProperty == null ? 
701                format(formatters, componentName + '.', o) : format(formatters, 
702                componentName + '.' + valueProperty, PropertyUtils.getProperty(o, 
703                valueProperty));
704 
705          add(component, createItemOfType(key, description, type));
706       }
707    }

721    protected void populateFromCollection(Object component, Collection c, 
722          Map formatters) throws IllegalAccessException, 
723          InvocationTargetException, NoSuchMethodException {
724       if (!getClass(component).equals(TABLE)) {
725          throw new UnsupportedOperationException();
726       }
727 
728       removeAll(component);
729 
730       final Collection propertyNames = new ArrayList();
731       final Object[] columns = getTableColumns(component); 
732       
733       for (int i = 0; i < columns.length; i++) {
734          final String name = getName(columns[i]);
735 
736          if (name == null) {
737             throw new IllegalArgumentException("column (index " + i + ") in " + 
738                   "table " + getName(component) + " does not have a name");
739          }
740 
741          propertyNames.add(name);
742       }
743 
744       final String componentName = getName(component);
745 
746       String propertyName;
747       Object row;
748       Object bean;
749       int indexOfDot;
750       boolean skip;
751 
752       for (final Iterator i = c.iterator(); i.hasNext(); ) {
753          bean = i.next();
754          row = createRow();
755 
756          for (final Iterator it = propertyNames.iterator(); it.hasNext(); ) {
757             propertyName = it.next().toString();
758 
759             indexOfDot = 0;
760             skip = false;
761 
762             while ((indexOfDot = propertyName.indexOf('.', indexOfDot)) != -1) {
763                if (PropertyUtils.getProperty(bean, 
764                      propertyName.substring(0, indexOfDot)) == null) {
765                   add(row, createCell(propertyName, 
766                         format(formatters, componentName + '.' + propertyName, 
767                         null)));
768                   skip = true;
769                   break;
770                }
771 
772                indexOfDot += 1;
773             }
774 
775             if (skip) {
776                continue;
777             }
778 
779             add(row, createCell(propertyName, format(formatters, 
780                   componentName + '.' + propertyName, 
781                   PropertyUtils.getProperty(bean, propertyName))));
782          }
783 
784          add(component, row);
785       }
786    }

802       Map formPerClass = (Map)formPerClassPerComponent.get(widget);
803 
804       if (formPerClass == null) {
805          formPerClass = new HashMap();
806          formPerClassPerComponent.put(widget, formPerClass);

809 
810          if (oldForm != null) {
811             binderPerForm.remove(oldForm);
812          }
813       }
814 
815       formPerClass.put(form.getClass(), form);

950    public void handleException(Throwable throwable) {
951       if (throwable instanceof ScreenNotFoundException) {
952          handleException("The error screen file could not be found", throwable);
953 
954          return;
955       } else if (throwable instanceof ValidationException) {
956          showValidationErrors((ValidationException)throwable);
957          return;
958       } else if (throwable instanceof UIException) {
959          handleUIException((UIException)throwable);
960          return;
961       }
962 
963       try {
964          if (handleCustomException(throwable)) {
965             return;
966          }
967       } catch (Throwable t) {
968          LogFactory.getLog(getClass()).error("Unknown exception", t);
969       }
970 
971       handleUnknownException(throwable);
972    }
