34  /***
35   * @Aspect("perJVM")
36   */
37  public class TimeoutAspect {
38     private static final Log log = LogFactory.getLog(TimeoutAspect.class);
39  
40     private static final class LiveThreadWorkerThreadPair 
41           extends PhantomReference {
42        private final WeakReference reference;
43  
44        public LiveThreadWorkerThreadPair(Thread parentThread, ReferenceQueue queue,
45              WorkerThread workerThread) {
46           super(parentThread, queue);
47           reference = new WeakReference(workerThread);
48        }
49  
50        public WorkerThread getWorkerThread() {
51           return (WorkerThread)reference.get();
52        }
53     }

55     private static final class WorkerThreadDisposer extends Thread {
56        private static WorkerThreadDisposer instance;
57        private final ReferenceQueue queue = new ReferenceQueue();
58        private final Collection references = new LinkedList();
59  
60        public WorkerThreadDisposer() {
61           super("WorkerThreadDisposer-Daemon");
62           setDaemon(true);
63           setPriority(Thread.MIN_PRIORITY);
64           start();
65        }
66  
67        public static WorkerThreadDisposer getInstance() {
68           synchronized (WorkerThreadDisposer.class) {
69              if (instance == null) {
70                 instance = new WorkerThreadDisposer();
71              }
72           }
73  
74           return instance;
75        }
76  
77        public void enqueue(Thread parentThread, WorkerThread workerThread) {
78           LiveThreadWorkerThreadPair pair = new LiveThreadWorkerThreadPair(
79                 parentThread, queue, workerThread);
80  
81           synchronized (references) {
82              references.add(pair);
83           }
84        }
85  
86        public void run() {
87           while (true) {
88              try {
89                 LiveThreadWorkerThreadPair pair = 
90                       (LiveThreadWorkerThreadPair)queue.remove();
91                 WorkerThread worker = pair.getWorkerThread();
92  
93                 if (worker != null) {
94                    worker.interrupt();
95                 }
96                 
97                 pair.clear();
98  
99                 synchronized (references) {
100                   references.remove(pair);
101                }
102             } catch (InterruptedException ie) {
103                log.error("WorkerThreadDisposed interrupted", ie);
104                return;
105             }
106          }
107       }
108    }

110    private static final class WorkerThread extends Thread {
111       private final boolean keepThreadInstance;
112       private Throwable throwable;
113       private Object returnValue;
114       private StaticJoinPoint jp;
115       private Object lock = new Object();
116       private boolean started;
117       private boolean running;
118       private boolean waiting;
119 
120       public WorkerThread(boolean keepThreadInstance) {
121          super("WorkerThread-" + Thread.currentThread().getName());
122          setDaemon(true);
123          this.keepThreadInstance = keepThreadInstance;
124       }
125 
126       public void execute(StaticJoinPoint jp, long timeout) {
127          this.jp = jp;
128          returnValue = null;
129          throwable = null;
130          running = true;
131          waiting = false;
132 
133          if (!started) {
134             started = true;
135             start();
136          } else {
137             synchronized (lock) {
138                lock.notifyAll();
139             }
140          }
141 
142          synchronized (lock) {
143             try {
144                waiting = true;
145                lock.wait(timeout);
146             } catch (InterruptedException ie) {
147                log.info(ie);
148             }
149          }
150       }
151 
152       public void run() {
153          do {
154             try {
155                returnValue = jp.proceed();
156             } catch (final Throwable t) {
157                throwable = t;
158             }
159 
160             running = false;
161             jp = null;
162 
163             while (!waiting) {
164                try {
165                   Thread.sleep(10);
166                } catch (InterruptedException ie) {
167                   if (log.isTraceEnabled()) {
168                      log.trace(ie);
169                   }
170 
171                   return;
172                }
173             }
174 
175             synchronized (lock) {
176                lock.notifyAll();
177 
178                if (!keepThreadInstance) {
179                   return;
180                }
181 
182                try {
183                   lock.wait();
184                } catch (InterruptedException ie) {
185                   if (log.isTraceEnabled()) {
186                      log.trace(ie);
187                   }
188 
189                   return;
190                }
191             }
192          } while (true);
193       }
194 
195       public Object getReturnValue() {
196          return returnValue;
197       }
198 
199       public Throwable getThrowable() {
200          return throwable;
201       }
202 
203       public boolean isRunning() {
204          return running;
205       }
206 
207       public void cleanUp() {
208          returnValue = null;
209          throwable = null;
210       }
211    }

213    private final long timeout;
214    private final boolean keepThreadInstance;
215    private final ThreadLocal threadLocal = new ThreadLocal();
216    
217    public TimeoutAspect(final AspectContext ctx) {
218       this.timeout = Long.parseLong(ctx.getParameter("timeout"));
219       this.keepThreadInstance = "true".equals(
220             ctx.getParameter("keepThreadInstance"));
221    }
222 
223    private WorkerThread getWorkerThread() {
224       return (WorkerThread)threadLocal.get();
225    }
226 
227    private void setWorkerThread(WorkerThread workerThread) {
228       threadLocal.set(workerThread);
229    }

231    /***
232     * @Around("timeout")
233     */
234    public Object timeoutAdvice(final StaticJoinPoint jp) throws Throwable {
235       WorkerThread thread = getWorkerThread();
236 
237       if (thread == null) {
238          thread = new WorkerThread(keepThreadInstance);
239          setWorkerThread(thread);
240 
241          if (keepThreadInstance) {
242             if (log.isDebugEnabled()) {
243                log.debug("New thread created: " + thread);
244             }
245 
246             WorkerThreadDisposer.getInstance().enqueue(Thread.currentThread(), 
247                   thread);
248          }
249       } else {
250          if (keepThreadInstance && log.isDebugEnabled()) {
251             log.debug("Reusing thread: " + thread);
252          }
253       }
254 
255       thread.execute(jp, timeout);
256 
257       if (thread.isRunning()) {
258          try {
259             thread.interrupt();
260          } catch (Throwable t) {
261             log.error(t);
262          }
263 
264          setWorkerThread(null);
265 
266          throw new TimeoutException("Execution took more than " + timeout + 
267                " ms");
268       }
269  
270       final Throwable throwable = thread.getThrowable();
271       final Object returnValue = thread.getReturnValue();
272 
273       thread.cleanUp();
274 
275       if (!keepThreadInstance) {
276          setWorkerThread(null);
277       }
278 
279       if (throwable != null) {
280          throw throwable;
281       }
282 
283       return returnValue;
284    }
