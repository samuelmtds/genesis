75     public abstract class AbstractBoundMember implements BoundMember {
76        private final Widget widget;
77        private final SWTBinder binder;
78        private final Set enabledWidgetGroupSet = new HashSet();
79        private final Set visibleWidgetGroupSet = new HashSet();
80  
81        public AbstractBoundMember(SWTBinder binder, Widget widget) {
82           this.widget = widget;
83           this.binder = binder;
84  
85           createGroups();
86        }
87  
88        protected SWTBinder getBinder() {
89           return binder;
90        }
91  
92        protected Set getEnabledWidgetGroupSet() {
93           return enabledWidgetGroupSet;
94        }
95  
96        protected Set getVisibleWidgetGroupSet() {
97           return visibleWidgetGroupSet;
98        }
99  
100       protected void createGroups() {
101          createWidgetGroup();
102          createEnabledGroup();
103          createVisibleGroup();
104       }
105 
106       protected void createWidgetGroup() {
107          createGroup(widget
108                .getData(SWTBinder.WIDGET_GROUP_PROPERTY), true, true);
109       }
110 
111       protected void createEnabledGroup() {
112          createGroup(widget
113                .getData(SWTBinder.ENABLED_GROUP_PROPERTY), true,
114                false);
115       }
116 
117       protected void createVisibleGroup() {
118          createGroup(widget
119                .getData(SWTBinder.VISIBLE_GROUP_PROPERTY), false,
120                true);
121       }

123       protected void createGroup(Object group, boolean enabled, boolean visible) {
124          if (group == null) {
125             return;
126          }
127 
128          boolean instanceofString = group instanceof String;
129          if (instanceofString || group instanceof String[]) {
130         	
131             String[] widgetNames = instanceofString ?
132             		((String) group).split("//s*,//s*") : (String[]) group;
133 
134             for (int i = 0; i < widgetNames.length; i++) {
135                Widget w = (Widget) binder.lookup(widgetNames[i]);
136 
137                if (w != null) {
138                   if (enabled) {
139                      enabledWidgetGroupSet.add(w);
140                   }
141 
142                   if (visible) {
143                      visibleWidgetGroupSet.add(w);
144                   }
145                }
146             }
147          } else if (group instanceof Collection) {
148             Collection groupCollection = (Collection) group;
149 
150             if (enabled) {
151                enabledWidgetGroupSet.addAll(groupCollection);
152             }
153 
154             if (visible) {
155                visibleWidgetGroupSet.addAll(groupCollection);
156             }
157          } else if (group instanceof Object[]) {
158             Object[] groupArray = (Object[]) group;
159 
160             if (enabled) {
161                enabledWidgetGroupSet.addAll(Arrays.asList(groupArray));
162             }
163 
164             if (visible) {
165                visibleWidgetGroupSet.addAll(Arrays.asList(groupArray));
166             }
167          } else if (group instanceof Widget) {
168             Widget widget = (Widget) group;
169 
170             if (enabled) {
171                enabledWidgetGroupSet.add(widget);
172             }
173 
174             if (visible) {
175                visibleWidgetGroupSet.add(widget);
176             }
177          } else {
178             throw new IllegalArgumentException("Group property must be a comma-separated string, array of strings, a collection of widgets, an array of widgets or a Widget");
179          }
180       }
181 
182       protected boolean isBlank(Widget widget) {
183          return isBoolean(widget, SWTBinder.BLANK_PROPERTY);
184       }
185 
186       protected boolean isBoolean(Widget widget, String propertyName) {
187          final Object booleanObject = widget.getData(propertyName);
188 
189          boolean ret;
190 
191          if (booleanObject == null) {
192             ret = false;
193          } else if (booleanObject instanceof String) {
194             ret = Boolean.valueOf(booleanObject.toString()).booleanValue();
195          } else if (booleanObject instanceof Boolean) {
196             ret = ((Boolean) booleanObject).booleanValue();
197          } else {
198             throw new PropertyMisconfigurationException("Property '" +
199                propertyName + "' " + "for the widget named " +
200                getName() + " must " +
201                "either be left empty or contain a boolean value");
202          }
203 
204          return ret;
205       }
206 
207       public void setEnabled(boolean enabled) {
208          ((Control)widget).setEnabled(enabled);
209 
210          for (Iterator iter = enabledWidgetGroupSet.iterator(); iter.hasNext();) {
211             ((Control) iter.next()).setEnabled(enabled);
212          }
213       }
214 
215       public void setVisible(boolean visible) {
216          ((Control)widget).setVisible(visible);
217 
218          for (Iterator iter = visibleWidgetGroupSet.iterator(); iter.hasNext();) {
219             ((Control) iter.next()).setVisible(visible);
220          }
221       }

223       protected Object getProperty(Object bean, String propertyName)
224             throws IllegalAccessException, InvocationTargetException {
225          try {
226             return PropertyUtils.getProperty(bean, propertyName);
227          } catch (NoSuchMethodException e) {
228             IllegalArgumentException iae = new IllegalArgumentException(
229                   "The widget named '" + getName() + "' expected "  + 
230                   bean.getClass().getName() + " to have a property named '" + 
231                   propertyName + "' (at bean " + bean + ")");
232             iae.initCause(e);
233             throw iae;
234          }
235       }
236 
237       public String getName() {
238          return binder.getLookupStrategy().getName(widget);
239       }

242       }
243    }
