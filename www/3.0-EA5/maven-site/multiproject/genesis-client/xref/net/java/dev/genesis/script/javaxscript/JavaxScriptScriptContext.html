19  package net.java.dev.genesis.script.javaxscript;
20  
21  import java.lang.reflect.InvocationTargetException;
22  import java.util.Collections;
23  import java.util.Map;
24  
25  import net.java.dev.genesis.script.ScriptContext;
26  import net.java.dev.genesis.script.ScriptException;
27  import net.java.dev.genesis.script.ScriptExpression;
28  import net.java.dev.genesis.script.ScriptFunctionsAdapter;
29  import net.java.dev.genesis.script.javaxscript.bridge.JavaxScriptBridge;
30  import net.java.dev.genesis.script.javaxscript.bridge.ScriptEngine;
31  
32  public class JavaxScriptScriptContext extends ScriptContext {
33     public static final String FORM_NS = "form";
34     public static final String GENESIS_FUNCTIONS_NS = "genesis";
35  
36     private final ScriptEngine scriptEngine;
37     private final Map unmodifiableMap;
38     private Object realContext;
39  
40     protected JavaxScriptScriptContext(ScriptEngine engine, Object root) {
41        this.scriptEngine = engine;
42        this.realContext = engine.getContext().getRealContext();
43        this.unmodifiableMap = Collections.unmodifiableMap(scriptEngine.getContext().getBindings(
44              JavaxScriptBridge.getInstance().getEngineScope()));
45        declare(FORM_NS, root);
46        registerFunctions(GENESIS_FUNCTIONS_NS, getFunctions());
47     }
48     
49     protected ScriptEngine getScriptEngine() {
50        return scriptEngine;
51     }
52  
53     protected Object doEval(ScriptExpression expr) {
54        try {
55           return ((JavaxScriptExpression) expr).evalCompiled(scriptEngine,
56                 realContext);
57        } catch (InvocationTargetException ite) {
58           throw new ScriptException(ite.getTargetException().getMessage(), ite.getTargetException());
59        } catch (Exception e) {
60           throw new ScriptException(e.getMessage(), e);
61        }
62     }
63  
64     protected ScriptExpression newScriptExpression(String expression) {
65        return new JavaxScriptExpression(expression, scriptEngine);
66     }
67  
68     public void registerFunctions(String prefix, Class functionClass) {
69        try {
70           scriptEngine.getContext().setAttribute(prefix,
71                 functionClass.newInstance(),
72                 JavaxScriptBridge.getInstance().getGlobalScope());
73        } catch (Exception e) {
74           throw new ScriptException(e.getMessage(), e);
75        }
76     }
77  
78     public void declare(String name, Object value) {
79        scriptEngine.getContext().setAttribute(name, value,
80              JavaxScriptBridge.getInstance().getEngineScope());
81     }
82  
83     public Map getContextMap() {
84        return unmodifiableMap;
85     }
86  
87     public Object lookup(String name) {
88        return scriptEngine.getContext().getAttribute(name);
89     }
90  
91     public void undeclare(String name) {
92        scriptEngine.getContext().removeAttribute(name,
93              JavaxScriptBridge.getInstance().getEngineScope());
94     }
95  
96     protected Class getFunctions() {
97        return ScriptFunctionsAdapter.class;
98     }
99  }
