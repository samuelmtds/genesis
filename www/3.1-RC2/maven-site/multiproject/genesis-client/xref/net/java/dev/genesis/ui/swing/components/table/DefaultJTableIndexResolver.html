21  import java.lang.reflect.InvocationTargetException;
22  import java.lang.reflect.Method;
23  import java.util.Map;
24  import java.util.WeakHashMap;
25  import javax.swing.JComponent;
26  import net.java.dev.genesis.registry.Registry;

28  public class DefaultJTableIndexResolver implements JTableIndexResolver {
29     protected static class JTableIndexResolverInfo {
30        private Method convertRowIndexToModel;
31        private Method convertRowIndexToView;
32  
33        JTableIndexResolverInfo(final Class clazz) {
34           try {
35              convertRowIndexToModel = clazz.getMethod(
36                    "convertRowIndexToModel", new Class[] {int.class});
37              convertRowIndexToView = clazz.getMethod(
38                    "convertRowIndexToView", new Class[] {int.class});
39           } catch (SecurityException se) {
40              IllegalStateException ise = new IllegalStateException(
41                    "Could not load convertRowIndexTo* methods");
42              ise.initCause(se);
43              throw ise;
44           } catch (NoSuchMethodException nsme) {
45              // Expected
46           }
47        }
48  
49        public boolean needsConversion() {
50           return convertRowIndexToModel != null;
51        }
52  
53        public int convertRowIndexToModel(final JComponent component,
54              final int index) throws IllegalAccessException,
55              IllegalArgumentException, InvocationTargetException {
56           return ((Integer)convertRowIndexToModel.invoke(component,
57                 new Object[] {new Integer(index)})).intValue();
58        }
59  
60        public int convertRowIndexToView(final JComponent component,
61              final int index) throws IllegalAccessException,
62              IllegalArgumentException, InvocationTargetException {
63           return ((Integer)convertRowIndexToView.invoke(component,
64                 new Object[] {new Integer(index)})).intValue();
65        }
66     }
67  
68     private static class JTableIndexResolverInfoRegistry {
69        private final Registry registry = new Registry();
70  
71        public void deregister() {
72           registry.deregister();
73        }
74  
75        public void deregister(Class clazz) {
76           registry.deregister(clazz);
77        }
78  
79        public JTableIndexResolverInfo register(Class clazz,
80              JTableIndexResolverInfo info) {
81           return (JTableIndexResolverInfo)registry.register(clazz, info);
82        }
83  
84        public JTableIndexResolverInfo get(Class clazz) {
85           return (JTableIndexResolverInfo)registry.get(clazz);
86        }
87  
88        public JTableIndexResolverInfo get(Class clazz, boolean superClass) {
89           return (JTableIndexResolverInfo)registry.get(clazz, superClass);
90        }
91  
92        public JTableIndexResolverInfo get(Object o) {
93           return (JTableIndexResolverInfo)registry.get(o);
94        }
95     }
96     private final Map resolverInfoMap = new WeakHashMap();
97     private final JTableIndexResolverInfoRegistry resolverInfoRegistry =
98           new JTableIndexResolverInfoRegistry();
99  
100    protected JTableIndexResolverInfo initialize(final Class clazz) {
101       JTableIndexResolverInfo info = resolverInfoMap.get(clazz) != null ? (JTableIndexResolverInfo)resolverInfoMap.get(clazz)
102             : resolverInfoRegistry.get(clazz, true);
103       if (info == null) {
104          info = new JTableIndexResolverInfo(clazz);
105          if (info.needsConversion()) {
106             resolverInfoRegistry.register(clazz, info);
107          } else {
108             resolverInfoMap.put(clazz, info);
109          }
110       }
111 
112       return info;
113    }
114 
115    public boolean needsConversion(JComponent component) {
116       return initialize(component.getClass()).needsConversion();
117    }
118 
119    public int convertRowIndexToModel(JComponent component, int index) throws IllegalAccessException,
120          IllegalArgumentException, InvocationTargetException {
121       return initialize(component.getClass()).convertRowIndexToModel(component,
122             index);
123    }
124 
125    public int convertRowIndexToView(JComponent component, int index) throws IllegalAccessException,
126          IllegalArgumentException, InvocationTargetException {
127       return initialize(component.getClass()).convertRowIndexToView(component,
128             index);
129    }
