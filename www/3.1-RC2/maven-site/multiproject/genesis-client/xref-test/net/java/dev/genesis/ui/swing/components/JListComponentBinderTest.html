226    public void testUpdateList() throws Exception {
227       assertNull(componentBinder.bind(binder, list, (ActionMetadata) null));
228       assertNull(componentBinder.bind(binder, list, (FieldMetadata) null));
229 
230       boundDataProvider = componentBinder.bind(binder, list, dataMeta);
231       assertNotNull(boundDataProvider);
232 
233       String[] newList = new String[] { "newOne", "newTwo", "newThree" };
234       boundDataProvider.updateList(Arrays.asList(newList));
235       int count = list.getModel().getSize();
236       assertEquals(-1, list.getSelectedIndex());
237       assertEquals(count, newList.length);
238       for (int i = 0; i < newList.length; i++) {
239          assertEquals(list.getModel().getElementAt(i), newList[i]);
240       }
241 
242       list.setSelectedIndices(new int[] {0, 1, 2});
243       boundDataProvider.updateList(Arrays.asList(newList));
244       count = list.getModel().getSize();
245       assertEquals(0, list.getSelectedIndices().length);
246       assertEquals(count, newList.length);
247       for (int i = 0; i < newList.length; i++) {
248          assertEquals(list.getModel().getElementAt(i), newList[i]);
249       }
250 
251       dataMeta.setResetSelection(false);
252       list.setSelectedIndices(new int[] {0, 1, 2});
253       boundDataProvider.updateList(Arrays.asList(newList));
254       count = list.getModel().getSize();
255       assertTrue(Arrays.equals(new int[] {0, 1, 2}, list.getSelectedIndices()));
256       assertEquals(count, newList.length);
257       for (int i = 0; i < newList.length; i++) {
258          assertEquals(list.getModel().getElementAt(i), newList[i]);
259       }
260 
261       newList = new String[] { "other" };
262       boundDataProvider.updateList(Arrays.asList(newList));
263       count = list.getModel().getSize();
264       assertTrue(Arrays.equals(new int[] {0}, list.getSelectedIndices()));
265       assertEquals(count, newList.length);
266       for (int i = 0; i < newList.length; i++) {
267          assertEquals(list.getModel().getElementAt(i), newList[i]);
268       }
269 
270       newList = new String[0];
271       boundDataProvider.updateList(Arrays.asList(newList));
272       count = list.getModel().getSize();
273       assertEquals(count, newList.length);
274    }
