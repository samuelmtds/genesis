105    protected String getKey(Object value) throws Exception {
106       String keyPropertyName = (String) list.getData(SWTBinder.KEY_PROPERTY);
107 
108       if (keyPropertyName != null) {
109          Object o = (value == null) ? null : PropertyUtils.getProperty(value,
110                keyPropertyName);
111 
112          return binder.format(getName(), keyPropertyName, o);
113       } else if (EnumHelper.getInstance().isEnum(value)) {
114          return value.toString();
115       }
116 
117       return String.valueOf(System.identityHashCode(value));
118    }
119 
120    protected String[] getValues(Widget widget, Object[] values) throws Exception {
121       String[] array = new String[values.length];
122       
123       for (int i = 0; i < values.length; i++) {
124          array[i] = getValue(widget, values[i]);
125       }
126       
127       return array;
128    }
129 
130    protected String getValue(Widget widget, Object value) throws Exception {
131       String valueProperty = (String) widget.getData(SWTBinder.VALUE_PROPERTY);
132       if (value instanceof String) {
133          return (String) value;
134       } else if (valueProperty == null) {
135          return binder.format(binder.getLookupStrategy().getName(widget), null,
136                value);
137       }
138 
139       return binder.format(binder.getLookupStrategy().getName(widget),
140             valueProperty, PropertyUtils.getProperty(value, valueProperty));
141    }
