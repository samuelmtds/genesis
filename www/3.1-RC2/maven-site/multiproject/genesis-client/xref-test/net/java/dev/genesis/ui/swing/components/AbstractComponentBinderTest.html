37  public class AbstractComponentBinderTest extends GenesisTestCase {
38     private MockForm form;
39     private JTextField comp1;
40     private JTextField comp2;
41     private MockSwingBinder binder;
42     private AbstractComponentBinder componentBinder;
43     private AbstractBoundMember boundMember;
44     private ActionMetadata actionMeta;

46     public AbstractComponentBinderTest() {
47        super("Abstract Component Binder Unit Test");
48     }

50     protected void setUp() throws Exception {
51        form = new MockForm();
52        comp1 = new JTextField();
53        comp2 = new JTextField();
54  
55        binder = new MockSwingBinder(new JPanel(), form);
56        comp1.putClientProperty(SwingBinder.ENABLED_GROUP_PROPERTY, comp2);
57        comp1.putClientProperty(SwingBinder.VISIBLE_GROUP_PROPERTY, comp2);
58  
59        componentBinder = new AbstractComponentBinder() {
60        };
61        boundMember = componentBinder.new AbstractBoundMember(binder, comp1) {
62        };
63        actionMeta = (ActionMetadata) form.getFormMetadata().getActionMetadatas().get("someAction");
64     }
65  
66     public void testEnabled() {
67        assertTrue(comp1.isEnabled());
68        assertTrue(comp2.isEnabled());
69  
70        boundMember.setEnabled(false);
71  
72        assertFalse(comp1.isEnabled());
73        assertFalse(comp2.isEnabled());
74     }
75  
76     public void testVisible() {
77        assertTrue(comp1.isVisible());
78        assertTrue(comp2.isVisible());
79  
80        boundMember.setVisible(false);
81  
82        assertFalse(comp1.isVisible());
83        assertFalse(comp2.isVisible());
84     }
85  
86     public void testSimpleActionBinding() {
87        assertNotNull(componentBinder.bind(binder, comp1, (ActionMetadata) 
88              form.getFormMetadata().getActionMetadatas().get("someAction")));
89        comp1.postActionEvent();
90        assertSame(actionMeta, binder.get("invokeFormAction(ActionMetadata)"));
91     }
92     
93     public void testMultipleActions() {
94        assertNull(componentBinder.bind(binder, new InvalidMultipleActions(), 
95              (ActionMetadata)form.getFormMetadata().getActionMetadatas()
96              .get("someAction")));
97        
98        ValidMultipleActions valid = new ValidMultipleActions();
99        assertNotNull(componentBinder.bind(binder, valid, (ActionMetadata) 
100             form.getFormMetadata().getActionMetadatas().get("someAction")));
101 
102       binder.put("invokeFormAction(ActionMetadata)", null);
103       valid.fireActionOneEvent();
104       assertNull(binder.get("invokeFormAction(ActionMetadata)"));
105 
106       valid.fireActionEvent();
107       assertSame(actionMeta, binder.get("invokeFormAction(ActionMetadata)"));
108    }
