35  public class ButtonWidgetBinderTest extends GenesisTestCase {
36     private Shell root;
37     private Button button;
38     private MockSWTBinder binder;
39     private WidgetBinder widgetBinder;
40     private MockForm form;
41     private ActionMetadata actionMeta;
42  
43     public ButtonWidgetBinderTest() {
44        super("Button Component Binder Unit Test");
45     }
46  
47     protected void setUp() throws Exception {
48        button = new Button(root = new Shell(), SWT.NONE);
49        binder = new MockSWTBinder(root, form = new MockForm(), new Object());
50        actionMeta = (ActionMetadata) form.getFormMetadata().getActionMetadatas()
51              .get("someAction");
52        widgetBinder = binder.getWidgetBinder(button);
53     }
54  
55     public void testButtonClick() {
56        assertTrue(widgetBinder instanceof ButtonWidgetBinder);
57  
58        assertNull(widgetBinder.bind(binder, button, (DataProviderMetadata) null));
59        assertNull(widgetBinder.bind(binder, button, (FieldMetadata) null));
60        assertNotNull(widgetBinder.bind(binder, button, actionMeta));
61  
62        simulateClick(button);
63        assertSame(actionMeta, binder.get("invokeFormAction(ActionMetadata)"));
64     }
65  
66     private void simulateClick(Button button) {
67        Event event = new Event();
68        event.widget = button;
69        event.button = 1;
70        event.type = SWT.Selection;
71        button.notifyListeners(event.type, event);
72     }
73  }
