19  package net.java.dev.genesis.ui.metadata;
20  
21  import java.lang.reflect.Method;
22  import java.util.HashMap;
23  import java.util.Iterator;
24  import java.util.Map;
25  
26  import net.java.dev.genesis.reflection.MethodEntry;
27  
28  import org.apache.commons.jxpath.CompiledExpression;
29  
30  public class FormMetadata {
31     private final Class formClass;
32  
33     private final Map namedConditions;
34     private final Map fieldMetadatas;
35     private final Map actionMetadatas;
36     private final Map dataProviderMetadatas;
37  
38     public FormMetadata(final Class formClass) {
39        this.formClass = formClass;
40        this.namedConditions = new HashMap();
41        this.fieldMetadatas = new HashMap();
42        this.actionMetadatas = new HashMap();
43        this.dataProviderMetadatas = new HashMap();
44     }
45  
46     public Class getFormClass() {
47        return formClass;
48     }
49  
50     public Map getFieldMetadatas() {
51        return new HashMap(fieldMetadatas);
52     }
53  
54     public Map getNamedConditions() {
55        return new HashMap(namedConditions);
56     }
57  
58     public Map getActionMetadatas(){
59        return new HashMap(actionMetadatas);
60     }
61     
62     public Map getDataProviderMetadatas(){
63        return new HashMap(dataProviderMetadatas);
64     }
65  
66     public CompiledExpression getNamedCondition(final String namedConditionName) {
67        return (CompiledExpression) namedConditions.get(namedConditionName);
68     }
69  
70     public FieldMetadata getFieldMetadata(final String fieldName) {
71        return (FieldMetadata) fieldMetadatas.get(fieldName);
72     }
73     
74     public ActionMetadata getActionMetadata(final Method method) {
75        return (ActionMetadata) actionMetadatas.get(new MethodEntry(method));
76     }
77     
78     public ActionMetadata getActionMetadata(final String methodName) {
79        return (ActionMetadata) actionMetadatas.get(new MethodEntry(methodName, new String[0]));
80     }
81     
82     public DataProviderMetadata getDataProviderMetadata(final Method method) {
83        return (DataProviderMetadata) dataProviderMetadatas.get(new MethodEntry(method));
84     }
85  
86     public void addNamedCondition(final String key, final CompiledExpression value) {
87        namedConditions.put(key, value);
88     }
89  
90     public void addFieldMetadata(final String fieldName,
91           final FieldMetadata fieldMetadata) {
92        fieldMetadatas.put(fieldName, fieldMetadata);
93     }
94     
95     public void addActionMetadata(final Method method, final ActionMetadata actionMetadata) {
96        actionMetadatas.put(new MethodEntry(method), actionMetadata);
97     }
98     
99     public void addDataProviderMetadata(final Method method, final DataProviderMetadata dataProviderMetadata) {
100       dataProviderMetadatas.put(new MethodEntry(method), dataProviderMetadata);
101    }
102 
103    public String toString() {
104       StringBuffer buffer = new StringBuffer();
105       buffer.append(getClass().getName());
106       buffer.append(" = {\n");
107 
108       for (Iterator iter = namedConditions.entrySet().iterator(); iter
109             .hasNext();) {
110          Map.Entry element = (Map.Entry) iter.next();
111          buffer.append("\t");
112          buffer.append(element.getKey());
113          buffer.append("=");
114          buffer.append(element.getValue());
115          buffer.append("\n");
116       }
117 
118       for (Iterator iter = fieldMetadatas.values().iterator(); iter.hasNext();) {
119          buffer.append("\t");
120          buffer.append(iter.next());
121          buffer.append("\n");
122       }
123       
124       for (Iterator iter = actionMetadatas.values().iterator(); iter.hasNext();) {
125          buffer.append("\t");
126          buffer.append(iter.next());
127          buffer.append("\n");
128       }
129       
130       for (Iterator iter = dataProviderMetadatas.values().iterator(); iter.hasNext();) {
131          buffer.append("\t");
132          buffer.append(iter.next());
133          buffer.append("\n");
134       }
135 
136       buffer.append("}");
137 
138       return buffer.toString();
139    }
