136                                                 throws ScreenNotFoundException {
137       try {
138          InputStream is = getResourceAsStream(fileName, 
139                                              handler.getClass().getClassLoader());
140 
141          if (is == null) {
142             throw new ScreenNotFoundException(fileName);
143          }
144          
145          handler.setScreen(parse(is, handler));
146 
147          is.close();
148 
149          return handler.getScreen();
150       } catch (IOException ioe) {
151          throw new ScreenNotFoundException(fileName);
152       }
153    }

346    protected void displayBean(Object bean, Object root) throws IllegalAccessException,
347                               InvocationTargetException, NoSuchMethodException {
348       final Map properties = bean instanceof Map ? (Map)bean : 
349             PropertyUtils.describe(bean);
350 
351       if (root instanceof Thinlet) {
352          root = ((Thinlet)root).getDesktop();
353       }
354 
355       String propertyName;
356       String propertyValue = null;
357       String type;
358       Object component;
359       Collection groupComponents = null;
360 
361       for (final Iterator propertyNames = properties.keySet().iterator();
362                                                    propertyNames.hasNext(); ) {
363          propertyName = propertyNames.next().toString();
364          propertyValue = null;
365          component = find(root, propertyName);
366 
367          if (component == null) {
368             if (groupComponents == null) {
369                groupComponents = getAllOfClass(root, CHECKBOX);
370             }
371 
372             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
373                component = i.next();
374 
375                if (propertyName.equals(getGroup(component)) && 
376                       getName(component) != null ) {
377                   if (propertyValue == null) {
378                      propertyValue = getPropertyValue(properties, propertyName, true);
379                   }
380 
381                   if (getName(component).equals(propertyValue)) {
382                      setSelected(component, true);
383                      break;
384                   }
385                }
386             }
387 
388             continue;
389          }
390 
391          type = getClass(component);
392 
393          if (type.equals(COMBOBOX) || type.equals(LIST)) {
394             Object selectedComponent = getSelectedItem(component);
395 
396             if (selectedComponent != null && type.equals(LIST)) {
397                setSelected(selectedComponent, false);
398             }
399 
400             if (type.equals(COMBOBOX)) {
401                setSelected(component, -1);
402                setText(component, "");
403             }
404 
405             String valuePropertyName = (String)getProperty(component, "value");
406 
407             if (valuePropertyName != null) {
408                Object o = properties.get(propertyName);
409                propertyValue = FormatterRegistry.getInstance().format(
410                      o == null ? null : PropertyUtils.getProperty(
411                      properties.get(propertyName), valuePropertyName));
412             } else {
413                propertyValue = getPropertyValue(properties, propertyName, true);
414             }
415 
416             selectedComponent = find(component, propertyValue);
417 
418             if (selectedComponent != null) {
419                if (type.equals(LIST)) {
420                   setSelected(selectedComponent, true);
421                } else {
422                   setSelected(component, getIndexOf(component, selectedComponent));
423                }
424             }
425          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
426             setValue(component, getPropertyValue(properties, propertyName, false));
427          } else if (type.equals(PANEL)) {
428             displayBean(properties.get(propertyName), component);
429          } else if (!type.equals(TABLE)){
430             setText(component, getPropertyValue(properties, propertyName, false));
431          }
432       }
433    }

452    protected void populate(Form bean, Object root, final Map properties) 
453          throws IllegalAccessException, InvocationTargetException, NoSuchMethodException {
454       properties.putAll(BeanUtils.describe(bean));
455 
456       if (root instanceof Thinlet) {
457          root = ((Thinlet)root).getDesktop();
458       }
459 
460       String propertyName;
461       String type;
462       Object component;
463       Collection groupComponents = null;
464 
465       for (final Iterator propertyNames = properties.keySet().iterator();
466                                                    propertyNames.hasNext(); ) {
467          propertyName = propertyNames.next().toString();
468          component = find(root, propertyName);
469 
470          if (component == null) {
471             if (groupComponents == null) {
472                groupComponents = getAllOfClass(root, CHECKBOX);
473             }
474 
475             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
476                component = i.next();
477 
478                if (propertyName.equals(getGroup(component)) && 
479                      isSelected(component)) {
480                   properties.put(propertyName, getName(component));
481                   break;
482                }
483             }
484 
485             continue;
486          }
487 
488          type = getClass(component);
489 
490          if (type.equals(COMBOBOX) || type.equals(LIST)) {
491             component = getSelectedItem(component);
492 
493             if (component != null) {
494                properties.put(propertyName, getName(component));
495             } else {
496                properties.put(propertyName, null);
497             }
498          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
499             properties.put(propertyName, getValue(component));
500          } else {
501             properties.put(propertyName, getText(component));
502          }
503       }
504 
505       BeanUtils.populate(bean, properties);
506       bean.afterPopulate();
507    }

610    protected void populateFromCollection(Object component, Collection c)
611          throws IllegalAccessException, InvocationTargetException, 
612                 NoSuchMethodException {
613       if (!getClass(component).equals(TABLE)) {
614          throw new UnsupportedOperationException();
615       }
616 
617       removeAll(component);
618       final Collection propertyNames = new ArrayList();
619       final Object[] columns = getTableColumns(component); 
620       
621       for (int i = 0; i < columns.length; i++) {
622          final String name = getName(columns[i]);
623 
624          if (name == null) {
625             throw new IllegalArgumentException("column (index " + i + ") in " + 
626                   "table " + getName(component) + " does not have a name");
627          }
628 
629          propertyNames.add(name);
630       }
631 
632       String propertyName;
633       Object row;
634       Object bean;
635       int indexOfDot;
636       boolean skip;
637 
638       for (final Iterator i = c.iterator(); i.hasNext(); ) {
639          bean = i.next();
640          row = createRow();
641 
642          for (final Iterator it = propertyNames.iterator(); it.hasNext(); ) {
643             propertyName = it.next().toString();
644 
645             indexOfDot = 0;
646             skip = false;
647 
648             while ((indexOfDot = propertyName.indexOf('.', indexOfDot)) != -1) {
649                if (PropertyUtils.getProperty(bean, 
650                      propertyName.substring(0, indexOfDot)) == null) {
651                   add(row, createCell(propertyName, 
652                         FormatterRegistry.getInstance().format(null)));
653                   skip = true;
654                   break;
655                }
656 
657                indexOfDot += 1;
658             }
659 
660             if (skip) {
661                continue;
662             }
663 
664             add(row, createCell(propertyName, 
665                     FormatterRegistry.getInstance().format(
666                            PropertyUtils.getProperty(bean, propertyName))));
667          }
668 
669          add(component, row);
670       }
671    }

681    protected void bind(Object widget, Object form) throws Exception {
682       Map formPerClass = (Map)formPerClassPerComponent.get(widget);
683 
684       if (formPerClass == null) {
685          formPerClass = new HashMap();
686          formPerClassPerComponent.put(widget, formPerClass);
687       } else {
688          Object oldForm = formPerClass.get(form.getClass());
689 
690          if (oldForm != null) {
691             binderPerForm.remove(oldForm);
692          }
693       }
694 
695       formPerClass.put(form.getClass(), form);
696 
697       final ThinletBinder binder = createBinder(widget, form);
698       binderPerForm.put(form, binder);
699 
700       binder.bind();
701    }

709    public void invokeFormAction(String actionName) throws Exception {
710       final Map formPerClass = (Map)formPerClassPerComponent.get(getDesktop());
711 
712       if (formPerClass == null) {
713          throw new IllegalStateException("No form is currently bound to desktop");
714       }
715 
716       final Collection forms = formPerClass.values();
717 
718       if (forms.size() > 1) {
719          throw new UnsupportedOperationException("More than one form is " +
720                "bound to desktop");
721       }
722 
723       invokeFormAction(forms.iterator().next(), actionName);
724    }

726    public void invokeFormAction(Object form, String actionName) throws Exception {
727       final ThinletBinder binder = (ThinletBinder)binderPerForm.get(form);
728 
729       if (binder == null) {
730          throw new IllegalArgumentException(form + " is not bound");
731       }
732 
733       binder.invokeAction(actionName);
734    }

745       }

748          refreshViewFromForm(i.next());

766    protected int[] getSelectedIndexes(Object component) {
767       if (COMBOBOX.equals(getClass(component))) {
768          return new int[] {getSelected(component)};
769       }
770 
771       final int selectedCount = getSelectedItems(component).length;
772 
773       final Object[] items = getItems(component);
774       final int[] indexes = new int[selectedCount];
775 
776       int j = 0;
777 
778       for (int i = 0; i < items.length; i++) {
779          if (isSelected(items[i])) {
780             indexes[j] = i;
781             j++;
782          }
783       }
784 
785       return indexes;
786    }

800       if (throwable instanceof ScreenNotFoundException) {
801          handleException("The error screen file could not be found", throwable);
802 
803          return;
804       } else if (throwable instanceof ValidationException) {
805          showValidationErrors((ValidationException)throwable);
806          return;
807       } else if (throwable instanceof UIException) {
808          handleUIException((UIException)throwable);
809          return;

811

813          if (handleCustomException(throwable)) {
814             return;
815          }
816       } catch (Throwable t) {
817          LogFactory.getLog(getClass()).error("Unknown exception", t);
818       }
819 
820       handleUnknownException(throwable);
821    }
