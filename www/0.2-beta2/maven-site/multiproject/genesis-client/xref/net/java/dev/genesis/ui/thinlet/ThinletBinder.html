87     public void bind() throws Exception {
88        final FormMetadata formMetadata = getFormMetadata(form);
89        controller.setForm(form);
90        controller.setFormMetadata(formMetadata);
91        controller.addFormControllerListener(this);
92  
93        final Collection dataProviders = new ArrayList(formMetadata
94              .getDataProviderMetadatas().values());
95        dataProviders.addAll(formMetadata.getActionMetadatas().values());
96  
97        bindFieldMetadatas(formMetadata);
98        bindActionMetadatas(formMetadata);
99        bindDataProviders(dataProviders);
100 
101       controller.setup();
102    }

104    protected FormMetadata getFormMetadata(final Object form) {
105       if (!(form instanceof FormMetadataFactory)) {
106          throw new IllegalArgumentException(form + " should implement " + 
107                "FormMetadataFactory; probably it should have been annotated " +
108                "with @Form or your aop.xml/weaving process should be " +
109                "properly configured.");
110       }
111 
112       return ((FormMetadataFactory)form).getFormMetadata(form.getClass());
113    }

115    protected void bindFieldMetadatas(final FormMetadata formMetadata) {
116       for (final Iterator i = formMetadata.getFieldMetadatas().entrySet()
117             .iterator(); i.hasNext(); ) {
118          final Map.Entry entry = (Map.Entry)i.next();
119          final String name = entry.getKey().toString();
120          final List components = findComponents(name);
121 
122          if (components.isEmpty()) {
123             log.warn(name + " could not be found while binding " + form.getClass());
124             continue;
125          }
126 
127          final Object mainComponent = components.get(0);
128          final String className = Thinlet.getClass(mainComponent);
129          final FieldMetadata fieldMetadata = (FieldMetadata)entry.getValue();
130 
131          if (Arrays.binarySearch(supportedFieldWidgets, className) < 0) {
132             if (!fieldMetadata.isDisplayOnly()) {
133                log.warn(name + " is not represented by an editable widget (" + 
134                      className + ")");
135             }
136 
137             continue;
138          }
139 
140          if (fieldMetadata.isDisplayOnly()) {
141             if (log.isDebugEnabled()) {
142                log.debug(name + " is a display only field, skipping...");
143             }
144             
145             continue;
146          }
147 
148          if (log.isTraceEnabled()) {
149             log.trace("Binding field " + name + " in " + 
150                   form.getClass().getName() + " to " + className);
151          }
152 
153          boundFieldNames.add(name);
154 
155          createWidgetGroup(mainComponent, name);
156 
157          for (final Iterator it = components.iterator(); it.hasNext(); ) {
158             final Object component = it.next();
159 
160             if (className.equals(BaseThinlet.CHECKBOX)) {
161                thinlet.setMethod(component, "action", "setValue(" + 
162                      thinlet.getName(component) + "," + 
163                      thinlet.getName(component) + ".group)", root, this);
164             } else if (Arrays.binarySearch(fieldsChangedByAction, className) > -1) {
165                thinlet.setMethod(component, "action", "setValue(" + name + "," +
166                      name + ".name)", root, this);
167             } else {
168                thinlet.setMethod(component, "focuslost", "setValue(" + name + 
169                      "," + name + ".name)", root, this);
170             }
171          }
172       }
173    }

175    // TODO: consider caching components
176    private List findComponents(final String name) {
177       final Object widget = thinlet.find(root, name);
178 
179       if (widget != null) {
180          return Collections.singletonList(widget);
181       }
182 
183       final List groupComponents = thinlet.getAllOfClass(root, 
184             BaseThinlet.CHECKBOX);
185 
186       for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
187          final Object component = i.next();
188 
189          if (!name.equals(thinlet.getGroup(component))) {
190             i.remove();
191          }
192       }
193 
194       return groupComponents;
195    }

197    private void createWidgetGroup(Object component, String name) {
198       final Collection widgetGroupCollection = new ArrayList();
199       widgetGroupCollection.add(name);
200 
201       final String widgetGroup = (String)thinlet.getProperty(component, 
202             "widgetGroup");
203 
204       if (widgetGroup != null) {
205          widgetGroupCollection.addAll(Arrays.asList(widgetGroup.split(",")));
206 
207          if (log.isDebugEnabled()) {
208             log.debug("Adding " + widgetGroup + " as a widget group for " +
209                   name + " in class " + form.getClass());
210          }
211       }
212 
213       widgetGroupMap.put(name, widgetGroupCollection);
214    }

216    public void setValue(Object component, String name) throws Exception {
217       final String type = Thinlet.getClass(component);
218 
219       String value = null;
220 
221       if (type.equals(BaseThinlet.CHECKBOX)) {
222          if (!thinlet.isSelected(component)) {
223             return;
224          }
225 
226          value = name;
227       } else if (type.equals(BaseThinlet.COMBOBOX) || type.equals(BaseThinlet.LIST)) {
228          component = thinlet.getSelectedItem(component);
229          
230          value = thinlet.getName(component != null ? component : null);
231       } else if (type.equals(BaseThinlet.PROGRESS_BAR) || type.equals(BaseThinlet.SLIDER)) {
232          value = thinlet.getValue(component);
233       } else {
234          value = thinlet.getText(component);
235       }
236 
237       final Map properties = new HashMap();
238       properties.put(name, value);
239 
240       controller.populate(properties);
241    }

243    protected void bindActionMetadatas(final FormMetadata formMetadata) {
244       for (final Iterator i = formMetadata.getActionMetadatas().entrySet()
245             .iterator(); i.hasNext(); ) {
246          final Map.Entry entry = (Map.Entry)i.next();
247          final String name = ((MethodEntry)entry.getKey()).getMethodName();
248          final Object component = thinlet.find(root, name);
249 
250          if (component == null) {
251             log.warn(name + " could not be found while binding " + form.getClass());
252             continue;
253          }
254 
255          final String className = Thinlet.getClass(component);
256 
257          if (!BaseThinlet.BUTTON.equals(className)) {
258             log.warn(name + " is not represented by a button; it is a " + 
259                   className);
260 
261             continue;
262          }
263 
264          createWidgetGroup(component, name);
265          
266          if (log.isTraceEnabled()) {
267             log.trace("Binding action " + name + " in " + form.getClass().getName());
268          }
269 
270          thinlet.setMethod(component, "action", "invokeAction(" + 
271                name + ".name)", root, this);
272       }
273    }

275    protected void bindDataProviders(final Collection dataProviders) {
276       for (final Iterator i = dataProviders.iterator(); i.hasNext();) {
277          final DataProviderMetadata dataProviderMetadata = 
278                (DataProviderMetadata)i.next();
279 
280          if (!dataProviderMetadata.isProvider()) {
281             continue;
282          }
283          
284          final String name = getFieldName(dataProviderMetadata);
285          final Object component = thinlet.find(root, name);
286 
287          if (component == null) {
288             log.warn(name + " could not be found while binding data provider in " + 
289                   form.getClass());
290             continue;
291          }
292 
293          final String className = Thinlet.getClass(component);
294 
295          if (!(className.equals(BaseThinlet.TABLE) || 
296                className.equals(BaseThinlet.COMBOBOX))) {
297             log.warn(className + " is not a supported widget for binding a " +
298                   "data provider");
299             continue;
300          }
301 
302          createWidgetGroup(component, name);
303 
304          if (log.isDebugEnabled()) {
305             log.debug("Binding an action method for " + name);
306          }
307 
308          thinlet.setMethod(component, "action", "updateSelection(" + name
309                + ".name," + name + ")", root, this);
310          dataProvided.put(name, dataProviderMetadata);
311       }
312    }

314    protected String getFieldName(final DataProviderMetadata dataProviderMetadata) {
315       return dataProviderMetadata.getObjectField() != null ? 
316             dataProviderMetadata.getObjectField().getFieldName() : 
317             dataProviderMetadata.getIndexField().getFieldName();

322       final int[] selected = thinlet.getSelectedIndexes(widget);
323       final DataProviderMetadata dataMeta = (DataProviderMetadata)dataProvided

356    public void dataProvidedListChanged(DataProviderMetadata metadata, List items)
357          throws Exception {
358       final String name = getFieldName(metadata);
359       final Object component = thinlet.find(root, name);
360 
361       if (component == null) {
362          log.warn(name + " could not be found while populating using "
363                + form.getClass());
364          return;
365       }
366 
367       if (log.isDebugEnabled()) {
368          log.debug("Populating " + name + " with " + metadata.getMethodEntry()
369                .getMethodName());
370       }
371 
372       final String className = Thinlet.getClass(component);
373 
374       if (className.equals(BaseThinlet.TABLE)) {
375          metadata.resetSelectedFields(form);
376          thinlet.populateFromCollection(component, items);
377       } else if (className.equals(BaseThinlet.COMBOBOX)) {
378          //TODO: This parsing shouldn't occur every time
379          final String key = (String) thinlet.getProperty(component, "key");
380 
381          if (key == null) {
382             throw new PropertyMisconfigurationException("Property 'key' "
383                   + "must be configured for the widget named " + name);
384          }
385 
386          final String value = (String) thinlet.getProperty(component, "value");
387          final Object blankObject = thinlet.getProperty(component, "blank");
388          boolean blank;
389 
390          if (blankObject == null) {
391             blank = false;
392          } else if (blankObject instanceof String) {
393             blank = Boolean.valueOf(blankObject.toString()).booleanValue();
394          } else if (blankObject instanceof Boolean) {
395             blank = ((Boolean) blankObject).booleanValue();
396          } else {
397             throw new PropertyMisconfigurationException("Property 'blank' "
398                   + "for the widget named " + name + " must either be left "
399                   + "empty or contain a boolean value");
400          }
401 
402          thinlet.populateFromCollection(component, items, key, value, blank);
403       } else {
404          // TODO: only table and combobox are implemented
405          throw new UnsupportedOperationException(className + " is not "
406                + "supported for data providing");
407       }
408    }

410    public void enabledConditionsChanged(Map updatedEnabledConditions) {
411       for (final Iterator i = updatedEnabledConditions.entrySet().iterator(); 
412             i.hasNext(); ) {
413          final Map.Entry entry = (Map.Entry)i.next();
414 
415          final Collection widgetGroup = (Collection)widgetGroupMap.get(
416                entry.getKey().toString());
417 
418          if (widgetGroup == null) {
419             log.warn(entry.getKey() + " enabled state should be changed, " +
420                   "but could not be found in the view component for " + 
421                   form.getClass());
422             continue;
423          }
424 
425          if (log.isDebugEnabled()) {
426             log.debug("Changing " + entry.getKey() + " enabled state to " + 
427                   entry.getValue());
428          }
429 
430          for (Iterator it = widgetGroup.iterator(); it.hasNext(); ) {
431             final Object widget = thinlet.find(root, it.next().toString());
432 
433             if (widget == null) {
434                log.warn(entry.getKey() + " enabled state should be changed, " +
435                      "but could not be found in the view component for " + 
436                      form.getClass());
437                continue;
438             }
439 
440             thinlet.setEnabled(widget, ((Boolean)entry.getValue())
441                   .booleanValue());
442          }
443       }
444    }
