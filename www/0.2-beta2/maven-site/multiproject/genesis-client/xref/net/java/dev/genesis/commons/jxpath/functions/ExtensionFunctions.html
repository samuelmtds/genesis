36  public class ExtensionFunctions {
37     private static Log log = LogFactory.getLog(ExtensionFunctions.class);

39     private static Object getValue(final ExpressionContext ctx,
40           final String propertyName) throws Exception {
41  
42        return PropertyUtils.getProperty(ctx.getJXPathContext().getContextBean(),
43              propertyName);
44     }
45  
46     private static Object getValue(final ExpressionContext ctx, final Object obj)
47           throws Exception {
48        return obj instanceof GenesisNodeSet ? getValue(ctx,
49              getSimplePropertyName((GenesisNodeSet) obj)) : obj;
50     }
51  
52     private static String getSimplePropertyName(final GenesisNodeSet nodeSet) {
53        final String path = nodeSet.getPath();
54        int index;
55  
56        return (index = path.indexOf('/')) > 0 ? path.substring(0, index) : path;
57     }
58  
59     private static EqualityComparator getEqualityComparator(
60           final ExpressionContext ctx, final Object obj) {
61  
62        if (obj instanceof GenesisNodeSet) {
63           final FormMetadata meta = (FormMetadata) ctx.getJXPathContext()
64                 .getVariables().getVariable(FormController.FORM_METADATA_KEY);
65           return meta.getFieldMetadata(
66                 getSimplePropertyName((GenesisNodeSet) obj))
67                 .getEqualityComparator();
68        }
69  
70        return EqualityComparatorRegistry.getInstance()
71              .getDefaultEqualityComparatorFor(obj.getClass());
72     }
73  
74     public static boolean hasChanged(final ExpressionContext context,
75           final GenesisNodeSet nodeSet) {
76  
77        final Map changedMap = ((FormState)context.getJXPathContext()
78              .getVariables().getVariable(FormController.CURRENT_STATE_KEY))
79              .getChangedMap();
80        final String propertyName = getSimplePropertyName(nodeSet);
81        final boolean result = changedMap.containsKey(propertyName);
82  
83        if (log.isDebugEnabled()) {
84           log.debug("Property '" + propertyName
85                 + (result ? "' changed" : "' didn´t change"));
86        }
87  
88        return result;
89     }
90  
91     public static boolean hasNotChanged(final ExpressionContext context,
92           final GenesisNodeSet nodeSet) {
93        return !hasChanged(context, nodeSet);
94     }
95  
96     public static boolean isEmpty(final ExpressionContext ctx, final Object obj)
97           throws Exception {
98  
99        if (!(obj instanceof GenesisNodeSet)) {
100          return EmptyResolverRegistry.getInstance().getDefaultEmptyResolverFor(
101                obj.getClass()).isEmpty(obj);
102       }
103 
104       final GenesisNodeSet nodeSet = (GenesisNodeSet) obj;
105       final FormMetadata formMeta = (FormMetadata) ctx.getJXPathContext()
106             .getVariables().getVariable(FormController.FORM_METADATA_KEY);
107       final String propertyName = getSimplePropertyName(nodeSet);
108 
109       return formMeta.getFieldMetadata(getSimplePropertyName(nodeSet))
110             .getEmptyResolver().isEmpty(getValue(ctx, propertyName));
111    }
112 
113    public static boolean isNotEmpty(final ExpressionContext ctx, final Object obj)
114    		throws Exception {
115       return !isEmpty(ctx, obj);
116    }

118    public static boolean equals(final ExpressionContext ctx, final Object arg1,
119          final Object arg2) throws Exception {
120 
121       final EqualityComparator comp1 = getEqualityComparator(ctx, arg1);
122       final EqualityComparator comp2 = getEqualityComparator(ctx, arg2);
123 
124       final Object value1 = getValue(ctx, arg1);
125       final Object value2 = getValue(ctx, arg2);
126 
127       if (comp1 == comp2) {
128          if (log.isDebugEnabled()) {
129             log.debug("Evaluation equals for '" + value1 + "' and '" + value2
130                   + "' with same comparator " + comp1);
131          }
132          return comp1.equals(value1, value2);
133       } else if (value1 == null) {
134          if (log.isDebugEnabled()) {
135             log.debug("Evaluation equals for '" + value1 + "' and '" + value2
136                   + "'");
137          }
138          return value2 == null;
139       } else if (value1.getClass().equals(value2.getClass())
140             || value1.getClass().isAssignableFrom(value2.getClass())) {
141          if (log.isDebugEnabled()) {
142             log.debug("Evaluation equals for '" + value1 + "' and '" + value2
143                   + "' with first comparator " + comp1);
144          }
145          return comp1.equals(value1, value2);
146       } else if (value2.getClass().isAssignableFrom(value1.getClass())) {
147          if (log.isDebugEnabled()) {
148             log.debug("Evaluation equals for '" + value1 + "' and '" + value2
149                   + "' with second comparator " + comp2);
150          }
151          return comp2.equals(value2, value1);
152       }
153       if (log.isDebugEnabled()) {
154          log.debug("'" + value1 + "' and '" + value2
155                + "' have different classes");
156       }
157       return false;
158 
159    }

161    public static boolean notEquals(final ExpressionContext ctx, final Object arg1,
162          final Object arg2) throws Exception {
163       return !equals(ctx, arg1, arg2);
