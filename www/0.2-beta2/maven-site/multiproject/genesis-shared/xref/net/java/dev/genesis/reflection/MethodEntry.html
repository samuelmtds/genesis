19  package net.java.dev.genesis.reflection;
20  
21  import java.io.Serializable;
22  import java.lang.reflect.Method;
23  import java.util.Arrays;
24  
25  public class MethodEntry implements Serializable {
26     private final String methodName;
27     private final String[] argsClassesNames;
28  
29     public MethodEntry(final String methodName, final String[] argsClassesNames) {
30        this.methodName = methodName;
31        this.argsClassesNames = argsClassesNames;
32     }
33     
34     public MethodEntry(final Method method){
35        final Class[] parameterTypes = method.getParameterTypes();
36        final String[] classNames = new String[parameterTypes.length];
37  
38        for(int i = 0; i < parameterTypes.length; i++){
39            classNames[i] = parameterTypes[i].getName();
40        }
41        this.methodName = method.getName();
42        this.argsClassesNames = classNames;
43     }
44  
45     public String[] getArgsClassesNames() {
46        return argsClassesNames;
47     }
48  
49     public String getMethodName() {
50        return methodName;
51     }
52  
53     public boolean equals(Object o) {
54        MethodEntry that = (MethodEntry) o;
55  
56        return that.methodName.equals(this.methodName)
57              && Arrays.equals(that.argsClassesNames, this.argsClassesNames);
58     }
59  
60     public int hashCode() {
61        return methodName.hashCode();
62     }
63     
64     public String toString() {
65        StringBuffer buffer = new StringBuffer();
66        buffer.append(methodName);
67        buffer.append("(");
68        for (int i = 0; i < argsClassesNames.length; i++) {
69           if(i > 0){
70              buffer.append(",");
71           }
72           buffer.append(argsClassesNames[i]);
73        }
74        buffer.append(")");
75        return buffer.toString();
76     }
77  }
