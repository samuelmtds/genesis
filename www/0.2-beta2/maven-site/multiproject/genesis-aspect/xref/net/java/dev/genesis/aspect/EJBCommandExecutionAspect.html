19  package net.java.dev.genesis.aspect;
20  
21  import java.lang.reflect.InvocationTargetException;
22  import java.rmi.NoSuchObjectException;
23  import java.rmi.RemoteException;
24  
25  import javax.naming.InitialContext;
26  import javax.rmi.PortableRemoteObject;
27  import net.java.dev.genesis.ejb.CommandExecutor;
28  import net.java.dev.genesis.ejb.CommandExecutorHome;
29  
30  import org.apache.commons.logging.Log;
31  import org.apache.commons.logging.LogFactory;
32  import org.codehaus.aspectwerkz.CrossCuttingInfo;
33  import org.codehaus.aspectwerkz.joinpoint.JoinPoint;
34  import org.codehaus.aspectwerkz.joinpoint.MethodRtti;
35  import org.codehaus.aspectwerkz.joinpoint.impl.MethodRttiImpl;
36  
37  /***
38   * @Aspect perJVM
39   */
40  public class EJBCommandExecutionAspect extends CommandInvocationAspect {
41     private static Log log = LogFactory.getLog(EJBCommandExecutionAspect.class);
42  
43     private CommandExecutorHome home;
44     private CommandExecutor session;
45     private boolean retryOnNoSuchObject;
46  
47     public EJBCommandExecutionAspect(CrossCuttingInfo ccInfo) {
48        super(ccInfo);
49  
50        if (!ccInfo.isPrototype()) {
51           retryOnNoSuchObject = !"false".equals(ccInfo
52                 .getParameter("retryOnNoSuchObject"));
53        }
54     }
55  
56     private CommandExecutorHome getHome() throws Exception {
57        if (home == null) {
58           home = (CommandExecutorHome)PortableRemoteObject.narrow(
59                 new InitialContext().lookup(ccInfo.getParameter("jndiName")),
60                 CommandExecutorHome.class);
61        }
62  
63        return home;
64     }
65  
66     private CommandExecutor getCommandExecutor() {
67        if (session == null) {
68           try {
69              session = getHome().create();
70           } catch (final Exception e) {
71              home = null;
72              log.error(e.getMessage(), e);
73  
74              throw new RuntimeException(e);
75           }
76        }
77  
78        return session;
79     }
80  
81     /***
82      * 
83      * @param jp
84      *            the join point
85      * @return the resulting object
86      * @throws Throwable
87      * 
88      * @Around ejbCommandExecution
89      */
90     public Object commandExecution(final JoinPoint jp) throws Throwable {
91        final MethodRtti rtti = (MethodRtti)jp.getRtti();
92        final CommandResolver obj = (CommandResolver)jp.getTarget();
93        final Class[] classes = rtti.getParameterTypes();
94        final String[] classNames = new String[classes.length];
95        final String methodName = ((MethodRttiImpl)rtti).getMethodTuple()
96              .getWrapperMethod().getName();
97        final Object[] parameterValues = rtti.getParameterValues();
98  
99        for (int i = 0; i < classes.length; i++) {
100          classNames[i] = classes[i].getName();
101       }
102 
103       CommandExecutor commandExecutor = null;
104 
105       boolean retry = retryOnNoSuchObject;
106 
107       while (true) {
108          try {
109             commandExecutor = getCommandExecutor();
110             if (obj.isTransactional(rtti.getMethod())) {
111                return commandExecutor.executeTransaction(obj, methodName,
112                      classNames, parameterValues);
113             } else if (obj.isRemotable(rtti.getMethod())) {
114                return commandExecutor.executeQuery(obj, methodName, classNames,
115                      parameterValues);
116             } else {
117                return jp.proceed();
118             }
119          } catch (final NoSuchObjectException nsoe) {
120             log.error("NoSuchObjectException occured", nsoe);
121 
122             if (retry) {
123                log.info("Retrying on NoSuchObjectException...");
124                session = null;
125                home = null;
126                retry = false;
127 
128                continue;
129             }
130 
131             throw nsoe;
132          } catch (final RemoteException re) {
133             log.error("RemoteException occured", re);
134             cleanUp();
135 
136             throw re;
137          } catch (final InvocationTargetException ite) {
138             throw ite.getTargetException();
139          }
140       }
141    }
142    
143    private void cleanUp() {
144       if (session != null) {
145          try {
146             session.remove();
147          } catch (Throwable t) {
148             log.error("An error ocurred while removing EJB instance", t);
149          }
150 
151          session = null;
152       }
153    }
