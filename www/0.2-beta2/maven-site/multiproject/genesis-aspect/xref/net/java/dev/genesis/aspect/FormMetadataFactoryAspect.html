74           public static class ConditionAnnotationHandler implements
75                 AnnotationHandler {
76              public void processFormAnnotation(final FormMetadata formMetadata,
77                    final Annotation annotation) {
78  
79                 final String value = ((UntypedAnnotationProxy) annotation)
80                       .getValue();
81                 final int equalIndex = value.indexOf('=');
82                 if (equalIndex > 0) {
83                    formMetadata.addNamedCondition(value.substring(0, equalIndex)
84                          .trim(), JXPathContext.compile(value
85                          .substring(equalIndex + 1)));
86                 }
87              }
88  
89              public void processFieldAnnotation(final FormMetadata formMetadata,
90                    final FieldMetadata fieldMetadata, final Annotation annotation) {
91                 processFormAnnotation(formMetadata, annotation);
92              }
93  
94              public void processMethodAnnotation(
95                    final FormMetadata formMetadata,
96                    final MethodMetadata methodMetadata,
97                    final Annotation annotation) {
98                 processFormAnnotation(formMetadata, annotation);
99              }

230             public void processMethodAnnotation(
231                   final FormMetadata formMetadata,
232                   final MethodMetadata methodMetadata,
233                   final Annotation annotation) {
234 
235                final Map attributesMap = GenesisUtils.getAttributesMap(
236                      ((UntypedAnnotationProxy) annotation).getValue());
237                final String objectFieldName = (String)attributesMap
238                      .get("objectField");
239                final String indexFieldName = (String)attributesMap
240                      .get("indexField");
241                final String callOnInit = (String)attributesMap
242                      .get("callOnInit");
243 
244                final DataProviderMetadata dataProviderMetadata = 
245                      ((DataProviderMetadata) methodMetadata);
246 
247                if (objectFieldName == null && indexFieldName == null) {
248                   throw new RuntimeException("At least one of objectField or " + 
249                         "indexField must be specified for @DataProvider in " +
250                         methodMetadata.getMethodEntry().getMethodName());
251                }
252 
253                dataProviderMetadata.setCallOnInit(!"false".equals(callOnInit));
254 
255                PropertyDescriptor[] descriptors = PropertyUtils
256                      .getPropertyDescriptors(formMetadata.getFormClass());
257 
258                Map descriptorsPerPropertyName = new HashMap();
259 
260                for (int i = 0; i < descriptors.length; i++) {
261                   descriptorsPerPropertyName.put(descriptors[i].getName(), 
262                         descriptors[i]);
263                }
264 
265                if (objectFieldName != null) {
266                   PropertyDescriptor descriptor = (PropertyDescriptor)
267                         descriptorsPerPropertyName.get(objectFieldName);
268                   
269                   if (descriptor == null) {
270                      throw new RuntimeException("The object "
271                            + formMetadata.getFormClass()
272                            + " doesn´t have a field called " + objectFieldName);
273                   }
274 
275                   dataProviderMetadata.setObjectField(new FieldEntry(
276                         descriptor.getName(), descriptor.getPropertyType()));
277                }
278 
279                if (indexFieldName != null) {
280                   PropertyDescriptor descriptor = (PropertyDescriptor)
281                         descriptorsPerPropertyName.get(indexFieldName);
282                   
283                   if (descriptor == null) {
284                      throw new RuntimeException("The object "
285                            + formMetadata.getFormClass()
286                            + " doesn´t have a field called " + indexFieldName);
287                   }
288 
289                   dataProviderMetadata.setIndexField(new FieldEntry(
290                         descriptor.getName(), descriptor.getPropertyType()));
291                }
292             }

488       public FormMetadata getFormMetadata(final Class formClass) {
489          FormMetadata formMetadata = (FormMetadata) cache.get(formClass);
490          if (formMetadata == null) {
491             formMetadata = new FormMetadata(formClass);
492             processAnnotations(formMetadata);
493             cache.put(formClass, formMetadata);
494          }
495          return formMetadata;
496       }

504       private void processFormAnnotations(final FormMetadata formMetadata) {
505          final List annotations = Annotations.getAnnotationInfos(formMetadata
506                .getFormClass());
507          AnnotationInfo info;
508          MetadataAttribute attr;
509          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
510             info = (AnnotationInfo) iter.next();
511             attr = MetadataAttribute.get(info.getName());
512             if (attr != null) {
513                attr.getHandler().processFormAnnotation(formMetadata,
514                      info.getAnnotation());
515             }
516 
517          }
518       }

520       private void processFieldsAnnotations(final FormMetadata formMetadata) {
521          final Class clazz = formMetadata.getFormClass();
522          final PropertyDescriptor[] propertyDescriptors = PropertyUtils
523                .getPropertyDescriptors(clazz);
524          PropertyDescriptor propDesc;
525          FieldMetadata fieldMetadata;
526          for (int i = 0; i < propertyDescriptors.length; i++) {
527             propDesc = propertyDescriptors[i];
528             // Ignoring java.lang.Object.getClass()
529             if (!propDesc.getName().equals("class") && 
530                   propDesc.getReadMethod() != null) {
531                fieldMetadata = new FieldMetadata(propDesc.getName(), propDesc
532                      .getPropertyType());
533                formMetadata.addFieldMetadata(propDesc.getName(), fieldMetadata);
534                processFieldAnnotations(formMetadata, fieldMetadata, 
535                      getReadMethod(clazz, propDesc));
536             }
537          }
538       }

556       private void processMethodsAnnotations(final FormMetadata formMetadata) {
557          Method[] methods = formMetadata.getFormClass().getMethods();
558          MethodMetadata methodMetadata;
559          for (int i = 0; i < methods.length; i++) {
560 
561             if (Annotations.getAnnotation("Action", methods[i]) != null) {
562                methodMetadata = new ActionMetadata(methods[i]);
563                formMetadata.addActionMetadata(methods[i],
564                      (ActionMetadata) methodMetadata);
565                processActionAnnotations(formMetadata,
566                      (ActionMetadata) methodMetadata, methods[i]);
567             } else if (Annotations.getAnnotation("DataProvider", methods[i]) != null) {
568                methodMetadata = new DataProviderMetadata(methods[i]);
569                formMetadata.addDataProviderMetadata(methods[i],
570                      (DataProviderMetadata) methodMetadata);
571                processDataProviderAnnotations(formMetadata,
572                      (DataProviderMetadata) methodMetadata, methods[i]);
573             }
574          }
575       }

609       // Work around for a bug in JDK 1.4.2
610       private Method getReadMethod(Class clazz, PropertyDescriptor pd) {
611          Method method = pd.getReadMethod();
612 
613          if (clazz.equals(method.getDeclaringClass())) {
614             return method;
615          }
616 
617          try {
618             method = clazz.getMethod(method.getName(), 
619                   method.getParameterTypes());
620             return (method == null) ? pd.getReadMethod() : method;
621          } catch (NoSuchMethodException nsme) {
622             return method;
623          }
624       }
