33     public static Object setSelectedIndexes(Object component, int[] indexes) {
34        if (BaseThinlet.COMBOBOX.equals(Thinlet.getClass(component))) {
35           setSelected(component, indexes.length > 0 ? indexes[0] : -1);
36           return component;
37        }
38  
39        final Object[] items = helper.getItems(component);
40        int j = 0;
41  
42        for (int i = 0; i < items.length; i++) {
43           if (indexes.length - j < 1 || indexes[j] != i) {
44              setSelected(items[i], false);
45              continue;
46           }
47  
48           setSelected(items[i], true);
49           j++;
50        }
51        return component;
52     }
