74        public AspectFormMetadataFactory() 
75              throws Exception {
76           Map parameters = Mixins.getParameters(getClass(), getClass()
77                 .getClassLoader());
78           String factoryName = (String)parameters.get("scriptFactory");
79           String scriptFactoryProperties = (String)parameters
80                 .get("scriptFactoryProperties");
81  
82           Map properties = new HashMap();
83  
84           if (factoryName == null) {
85              factoryName = "jxpath";
86           } 
87  
88           String scriptFactoryName = (String)FACTORIES.get(factoryName);
89           if (scriptFactoryName == null) {
90              scriptFactoryName = factoryName;
91           } else {
92              properties.put("lang", factoryName);
93           }
94  
95           scriptFactory = (ScriptFactory)ClassesCache
96                 .getClass(scriptFactoryName).newInstance();
97  
98           if (scriptFactoryProperties != null) {
99              properties.putAll(GenesisUtils
100                   .getAttributesMap(scriptFactoryProperties, ",", "="));
101          }
102 
103          if (!properties.isEmpty()) {
104             PropertyUtils.copyProperties(scriptFactory, properties);
105          }
106       }

109          if (script == null) {
110             script = scriptFactory.newScript();
111          }

287             public void processMethodAnnotation(
288                   final FormMetadata formMetadata,
289                   final MethodMetadata methodMetadata,
290                   final Annotation annotation) {
291 
292                final Map attributesMap = GenesisUtils.getAttributesMap(
293                      ((UntypedAnnotation) annotation).value());
294                final String widgetName = (String)attributesMap
295                      .get("widgetName");
296                final String objectFieldName = (String)attributesMap
297                      .get("objectField");
298                final String indexFieldName = (String)attributesMap
299                      .get("indexField");
300                final String callOnInit = (String)attributesMap
301                      .get("callOnInit");
302                final String resetSelection = (String)attributesMap
303                      .get("resetSelection");
304 
305                final DataProviderMetadata dataProviderMetadata = methodMetadata
306                      .getDataProviderMetadata();
307 
308                if (widgetName == null && objectFieldName == null && indexFieldName == null) {
309                   throw new RuntimeException("At least one of widgetName, objectField or " + 
310                         "indexField must be specified for @DataProvider in " +
311                         methodMetadata.getMethodEntry().getMethodName());
312                }
313 
314                dataProviderMetadata.setCallOnInit(!"false".equals(callOnInit));
315                dataProviderMetadata.setResetSelection(!"false".equals(resetSelection));
316 
317                PropertyDescriptor[] descriptors = PropertyUtils
318                      .getPropertyDescriptors(formMetadata.getFormClass());
319 
320                Map descriptorsPerPropertyName = new HashMap();
321 
322                for (int i = 0; i < descriptors.length; i++) {
323                   descriptorsPerPropertyName.put(descriptors[i].getName(), 
324                         descriptors[i]);
325                }
326 
327                if (objectFieldName != null) {
328                   PropertyDescriptor descriptor = (PropertyDescriptor)
329                         descriptorsPerPropertyName.get(objectFieldName);
330 
331                   if (descriptor == null) {
332                      throw new RuntimeException("The object "
333                            + formMetadata.getFormClass()
334                            + " doesn´t have a field called " + objectFieldName);
335                   }
336 
337                   final Class fieldType = descriptor.getPropertyType();
338 
339                   if (fieldType.isPrimitive()
340                         || (fieldType.isArray() &&
341                               fieldType.getComponentType().isPrimitive())) {
342                      throw new RuntimeException("The object " + objectFieldName
343                            + " has an object field called " + objectFieldName
344                            + " that cannot be primitive or array of primitives");
345                   }
346 
347                   dataProviderMetadata.setObjectField(new FieldEntry(
348                         descriptor.getName(), descriptor.getPropertyType()));
349                }
350 
351                if (indexFieldName != null) {
352                   PropertyDescriptor descriptor = (PropertyDescriptor)
353                         descriptorsPerPropertyName.get(indexFieldName);
354 
355                   if (descriptor == null) {
356                      throw new RuntimeException("The object "
357                            + formMetadata.getFormClass()
358                            + " doesn´t have a field called " + indexFieldName);
359                   }
360 
361                   final Class fieldType = descriptor.getPropertyType()
362                         .isArray() ? descriptor.getPropertyType()
363                         .getComponentType() : descriptor.getPropertyType();
364 
365                   if (!Collection.class.isAssignableFrom(descriptor
366                         .getPropertyType())
367                         && !Integer.TYPE.isAssignableFrom(fieldType)
368                         && !Integer.class.isAssignableFrom(fieldType)) {
369                      throw new RuntimeException(
370                            "The object "
371                                  + objectFieldName
372                                  + " has an index field called "
373                                  + indexFieldName
374                                  + " that's not an Integer, int, Collection of them, or array of them.");
375                   }
376 
377                   dataProviderMetadata.setIndexField(new FieldEntry(descriptor
378                         .getName(), descriptor.getPropertyType()));
379                }
380 
381                dataProviderMetadata
382                      .setWidgetName(widgetName != null ? widgetName
383                            : objectFieldName != null ? objectFieldName
384                                  : indexFieldName);
385             }
386          }

609       public FormMetadata getFormMetadata(final Class formClass) {
610          FormMetadata formMetadata = (FormMetadata) cache.get(formClass);
611          if (formMetadata == null) {
612             formMetadata = new FormMetadata(formClass, getScript());
613             processAnnotations(formMetadata);
614             cache.put(formClass, formMetadata);
615          }
616          return formMetadata;

619       private void processAnnotations(final FormMetadata formMetadata) {
620          processFormAnnotations(formMetadata);
621          processFieldsAnnotations(formMetadata);
622          processMethodsAnnotations(formMetadata);
623       }

625       private void processFormAnnotations(final FormMetadata formMetadata) {
626          final List annotations = Annotations.getAnnotationInfos(formMetadata
627                .getFormClass());
628          AnnotationInfo info;
629          MetadataAttribute attr;
630          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
631             info = (AnnotationInfo) iter.next();
632             attr = MetadataAttribute.get(info.getName());
633 
634             if (attr == null) {
635                continue;
636             }
637 
638             attr.getHandler().processFormAnnotation(formMetadata,
639                   info.getAnnotation());
640          }
641       }
642 
643       private void processFieldsAnnotations(final FormMetadata formMetadata) {
644          final Class clazz = formMetadata.getFormClass();
645          final PropertyDescriptor[] propertyDescriptors = PropertyUtils
646                .getPropertyDescriptors(clazz);
647          PropertyDescriptor propDesc;
648          FieldMetadata fieldMetadata;
649          for (int i = 0; i < propertyDescriptors.length; i++) {
650             propDesc = propertyDescriptors[i];
651 
652             // Ignoring java.lang.Object.getClass()
653             if (propDesc.getName().equals("class")
654                   || propDesc.getReadMethod() == null
655                   || Annotations.getAnnotation("NotBound", propDesc
656                         .getReadMethod()) != null) {
657                continue;
658             }
659 
660             fieldMetadata = new FieldMetadata(propDesc.getName(), propDesc
661                   .getPropertyType(), propDesc.getWriteMethod() != null);
662             formMetadata.addFieldMetadata(propDesc.getName(), fieldMetadata);
663             processFieldAnnotations(formMetadata, fieldMetadata, getReadMethod(
664                   clazz, propDesc));
665 
666          }
667       }
668 
669       private void processFieldAnnotations(final FormMetadata formMetadata,
670             final FieldMetadata fieldMetadata, final Method fieldGetterMethod) {
671          final List annotations = Annotations
672                .getAnnotationInfos(fieldGetterMethod);
673          AnnotationInfo info;
674          MetadataAttribute attr;
675          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
676             info = (AnnotationInfo) iter.next();
677             attr = MetadataAttribute.get(info.getName());
678 
679             if (attr == null) {
680                continue;
681             }
682 
683             attr.getHandler().processFieldAnnotation(formMetadata,
684                   fieldMetadata, info.getAnnotation());
685          }
686       }
687 
688       private void processMethodsAnnotations(final FormMetadata formMetadata) {
689          final Method[] methods = formMetadata.getFormClass().getMethods();
690          boolean isAction;
691          boolean isProvider;
692          MethodMetadata methodMetadata;
693 
694          for (int i = 0; i < methods.length; i++) {
695             isAction = Annotations.getAnnotation("Action", methods[i]) != null;
696             isProvider = Annotations.getAnnotation("DataProvider", methods[i]) != null;
697 
698             if (!isAction && !isProvider) {
699                continue;
700             }
701 
702             methodMetadata = new MethodMetadata(methods[i], isAction, isProvider);
703             formMetadata.addMethodMetadata(methods[i], methodMetadata);
704             processMethodAnnotations(formMetadata, methodMetadata, methods[i]);
705          }
706       }
707 
708       private void processMethodAnnotations(final FormMetadata formMetadata,
709             final MethodMetadata methodMetadata, final Method actionMethod) {
710          final List annotations = Annotations.getAnnotationInfos(actionMethod);
711          AnnotationInfo info;
712          MetadataAttribute attr;
713          for (Iterator iter = annotations.iterator(); iter.hasNext();) {
714             info = (AnnotationInfo) iter.next();
715             attr = MetadataAttribute.get(info.getName());
716 
717             if (attr == null) {
718                continue;
719             }
720 
721             attr.getHandler().processMethodAnnotation(formMetadata,
722                   methodMetadata, info.getAnnotation());
723          }
724       }

726       // Work around for a bug in JDK 1.4.2
727       private Method getReadMethod(Class clazz, PropertyDescriptor pd) {
728          Method method = pd.getReadMethod();
729 
730          if (clazz.equals(method.getDeclaringClass())) {
731             return method;
732          }
733 
734          try {
735             method = clazz.getMethod(method.getName(),
