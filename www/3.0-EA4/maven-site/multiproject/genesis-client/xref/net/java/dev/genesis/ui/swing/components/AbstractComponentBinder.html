73     public abstract class AbstractBoundMember implements BoundMember {
74        private final JComponent component;
75        private final SwingBinder binder;
76        private final Set enabledWidgetGroupSet = new HashSet();
77        private final Set visibleWidgetGroupSet = new HashSet();
78  
79        public AbstractBoundMember(SwingBinder binder, JComponent component) {
80           this.component = component;
81           this.binder = binder;
82  
83           createGroups();
84  
85           storeBinder();
86        }
87  
88        protected void storeBinder() {
89           component.putClientProperty(SwingBinder.BINDER_KEY, binder);
90        }
91  
92        protected SwingBinder getBinder() {
93           return binder;
94        }
95  
96        protected Set getEnabledWidgetGroupSet() {
97           return enabledWidgetGroupSet;
98        }
99  
100       protected Set getVisibleWidgetGroupSet() {
101          return visibleWidgetGroupSet;
102       }
103 
104       protected void createGroups() {
105          createWidgetGroup();
106          createEnabledGroup();
107          createVisibleGroup();
108       }
109 
110       protected void createWidgetGroup() {
111          createGroup(component
112                .getClientProperty(SwingBinder.WIDGET_GROUP_PROPERTY), true, true);
113       }
114 
115       protected void createEnabledGroup() {
116          createGroup(component
117                .getClientProperty(SwingBinder.ENABLED_GROUP_PROPERTY), true,
118                false);
119       }
120 
121       protected void createVisibleGroup() {
122          createGroup(component
123                .getClientProperty(SwingBinder.VISIBLE_GROUP_PROPERTY), false,
124                true);
125       }
126 
127       protected void createGroup(Object group, boolean enabled, boolean visible) {
128          if (group == null) {
129             return;
130          }
131 
132          boolean instanceofString = group instanceof String;
133          if (instanceofString || group instanceof String[]) {
134         	
135             String[] componentNames = instanceofString ?
136             		((String) group).split("//s*,//s*") : (String[]) group;
137 
138             for (int i = 0; i < componentNames.length; i++) {
139                Component c = (Component) getBinder().lookup(componentNames[i]);
140 
141                if (c != null) {
142                   if (enabled) {
143                      enabledWidgetGroupSet.add(c);
144                   }
145 
146                   if (visible) {
147                      visibleWidgetGroupSet.add(c);
148                   }
149                }
150             }
151          } else if (group instanceof Collection) {
152             Collection groupCollection = (Collection) group;
153 
154             if (enabled) {
155                enabledWidgetGroupSet.addAll(groupCollection);
156             }
157 
158             if (visible) {
159                visibleWidgetGroupSet.addAll(groupCollection);
160             }
161          } else if (group instanceof Object[]) {
162             Object[] groupArray = (Object[]) group;
163 
164             if (enabled) {
165                enabledWidgetGroupSet.addAll(Arrays.asList(groupArray));
166             }
167 
168             if (visible) {
169                visibleWidgetGroupSet.addAll(Arrays.asList(groupArray));
170             }
171          } else if (group instanceof JComponent) {
172             JComponent jComponent = (JComponent) group;
173 
174             if (enabled) {
175                enabledWidgetGroupSet.add(jComponent);
176             }
177 
178             if (visible) {
179                visibleWidgetGroupSet.add(jComponent);
180             }
181          } else {
182             throw new IllegalArgumentException("Group property must be a comma-separated string, " +
183                   "array of strings, a collection of components, an array of components or a JComponent");
184          }
185       }
186 
187       protected boolean isBlank(JComponent component) {
188          return isBoolean(component, SwingBinder.BLANK_PROPERTY);
189       }
190 
191       protected boolean isBoolean(JComponent component, String propertyName) {
192          final Object booleanObject = component.getClientProperty(propertyName);
193 
194          boolean ret;
195 
196          if (booleanObject == null) {
197             ret = false;
198          } else if (booleanObject instanceof String) {
199             ret = Boolean.valueOf(booleanObject.toString()).booleanValue();
200          } else if (booleanObject instanceof Boolean) {
201             ret = ((Boolean) booleanObject).booleanValue();
202          } else {
203             throw new PropertyMisconfigurationException("Property '" +
204                propertyName + "' " + "for the component named " +
205                getName() + " must " +
206                "either be left empty or contain a boolean value");
207          }
208 
209          return ret;
210       }
211 
212       public void setEnabled(boolean enabled) {
213          component.setEnabled(enabled);
214 
215          for (Iterator iter = enabledWidgetGroupSet.iterator(); iter.hasNext();) {
216             ((JComponent) iter.next()).setEnabled(enabled);
217          }
218       }
219 
220       public void setVisible(boolean visible) {
221          component.setVisible(visible);
222 
223          for (Iterator iter = visibleWidgetGroupSet.iterator(); iter.hasNext();) {
224             ((JComponent) iter.next()).setVisible(visible);
225          }
226       }
227 
228       public String getName() {
229          return getBinder().getName(component);
230       }
231 
232       public void unbind() {
233       }
234    }
