19  package net.java.dev.genesis.ui.swing.components;

21  import java.awt.Component;
22  
23  import javax.swing.JSlider;
24  import javax.swing.event.ChangeEvent;
25  import javax.swing.event.ChangeListener;
26  
27  import org.apache.commons.beanutils.BeanUtilsBean;
28  
29  import net.java.dev.genesis.ui.binding.BoundField;
30  import net.java.dev.genesis.ui.metadata.FieldMetadata;
31  import net.java.dev.genesis.ui.swing.SwingBinder;
32  
33  public class JSliderComponentBinder extends AbstractComponentBinder {
34     public BoundField bind(SwingBinder binder, Component component,
35        FieldMetadata fieldMetadata) {
36        return new JSliderBoundField(binder, (JSlider) component, fieldMetadata);
37     }
38  
39     public class JSliderBoundField extends AbstractBoundMember
40           implements BoundField {
41        private final JSlider component;
42        private final FieldMetadata fieldMetadata;
43        private final ChangeListener listener;
44  
45        public JSliderBoundField(SwingBinder binder, JSlider component,
46           FieldMetadata fieldMetadata) {
47           super(binder, component);
48           this.component = component;
49           this.fieldMetadata = fieldMetadata;
50  
51           component.addChangeListener(listener = createChangeListener());
52        }
53  
54        protected JSlider getComponent() {
55           return component;
56        }
57  
58        protected FieldMetadata getFieldMetadata() {
59           return fieldMetadata;
60        }
61  
62        protected ChangeListener getListener() {
63           return listener;
64        }
65  
66        protected ChangeListener createChangeListener() {
67           return new ChangeListener() {
68              public void stateChanged(ChangeEvent event) {
69                 getBinder().populateForm(getFieldMetadata(), getValue());
70              }
71           };
72        }
73  
74        protected Object getValue() {
75           return new Integer(component.getValue());
76        }
77  
78        protected int toInt(Object value) throws Exception {
79           Integer integer = (Integer) BeanUtilsBean.getInstance()
80                 .getConvertUtils().lookup(Integer.TYPE).convert(Integer.TYPE,
81                       value);
82  
83           return integer.intValue();
84        }
85  
86        public void setValue(Object value) throws Exception {
87           deactivateListeners();
88           try {
89              component.setValue(toInt(value));
90           } finally {
91              reactivateListeners();
92           }
93        }
94        
95        protected void deactivateListeners() {
96           if (listener != null) {
97              component.removeChangeListener(listener);
98           }
99        }
100 
101       protected void reactivateListeners() {
102          if (listener != null) {
103             component.addChangeListener(listener);
104          }
105       }
106 
107       public void unbind() {
108          if (listener != null) {
109             component.removeChangeListener(listener);
110          }
111       }
112    }
