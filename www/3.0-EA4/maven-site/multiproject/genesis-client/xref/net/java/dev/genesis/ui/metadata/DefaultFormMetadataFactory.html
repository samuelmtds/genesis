35  public class DefaultFormMetadataFactory implements FormMetadataFactory {
36     public FormMetadata getFormMetadata(final Class formClass) {
37        FormMetadata formMetadata = FormMetadataRegistry.getInstance().get(formClass);
38  
39        if (formMetadata == null) {
40           formMetadata = new FormMetadata(formClass, ScriptRegistry.getInstance().getScript());
41           processAnnotations(formMetadata);
42           FormMetadataRegistry.getInstance().register(formClass, formMetadata);
43        }
44  
45        return formMetadata;
46     }
47  
48     private void processAnnotations(final FormMetadata formMetadata) {
49        processFormAnnotations(formMetadata);
50        processFieldsAnnotations(formMetadata);
51        processMethodsAnnotations(formMetadata);
52     }
53  
54     private void processFormAnnotations(final FormMetadata formMetadata) {
55        final Annotation[] annotations = Annotations.getAnnotations(formMetadata
56              .getFormClass());
57        AnnotationHandler handler;
58        for (int i = 0; i < annotations.length; i++) {
59           handler = AnnotationHandlerRegistry.getInstance().get(annotations[i].annotationType());
60  
61           if (handler == null) {
62              continue;
63           }
64  
65           handler.processFormAnnotation(formMetadata,
66                 annotations[i]);
67        }
68     }
69  
70     private void processFieldsAnnotations(final FormMetadata formMetadata) {
71        final Class clazz = formMetadata.getFormClass();
72        final PropertyDescriptor[] propertyDescriptors = PropertyUtils
73              .getPropertyDescriptors(clazz);
74        PropertyDescriptor propDesc;
75        FieldMetadata fieldMetadata;
76        for (int i = 0; i < propertyDescriptors.length; i++) {
77           propDesc = propertyDescriptors[i];
78  
79           // Ignoring java.lang.Object.getClass()
80           if (propDesc.getName().equals("class")) {
81              continue;
82           }
83  
84           final Method readMethod = getReadMethod(clazz, propDesc);
85  
86           if (readMethod == null || Annotations.getAnnotation(NotBound.class, 
87                 readMethod) != null) {
88              continue;
89           }
90  
91           fieldMetadata = new FieldMetadata(propDesc.getName(), propDesc
92                 .getPropertyType(), propDesc.getWriteMethod() != null);
93           formMetadata.addFieldMetadata(propDesc.getName(), fieldMetadata);
94           processFieldAnnotations(formMetadata, fieldMetadata, readMethod);
95        }
96     }
97  
98     private void processFieldAnnotations(final FormMetadata formMetadata,
99           final FieldMetadata fieldMetadata, final Method fieldGetterMethod) {
100       final Annotation[] annotations = Annotations.getAnnotations(fieldGetterMethod);
101       AnnotationHandler handler;
102       for (int i = 0; i < annotations.length; i++) {
103          handler = AnnotationHandlerRegistry.getInstance().get(annotations[i].annotationType());
104 
105          if (handler == null) {
106             continue;
107          }
108 
109          handler.processFieldAnnotation(formMetadata,
110                fieldMetadata, annotations[i]);
111       }
112    }
113 
114    private void processMethodsAnnotations(final FormMetadata formMetadata) {
115       final Method[] methods = formMetadata.getFormClass().getMethods();
116       boolean isAction;
117       boolean isProvider;
118       MethodMetadata methodMetadata;
119 
120       for (int i = 0; i < methods.length; i++) {
121          isAction = Annotations.getAnnotation(Action.class, methods[i]) != null;
122          isProvider = Annotations.getAnnotation(DataProvider.class, methods[i]) != null;
123 
124          if (!isAction && !isProvider) {
125             continue;
126          }
127 
128          if (formMetadata.getFieldMetadata(methods[i].getName()) != null) {
129             throw new IllegalArgumentException("An @Action/@DataProvider " +
130                   "cannot have the same name of a property: " + 
131                   methods[i].getName());
132          }
133 
134          methodMetadata = new MethodMetadata(methods[i], isAction, isProvider);
135          formMetadata.addMethodMetadata(methods[i], methodMetadata);
136          processMethodAnnotations(formMetadata, methodMetadata, methods[i]);
137       }
138    }
139 
140    private void processMethodAnnotations(final FormMetadata formMetadata,
141          final MethodMetadata methodMetadata, final Method actionMethod) {
142       final Annotation[] annotations = Annotations.getAnnotations(actionMethod);
143       AnnotationHandler handler;
144       for (int i = 0; i < annotations.length; i++) {
145          handler = AnnotationHandlerRegistry.getInstance().get(annotations[i].annotationType());
146 
147          if (handler == null) {
148             continue;
149          }
150 
151          handler.processMethodAnnotation(formMetadata,
152                methodMetadata, annotations[i]);
153       }
154    }
155 
156    // Work around for a bug in JDK 1.4.2
157    private Method getReadMethod(Class clazz, PropertyDescriptor pd) {
158       Method method = pd.getReadMethod();
159 
160       if (method == null || clazz.equals(method.getDeclaringClass())) {
161          return method;
162       }
163 
164       try {
165          method = clazz.getMethod(method.getName(), 
166                method.getParameterTypes());
167          return (method == null) ? pd.getReadMethod() : method;
168       } catch (NoSuchMethodException nsme) {
169          return method;
170       }
171    }
