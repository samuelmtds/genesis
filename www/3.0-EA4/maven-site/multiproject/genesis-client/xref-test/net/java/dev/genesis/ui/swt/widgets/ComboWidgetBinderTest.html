86     protected void setKey(int index, String key) throws Exception {
87        combo.setData(SWTBinder.KEY_PROPERTY + '-' + index, key);
88     }
89  
90     protected String getKey(Object value) throws Exception {
91        String keyPropertyName = (String) combo.getData(SWTBinder.KEY_PROPERTY);
92  
93        if (keyPropertyName != null) {
94           Object o = (value == null) ? null : PropertyUtils.getProperty(value,
95                 keyPropertyName);
96  
97           return binder.format(getName(), keyPropertyName, o);
98        } else if (EnumHelper.getInstance().isEnum(value)) {
99           return value.toString();
100       }
101 
102       throw new PropertyMisconfigurationException("Property 'key' "
103             + "must be configured for the widget named " + getName());
104    }
105 
106    protected String getValue(Widget widget, Object value) throws Exception {
107       String valueProperty = (String) widget.getData(SWTBinder.VALUE_PROPERTY);
108       if (value instanceof String) {
109          return (String) value;
110       } else if (valueProperty == null) {
111          return binder.format(binder.getLookupStrategy().getName(widget), null,
112                value);
113       }
114 
115       return binder.format(binder.getLookupStrategy().getName(widget),
116             valueProperty, PropertyUtils.getProperty(value, valueProperty));
117    }

326    private void simulateSelect(int index) {
327       if (index < 0) {
328          combo.deselectAll();
329       } else {
330          combo.select(index);
331       }
332       
333       Event event = new Event();
334       event.widget = combo;
335       event.button = 1;
336       event.type = SWT.Selection;
337       combo.notifyListeners(event.type, event);
338    }
