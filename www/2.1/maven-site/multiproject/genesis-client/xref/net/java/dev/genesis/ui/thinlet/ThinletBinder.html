114    public void bind() throws Exception {
115       final FormMetadata formMetadata = getFormMetadata(form);
116       final Collection dataProviders = new ArrayList(formMetadata
117             .getDataProviderMetadatas().values());
118 
119       bindFieldMetadatas(formMetadata);
120       bindActionMetadatas(formMetadata);
121       bindDataProviders(dataProviders);
122 
123       setupController();
124    }

126    protected void setupController() throws Exception {
127       if (!controller.isSetup()) {
128          controller.addFormControllerListener(this);
129          controller.setup();
130       } else {
131          controller.fireAllEvents(this);
132          controller.addFormControllerListener(this);
133       }
134    }

136    protected void bindFieldMetadatas(final FormMetadata formMetadata) {
137       for (final Iterator i = formMetadata.getFieldMetadatas().entrySet()
138             .iterator(); i.hasNext(); ) {
139          final Map.Entry entry = (Map.Entry)i.next();
140          final String name = entry.getKey().toString();
141          final List components = findComponents(name);
142 
143          if (components.isEmpty()) {
144             log.warn(name + " could not be found while binding " + form.getClass());
145             continue;
146          }
147 
148          final Object mainComponent = components.get(0);
149          final String className = Thinlet.getClass(mainComponent);
150          final FieldMetadata fieldMetadata = (FieldMetadata)entry.getValue();
151 
152          if (Arrays.binarySearch(supportedFieldWidgets, className) < 0) {
153             if (fieldMetadata.isWriteable()) {
154                log.warn(name + " is not represented by an editable widget (" + 
155                      className + ")");
156             }
157 
158             continue;
159          }
160 
161          if (!fieldMetadata.isWriteable()) {
162             if (log.isDebugEnabled()) {
163                log.debug(name + " is a display only field, skipping...");
164             }
165             
166             continue;
167          }
168 
169          if (log.isTraceEnabled()) {
170             log.trace("Binding field " + name + " in " + 
171                   form.getClass().getName() + " to " + className);
172          }
173 
174          boundFieldNames.add(name);
175 
176          createWidgetGroup(mainComponent, name);
177 
178          for (final Iterator it = components.iterator(); it.hasNext(); ) {
179             final Object component = it.next();
180 
181             if (className.equals(BaseThinlet.CHECKBOX) ||
182                   className.equals(BaseThinlet.TOGGLE_BUTTON)) {
183                thinlet.setMethod(component, "action", "setValue(" + 
184                      thinlet.getName(component) + "," + 
185                      thinlet.getName(component) + 
186                      (thinlet.getGroup(component) == null ? ".name)" : ".group)"),
187                      root, this);
188             } else if (Arrays.binarySearch(fieldsChangedByAction, className) > -1) {
189                thinlet.setMethod(component, "action", "setValue(" + name + "," +
190                      name + ".name)", root, this);
191             } else {
192                thinlet.setMethod(component, "focuslost", "setValue(" + name + 
193                      "," + name + ".name)", root, this);
194             }
195          }
196       }
197    }

199    protected List findComponents(final String name) {
200       final Object widget = thinlet.find(root, name);
201 
202       if (widget != null) {
203          return Collections.singletonList(widget);
204       }
205 
206       if (groupComponents == null) {
207          groupComponents = thinlet.getAllOfClass(root, new String[] {
208                BaseThinlet.CHECKBOX, BaseThinlet.TOGGLE_BUTTON });
209       }
210 
211       final List components = new ArrayList();
212 
213       for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
214          final Object component = i.next();
215 
216          if (name.equals(thinlet.getGroup(component))) {
217             components.add(component);
218          }
219       }
220 
221       return components;
222    }

224    protected void createWidgetGroup(Object component, String name) {
225       final Collection enabledWidgetGroupCollection = new ArrayList();
226       final Collection visibleWidgetGroupCollection = new ArrayList();
227 
228       enabledWidgetGroupCollection.add(name);
229       visibleWidgetGroupCollection.add(name);
230 
231       final String widgetGroup = (String)thinlet.getProperty(component,
232             "widgetGroup");
233       final String enabledWidgetGroup = (String)thinlet.getProperty(component,
234             "enabledWidgetGroup");
235       final String visibleWidgetGroup = (String)thinlet.getProperty(component,
236             "visibleWidgetGroup");
237 
238       if (widgetGroup != null) {
239          final List widgets = Arrays.asList(widgetGroup.split(","));
240          enabledWidgetGroupCollection.addAll(widgets);
241          visibleWidgetGroupCollection.addAll(widgets);
242 
243          if (log.isDebugEnabled()) {
244             log.debug("Adding " + widgetGroup + " as a widget group for " +
245                   name + " in class " + form.getClass());
246          }
247       }
248       if (enabledWidgetGroup != null) {
249          enabledWidgetGroupCollection.addAll(Arrays.asList(enabledWidgetGroup.split(",")));
250 
251          if (log.isDebugEnabled()) {
252             log.debug("Adding " + enabledWidgetGroup + " as a enabled widget group for " +
253                   name + " in class " + form.getClass());
254          }
255       }
256       if (visibleWidgetGroup != null) {
257          visibleWidgetGroupCollection.addAll(Arrays.asList(visibleWidgetGroup.split(",")));
258 
259          if (log.isDebugEnabled()) {
260             log.debug("Adding " + visibleWidgetGroup + " as a visible widget group for " +
261                   name + " in class " + form.getClass());
262          }
263       }
264 
265       enabledWidgetGroupMap.put(name, enabledWidgetGroupCollection);
266       visibleWidgetGroupMap.put(name, visibleWidgetGroupCollection);
267    }

269    public void setValue(Object component, String name) throws Exception {
270       final String type = Thinlet.getClass(component);
271 
272       String value = null;
273 
274       if (type.equals(BaseThinlet.CHECKBOX) || type.equals(BaseThinlet.TOGGLE_BUTTON)) {
275          if (thinlet.getGroup(component) != null) {
276             if (!thinlet.isSelected(component)) {
277                return;
278             }
279          
280             value = name;
281          } else {
282             value = String.valueOf(thinlet.isSelected(component));
283          }
284       } else if (type.equals(BaseThinlet.COMBOBOX) || type.equals(BaseThinlet.LIST)) {
285          component = thinlet.getSelectedItem(component);
286          
287          value = component != null ? thinlet.getName(component) : null;
288       } else if (type.equals(BaseThinlet.PROGRESS_BAR) || type.equals(BaseThinlet.SLIDER)) {
289          value = String.valueOf(thinlet.getValue(component));
290       } else {
291          value = thinlet.getText(component);
292       }
293 
294       final Map properties = new HashMap();
295       properties.put(name, value);
296 
297       controller.populate(properties, converters);
298    }

300    protected void bindActionMetadatas(final FormMetadata formMetadata) {
301       for (final Iterator i = formMetadata.getActionMetadatas().entrySet()
302             .iterator(); i.hasNext(); ) {
303          final Map.Entry entry = (Map.Entry)i.next();
304          final String name = ((MethodEntry)entry.getKey()).getMethodName();
305          final Object component = thinlet.find(root, name);
306 
307          if (component == null) {
308             log.warn(name + " could not be found while binding " + form.getClass());
309             continue;
310          }
311 
312          final String className = Thinlet.getClass(component);
313 
314          if (!BaseThinlet.BUTTON.equals(className)) {
315             log.warn(name + " is not represented by a button; it is a " + 
316                   className);
317 
318             continue;
319          }
320 
321          createWidgetGroup(component, name);
322          
323          if (log.isTraceEnabled()) {
324             log.trace("Binding action " + name + " in " + form.getClass().getName());
325          }
326 
327          thinlet.setMethod(component, "action", "invokeAction(" + 
328                name + ".name)", root, this);
329       }
330    }

332    protected void bindDataProviders(final Collection dataProviders) {
333       for (final Iterator i = dataProviders.iterator(); i.hasNext();) {
334          final DataProviderMetadata dataProviderMetadata = 
335                (DataProviderMetadata)i.next();
336 
337          final String name = dataProviderMetadata.getWidgetName();
338          final Object component = thinlet.find(root, name);
339 
340          if (component == null) {
341             log.warn(name + " could not be found while binding data provider in " + 
342                   form.getClass());
343             continue;
344          }
345 
346          final String className = Thinlet.getClass(component);
347 
348          if (!(className.equals(BaseThinlet.TABLE) || 
349                className.equals(BaseThinlet.COMBOBOX) ||
350                className.equals(BaseThinlet.LIST))) {
351             log.warn(className + " is not a supported widget for binding a " +
352                   "data provider");
353             continue;
354          }
355 
356          createWidgetGroup(component, name);
357 
358          if (log.isDebugEnabled()) {
359             log.debug("Binding an action method for " + name);
360          }
361 
362          thinlet.setMethod(component, "action", "updateSelection(" + name
363                + ".name," + name + ")", root, this);
364          dataProvided.put(name, dataProviderMetadata);
365       }
366    }

368    public void updateSelection(String name, Object widget) throws Exception {
369       int[] selected = thinlet.getSelectedIndexes(widget);
370       
371       final String widgetClass = BaseThinlet.getClass(widget);
372 
373       if (selected.length >= 1 && (BaseThinlet.COMBOBOX.equals(widgetClass) ||
374             BaseThinlet.LIST.equals(widgetClass)) && isBlank(widget, name)) {
375          boolean remove = false;
376 
377          for (int i = 0; i < selected.length; i++) {
378             selected[i] = selected[i] - 1;
379 
380             if (selected[i] == -1) {
381                remove = true;
382             }
383          }
384 
385          if (remove) {
386             if (selected.length == 1) {
387                selected = new int[0];
388             } else {
389                final int[] aux = selected;
390                selected = new int[aux.length - 1];
391                System.arraycopy(aux, 1, selected, 0, aux.length - 1);
392             }
393          }
394       }
395 
396       final DataProviderMetadata dataMeta = (DataProviderMetadata)dataProvided
397             .get(name);
398 
399       controller.updateSelection(dataMeta, selected);
400    }

402    public void invokeAction(String name) throws Exception {
403       Map stringProperties = null;
404 
405       if (name != null) {
406          final MethodMetadata methodMetadata = getMethodMetadata(name);
407 
408          if (methodMetadata != null && 
409                methodMetadata.getActionMetadata() != null && 
410                methodMetadata.getActionMetadata().isValidateBefore()) {
411             stringProperties = getStringProperties();
412             thinlet.populate(null, root, stringProperties, false);
413          }
414       }
415 
416       controller.invokeAction(name, stringProperties);
417    }

456    public void dataProvidedListChanged(DataProviderMetadata metadata, List items)
457          throws Exception {
458       final String name = metadata.getWidgetName();
459       final Object component = thinlet.find(root, name);
460 
461       if (component == null) {
462          log.warn(name + " could not be found while populating using "
463                + form.getClass());
464          return;
465       }
466 
467       if (log.isDebugEnabled()) {
468          log.debug("Populating " + name + " with " + metadata.getName());
469       }
470 
471       final String className = Thinlet.getClass(component);
472 
473       if (className.equals(BaseThinlet.TABLE)) {
474          resetSelectedFields(metadata);
475          thinlet.populateFromCollection(component, items, formatters);
476       } else if (className.equals(BaseThinlet.COMBOBOX) || 
477                className.equals(BaseThinlet.LIST)) {
478          //TODO: This parsing shouldn't occur every time
479          final String key = (String) thinlet.getProperty(component, "key");
480 
481          if (key == null) {
482             throw new PropertyMisconfigurationException("Property 'key' "
483                   + "must be configured for the widget named " + name);
484          }
485 
486          final String value = (String)thinlet.getProperty(component, "value");
487          final boolean blank = isBlank(component, name);
488          final String blankLabel = (String)thinlet.getProperty(component, 
489                "blankLabel");
490 
491          thinlet.populateFromCollection(component, items, key, value, blank, 
492                blankLabel, formatters);
493       } else {
494          throw new UnsupportedOperationException(className + " is not "
495                + "supported for data providing");
496       }
497    }

508    private boolean isBlank(Object component, String name) {
509       final Object blankObject = thinlet.getProperty(component, "blank");
510       boolean blank;
511 
512       if (blankObject == null) {
513          blank = false;
514       } else if (blankObject instanceof String) {
515          blank = Boolean.valueOf(blankObject.toString()).booleanValue();
516       } else if (blankObject instanceof Boolean) {
517          blank = ((Boolean) blankObject).booleanValue();
518       } else {
519          throw new PropertyMisconfigurationException("Property 'blank' "
520                + "for the widget named " + name + " must either be left "
521                + "empty or contain a boolean value");
522       }
523 
524       return blank;
525    }

535    protected void conditionsChanged(Map updatedConditions, boolean enabled) {
536       for (final Iterator i = updatedConditions.entrySet().iterator();
537             i.hasNext(); ) {
538          final Map.Entry entry = (Map.Entry)i.next();
539 
540          final Collection widgetGroup = getWidgetGroupCollection(
541                entry.getKey().toString(), enabled);
542 
543          if (widgetGroup == null) {
544             log.warn(entry.getKey() + (enabled ? " enabled" : " visible")
545                   + " state should be changed, "
546                   + "but could not be found in the view component for "
547                   + form.getClass());
548             continue;
549          }
550 
551          if (log.isDebugEnabled()) {
552             log.debug("Changing " + entry.getKey()
553                   + (enabled ? " enabled" : " visible") + " state to "
554                   + entry.getValue());
555          }
556 
557          for (Iterator it = widgetGroup.iterator(); it.hasNext(); ) {
558             final Object widget = thinlet.find(root, it.next().toString());
559 
560             if (widget == null) {
561                log.warn(entry.getKey() + (enabled ? " enabled" : " visible")
562                      + " state should be changed, "
563                      + "but could not be found in the view component for "
564                      + form.getClass());
565                continue;
566             }
567 
568             if (enabled) {
569                thinlet.setEnabled(widget, ((Boolean)entry.getValue())
570                      .booleanValue());
571             } else {
572                thinlet.setVisible(widget, ((Boolean)entry.getValue())
573                      .booleanValue());
574             }
575          }
576       }
577    }
