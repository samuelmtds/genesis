150                                                 throws ScreenNotFoundException {
151       try {
152          InputStream is = getResourceAsStream(fileName, 
153                                              handler.getClass().getClassLoader());
154 
155          if (is == null) {
156             throw new ScreenNotFoundException(fileName);
157          }
158          
159          handler.setScreen(parse(is, handler));
160 
161          is.close();
162 
163          return handler.getScreen();
164       } catch (IOException ioe) {
165          throw new ScreenNotFoundException(fileName);
166       }
167    }

376    protected void displayBean(Object bean, Object root, Map formatters) 
377          throws IllegalAccessException, InvocationTargetException, 
378                 NoSuchMethodException {
379       final Map properties = bean instanceof Map ? (Map)bean : 
380             PropertyUtils.describe(bean);
381 
382       if (root instanceof Thinlet) {
383          root = ((Thinlet)root).getDesktop();
384       }
385 
386       String propertyName;
387       String propertyValue = null;
388       String type;
389       Object component;
390       Collection groupComponents = null;
391 
392       for (final Iterator propertyNames = properties.keySet().iterator();
393                                                    propertyNames.hasNext(); ) {
394          propertyName = propertyNames.next().toString();
395          propertyValue = null;
396          component = find(root, propertyName);
397 
398          if (component == null) {
399             if (groupComponents == null) {
400                groupComponents = getAllOfClass(root, CHECKBOX);
401             }
402 
403             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
404                component = i.next();
405 
406                if (propertyName.equals(getGroup(component)) && 
407                       getName(component) != null ) {
408                   if (propertyValue == null) {
409                      propertyValue = getPropertyValue(properties, propertyName, 
410                            true, formatters);
411                   }
412 
413                   if (getName(component).equals(propertyValue)) {
414                      setSelected(component, true);
415                      break;
416                   }
417                }
418             }
419 
420             continue;
421          }
422 
423          type = getClass(component);
424 
425          if (type.equals(COMBOBOX) || (type.equals(LIST) && getSelection(component).equals(SINGLE))) {
426             Object selectedComponent = getSelectedItem(component);
427 
428             if (selectedComponent != null && type.equals(LIST)) {
429                setSelected(selectedComponent, false);
430             }
431 
432             if (type.equals(COMBOBOX)) {
433                setSelected(component, -1);
434                setText(component, "");
435             }
436 
437             String valuePropertyName = (String)getProperty(component, "value");
438 
439             if (valuePropertyName != null) {
440                Object o = properties.get(propertyName);
441                o = o == null ? null : PropertyUtils.getProperty(
442                      properties.get(propertyName), valuePropertyName);
443 
444                propertyValue = format(formatters, valuePropertyName, o);
445             } else {
446                propertyValue = getPropertyValue(properties, propertyName, true, 
447                      formatters);
448             }
449 
450             selectedComponent = find(component, propertyValue);
451 
452             if (selectedComponent != null) {
453                if (type.equals(LIST)) {
454                   setSelected(selectedComponent, true);
455                } else {
456                   setSelected(component, getIndexOf(component, selectedComponent));
457                }
458             }
459          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
460             setValue(component, Integer.parseInt(getPropertyValue(properties, propertyName, 
461                   false, formatters)));
462          } else if (type.equals(PANEL)) {
463             displayBean(properties.get(propertyName), component);
464          } else if (type.equals(CHECKBOX)) {
465             setSelected(component, Boolean.TRUE.equals(properties.get(propertyName)));
466          } else if (!type.equals(TABLE) && !type.equals(LIST)){
467             setText(component, getPropertyValue(properties, propertyName, 
468                   false, formatters));
469          }
470       }
471    }

473    protected String getPropertyValue(Map properties, String propertyName,
474          boolean enumKey, Map formatters) {
475       Object o = properties.get(propertyName);
476 
477       if (enumKey && o instanceof Enum) {
478         return o.toString();
479       }
480 
481       return format(formatters, propertyName, o);
482    }

489       }

492            formatter.format(value);

519    protected void populate(Object bean, Object root, final Map properties, 
520          final boolean changeBean) throws IllegalAccessException, 
521          InvocationTargetException, NoSuchMethodException {
522       if (changeBean) {
523          properties.putAll(BeanUtils.describe(bean));
524       }
525 
526       if (root instanceof Thinlet) {
527          root = ((Thinlet)root).getDesktop();
528       }
529 
530       String propertyName;
531       String type;
532       Object component;
533       Collection groupComponents = null;
534 
535       for (final Iterator propertyNames = properties.keySet().iterator();
536                                                    propertyNames.hasNext(); ) {
537          propertyName = propertyNames.next().toString();
538          component = find(root, propertyName);
539 
540          if (component == null) {
541             if (groupComponents == null) {
542                groupComponents = getAllOfClass(root, CHECKBOX);
543             }
544 
545             for (final Iterator  i = groupComponents.iterator(); i.hasNext(); ) {
546                component = i.next();
547 
548                if (propertyName.equals(getGroup(component)) && 
549                      isSelected(component)) {
550                   properties.put(propertyName, getName(component));
551                   break;
552                }
553             }
554 
555             continue;
556          }
557 
558          type = getClass(component);
559 
560          if (type.equals(COMBOBOX) || type.equals(LIST)) {
561             component = getSelectedItem(component);
562 
563             if (component != null) {
564                properties.put(propertyName, getName(component));
565             } else {
566                properties.put(propertyName, null);
567             }
568          } else if (type.equals(PROGRESS_BAR) || type.equals(SLIDER)) {
569             properties.put(propertyName, String.valueOf(getValue(component)));
570          } else if (type.equals(TABLE)) {
571             // skip it intentionally
572          } else {
573             properties.put(propertyName, getText(component));
574          }
575       }
576 
577       if (changeBean) {
578          BeanUtils.populate(bean, properties);
579 
580          if (bean instanceof Form) {
581             ((Form)bean).afterPopulate();
582          }
583       }
584    }

621    protected void populateFromEnum(Object component, Class clazz, boolean blank) {
622       final boolean combobox = getClass(component).equals(COMBOBOX);
623 
624       if (!combobox && !getClass(component).equals(LIST)) {
625          throw new UnsupportedOperationException();
626       }
627 
628       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
629 
630       String key;
631       Object enumInstance;
632 
633       removeAll(component);
634 
635       if (blank) {
636          add(component, createItemOfType("", "", type));
637       }
638 
639       for (final Iterator i = Enum.getInstances(clazz).iterator(); i.hasNext(); ) {
640          enumInstance = i.next();
641          key = enumInstance.toString();
642 
643          add(component, createItemOfType(key, FormatterRegistry.getInstance()
644                .format(enumInstance), type));
645       }
646    }

679    protected void populateFromCollection(Object component, Collection c, 
680                         String keyProperty, String valueProperty, boolean blank,
681                         String blankLabel, Map formatters) 
682          throws IllegalAccessException, InvocationTargetException, 
683                 NoSuchMethodException {
684       final boolean combobox = getClass(component).equals(COMBOBOX);
685 
686       if (!combobox && !getClass(component).equals(LIST)) {
687          throw new UnsupportedOperationException();
688       }
689 
690       final ItemType type = combobox ? ItemType.CHOICE : ItemType.ITEM;
691 
692       String key;
693       String description;
694       Object o;
695 
696       removeAll(component);
697 
698       if (blank) {
699          add(component, createItemOfType("", blankLabel == null ? 
700                "" : blankLabel, type));
701       }
702 
703       final String componentName = getName(component);
704 
705       for (final Iterator i = c.iterator(); i.hasNext(); ) {
706          o = i.next();
707 
708          key = format(formatters, componentName + '.' + keyProperty, PropertyUtils
709                .getProperty(o, keyProperty));
710          description = valueProperty == null ? 
711                format(formatters, componentName + '.', o) : format(formatters, 
712                componentName + '.' + valueProperty, PropertyUtils.getProperty(o, 
713                valueProperty));
714 
715          add(component, createItemOfType(key, description, type));
716       }
717    }

731    protected void populateFromCollection(Object component, Collection c, 
732          Map formatters) throws IllegalAccessException, 
733          InvocationTargetException, NoSuchMethodException {
734       if (!getClass(component).equals(TABLE)) {
735          throw new UnsupportedOperationException();
736       }
737 
738       removeAll(component);
739 
740       final Collection propertyNames = new ArrayList();
741       final Object[] columns = getTableColumns(component); 
742       
743       for (int i = 0; i < columns.length; i++) {
744          final String name = getName(columns[i]);
745 
746          if (name == null) {
747             throw new IllegalArgumentException("column (index " + i + ") in " + 
748                   "table " + getName(component) + " does not have a name");
749          }
750 
751          propertyNames.add(name);
752       }
753 
754       final String componentName = getName(component);
755 
756       String propertyName;
757       Object row;
758       Object bean;
759       int indexOfDot;
760       boolean skip;
761 
762       for (final Iterator i = c.iterator(); i.hasNext(); ) {
763          bean = i.next();
764          row = createRow();
765 
766          for (final Iterator it = propertyNames.iterator(); it.hasNext(); ) {
767             propertyName = it.next().toString();
768 
769             indexOfDot = 0;
770             skip = false;
771 
772             while ((indexOfDot = propertyName.indexOf('.', indexOfDot)) != -1) {
773                if (PropertyUtils.getProperty(bean, 
774                      propertyName.substring(0, indexOfDot)) == null) {
775                   add(row, createCell(propertyName, 
776                         format(formatters, componentName + '.' + propertyName, 
777                         null)));
778                   skip = true;
779                   break;
780                }
781 
782                indexOfDot += 1;
783             }
784 
785             if (skip) {
786                continue;
787             }
788

821          Object handler) throws Exception {
822       if (widget instanceof Thinlet) {
823          widget = ((Thinlet)widget).getDesktop();

830          formPerClassPerComponent.put(widget, formPerClass);
831       } else {

834          if (oldForm != null) {

975    public void handleException(Throwable throwable) {
976       if (throwable instanceof ScreenNotFoundException) {
977          handleException("The error screen file could not be found", throwable);
978 
979          return;
980       } else if (throwable instanceof ValidationException) {
981          showValidationErrors((ValidationException)throwable);
982          return;
983       } else if (throwable instanceof UIException) {
984          handleUIException((UIException)throwable);
985          return;

987 
988       try {
989          if (handleCustomException(throwable)) {
990             return;
991          }
992       } catch (Throwable t) {
993          LogFactory.getLog(getClass()).error("Unknown exception", t);
994       }
995 
996       handleUnknownException(throwable);
997    }
