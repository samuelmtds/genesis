154    protected FormState createFormState(FormState state) {
155       Map clonedValues = new HashMap(state.getValuesMap().size());
156 
157       for (Iterator iter = state.getValuesMap().entrySet().iterator(); iter
158             .hasNext();) {
159          Map.Entry entry = (Map.Entry)iter.next();
160 
161          FieldMetadata fieldMeta = formMetadata.getFieldMetadata(entry.getKey()
162                .toString());
163 
164          clonedValues.put(entry.getKey(), fieldMeta.getCloner().clone(
165                entry.getValue()));
166       }
167 
168       return new FormStateImpl(state, clonedValues);
169    }

171    public void setup() throws Exception {
172       if (setup) {
173          throw new IllegalStateException("setup() has already been called");
174       }
175 
176       setup = true;
177 
178       JXPathContext ctx = createJXPathContext();
179       ctx.getVariables().declareVariable(FORM_METADATA_KEY, formMetadata);
180 
181       currentState = createFormState();
182       ctx.getVariables().declareVariable(CURRENT_STATE_KEY, currentState);
183 
184       setContext(ctx);
185 
186       if (PropertyUtils.isWriteable(getForm(), "context")
187             && Map.class.isAssignableFrom(PropertyUtils.getPropertyType(getForm(),
188                   "context"))) {
189          PropertyUtils.setProperty(getForm(), "context", getVariablesMap());
190       }
191 
192       evaluate(true);
193    }

219    protected void populate(Map properties, boolean stringMap, Map converters) 
220          throws Exception {
221 
222       final boolean createPreviousState = createPreviousState();
223 
224       try {
225          if (properties == null) {
226             properties = PropertyUtils.describe(getForm());
227          }
228 
229          updateChangedMap(properties, stringMap, converters);
230 
231          if (currentState.getChangedMap().isEmpty()) {
232             log.debug("Nothing changed.");
233 
234             return;
235          }
236 
237          if (stringMap) {
238             PropertyUtils.copyProperties(getForm(), currentState.getChangedMap());
239          }
240 
241          evaluate(false);
242       } catch (Exception e) {
243          resetPreviousState(createPreviousState);
244 
245          throw e;
246       } finally {
247          releasePreviousState(createPreviousState);
248       }
249    }

251    protected boolean updateChangedMap(Map newData, boolean stringMap, 
252             Map converters) {
253       boolean changed = false;
254 
255       Object value;
256       FieldMetadata fieldMeta;
257       Map.Entry entry;
258 
259       final Map changedMap = currentState.getChangedMap();
260 
261       changedMap.clear();
262       changedMap.putAll(GenesisUtils.normalizeMap(newData));
263 
264       if (log.isDebugEnabled()) {
265          log.debug("changedMap started as: " + changedMap);
266       }
267 
268       for (final Iterator i = changedMap.entrySet().iterator(); i.hasNext();) {
269          entry = (Map.Entry) i.next();
270          fieldMeta = formMetadata.getFieldMetadata(entry.getKey().toString());
271 
272          if (fieldMeta == null) {
273             continue;
274          }
275 
276          value = (stringMap) ? getConverter(fieldMeta, converters).convert(
277                fieldMeta.getFieldClass(), entry.getValue()) : entry.getValue();
278 
279          if (previousState != null && fieldMeta.getEqualityComparator().equals(
280                value, previousState.getValuesMap().get(entry.getKey()))
281                && fieldMeta.getEqualityComparator().equals(value,
282                      currentState.getValuesMap().get(entry.getKey()))) {
283             i.remove();
284 
285             if (log.isDebugEnabled()) {
286                log.debug(entry.getKey() + " removed from changedMap for being "
287                      + " equal to previousValue; is " + entry.getValue()
288                      + "; was:" + previousState.getValuesMap().get(entry
289                      .getKey()));
290             }
291 
292             continue;
293          }
294 
295          if (log.isDebugEnabled()) {
296             log.debug("Field '" + entry.getKey() + "' changed to '"
297                   + entry.getValue() + "'; was " + (previousState == null ? 
298                   "undefined" : previousState.getValuesMap().get(entry
299                   .getKey())));
300          }
301 
302          entry.setValue(value);
303          currentState.getValuesMap().put(entry.getKey(), entry.getValue());
304          
305          changed = true;
306       }
307 
308       return changed;
309    }

311    private Converter getConverter(FieldMetadata fieldMetadata, Map converters) {
312       if (converters == null) {
313          return fieldMetadata.getConverter();
314       }
315 
316       final Converter converter = (Converter)converters.get(
317             fieldMetadata.getName());
318 
319       return (converter == null) ? fieldMetadata.getConverter() : converter;
320    }

322    protected void evaluate(boolean firstCall) throws Exception {
323       boolean changed = true;
324 
325       for (int times = 0; changed && times < getMaximumEvaluationTimes(); 
326             times++) {
327          changed = doEvaluate(firstCall);
328       }
329    }

331    protected boolean doEvaluate(boolean firstCall) throws Exception {
332       boolean changed = false;
333       evaluateNamedConditions();
334 
335       if (evaluateClearOnConditions()) {
336          changed = true;
337          evaluateNamedConditions();
338       }
339 
340       if (evaluateCallWhenConditions(firstCall)) {
341          changed = true;
342          evaluateNamedConditions();
343       }
344 
345       if (evaluateDataProvidedIndexes(firstCall)) {
346          changed = true;
347          evaluateNamedConditions();
348       }
349 
350       final Map newData = PropertyUtils.describe(getForm());
351       changed |= updateChangedMap(newData, false, null);
352 
353       fireValuesChanged(currentState.getChangedMap());
354 
355       changed |= evaluateEnabledWhenConditions();
356       changed |= evaluateVisibleWhenConditions();
357       
358       return changed;
359    }

367    protected boolean evaluateClearOnConditions() throws Exception {
368       final Map fieldMetadatas = formMetadata.getFieldMetadatas();
369       final Map dataProviderMetadatas = formMetadata.getDataProviderMetadatas();
370       final Map toCopy = new HashMap();
371 
372       boolean changed;
373       FieldMetadata fieldMetadata;
374       DataProviderMetadata dataProviderMeta;
375       List items;
376       Map.Entry entry;
377       final Map properties = PropertyUtils.describe(getForm());
378 
379       do {
380          changed = false;
381          properties.putAll(toCopy);
382 
383          for (final Iterator i = fieldMetadatas.entrySet().iterator(); i
384                .hasNext();) {
385             entry = (Map.Entry) i.next();
386             fieldMetadata = (FieldMetadata) entry.getValue();
387 
388             if (fieldMetadata.getClearOnCondition() == null
389                   || fieldMetadata.getEmptyResolver().isEmpty(
390                         properties.get(entry.getKey()))) {
391                i.remove();
392                continue;
393             }
394 
395             if (isConditionSatisfied(fieldMetadata.getClearOnCondition())) {
396                if (log.isDebugEnabled()) {
397                   log.debug("ClearOn Condition for field '" + entry.getKey()
398                         + "' satisfied. Setting '"
399                         + fieldMetadata.getEmptyValue() + "'");
400                }
401                changed = true;
402                toCopy.put(entry.getKey(), fieldMetadata.getEmptyValue());
403                i.remove();
404             }
405          }
406          
407          for (Iterator i = dataProviderMetadatas.entrySet().iterator(); 
408                i.hasNext();) {
409             entry = (Map.Entry) i.next();
410             dataProviderMeta = (DataProviderMetadata) entry.getValue();
411             
412             if (dataProviderMeta.getClearOnCondition() == null) {
413                i.remove();
414                continue;
415             }
416 
417             if (isConditionSatisfied(dataProviderMeta.getClearOnCondition())) {
418                if (log.isDebugEnabled()) {
419                   log.debug("ClearOn Condition for data provider '" + 
420                         entry.getKey() + "' satisfied. Clearing data " +
421                         "provider list.");
422                }
423 
424                changed = true;
425                updateDataProviderCurrentSelection(dataProviderMeta,
426                      items = EMPTY_LIST, EMPTY_INT_ARRAY);
427                fireDataProvidedListMetadataChanged(dataProviderMeta, items, false);
428                i.remove();
429             }
430          }
431       } while (changed);
432 
433       if (toCopy.isEmpty()) {
434          if (log.isDebugEnabled()) {
435             log.debug("Nothing changed in ClearOn conditions.");
436          }
437 
438          return false;
439       }
440 
441       PropertyUtils.copyProperties(getForm(), toCopy);
442       currentState.getChangedMap().putAll(toCopy);
443 
444       return true;
445    }

478    protected boolean evaluateConditions(boolean enabled) {
479       Map.Entry entry;
480 
481       final Map updatedConditions = new HashMap();
482       final Map toEvaluate = new HashMap(formMetadata.getFieldMetadatas());
483       toEvaluate.putAll(formMetadata.getActionMetadatas());
484 
485       for (final Iterator i = toEvaluate.entrySet().iterator(); i.hasNext();) {
486          entry = (Map.Entry) i.next();
487          evaluateCondition((MemberMetadata) entry.getValue(), updatedConditions, enabled);
488       }
489 
490       if (enabled) {
491          fireEnabledConditionChanged(updatedConditions);
492       } else {
493          fireVisibleConditionChanged(updatedConditions);
494       }
495 
496       return !updatedConditions.isEmpty();
497    }

499    protected void evaluateCondition(MemberMetadata memberMetadata, Map updatedConditions, boolean enabled) {
500       if ((enabled && memberMetadata.getEnabledCondition() == null)
501             || (!enabled && memberMetadata.getVisibleCondition() == null)) {
502          return;
503       }
504 
505       final Object newValue = isSatisfied(enabled ? memberMetadata.getEnabledCondition()
506             : memberMetadata.getVisibleCondition());
507       
508       final Map conditionMap = enabled ?
509             currentState.getEnabledMap() : currentState.getVisibleMap();
510 
511       if (!newValue.equals(conditionMap.get(memberMetadata
512             .getName()))) {
513          conditionMap.put(memberMetadata.getName(), newValue);
514          updatedConditions.put(memberMetadata.getName(), newValue);
515       }
516 
517       if (log.isDebugEnabled()) {
518          log.debug((enabled ? "EnabledWhen" : "VisibleWhen")
519                + " Condition for member '" + memberMetadata.getName()
520                + "' evaluated as '"
521                + conditionMap.get(memberMetadata.getName()) + "'");
522       }
523    }

563    protected boolean evaluateDataProvidedIndex(DataProviderMetadata dataMeta,
564          Map currentMap) throws Exception {
565       final Object indexes = currentMap.get(dataMeta.getIndexField()
566             .getFieldName());
567 
568       if (indexes == null) {
569          if (log.isDebugEnabled()) {
570             log.debug("Index field " + dataMeta.getIndexField().getFieldName()
571                   + " haven't been changed.");
572          }
573 
574          return false;
575       }
576 
577       final int[] selectedIndexes = dataMeta.getSelectedIndexes(indexes);
578 
579       final Object objectFieldValue = dataMeta.populateSelectedFields(getForm(),
580             (List)currentState.getDataProvidedMap().get(dataMeta),
581             selectedIndexes);
582       fireDataProvidedIndexesChanged(dataMeta, selectedIndexes);
583 
584       currentState.getDataProvidedIndexesMap().put(dataMeta, selectedIndexes);
585 
586       if (dataMeta.getObjectField() != null) {
587          currentState.getChangedMap().put(
588                dataMeta.getObjectField().getFieldName(), objectFieldValue);
589       }
590 
591       return true;
592    }

594    protected boolean invokeAction(MethodMetadata methodMetadata, 
595          boolean firstCall, boolean conditionally) throws Exception {
596       List items = null;
597       DataProviderMetadata dataProviderMeta = methodMetadata
598             .getDataProviderMetadata();
599 
600       if ((methodMetadata.getCallCondition() == null || firstCall) 
601             && conditionally) {
602          if (dataProviderMeta == null || (conditionally && !firstCall)) {
603             return false;
604          }
605 
606          if (!dataProviderMeta.isCallOnInit()) {
607             updateDataProviderCurrentSelection(dataProviderMeta,
608                   items = EMPTY_LIST, EMPTY_INT_ARRAY);
609             fireDataProvidedListMetadataChanged(dataProviderMeta, items, false);
610 
611             return true;
612          }
613          conditionally = false;
614       }
615 
616       final boolean satisfied = !conditionally || isConditionSatisfied(
617             methodMetadata.getCallCondition());
618 
619       if (satisfied) {
620          if (dataProviderMeta == null && !beforeInvokingMethod(methodMetadata)) {
621             return false;
622          }
623 
624          final Object ret = methodMetadata.invoke(getForm());
625 
626          if (dataProviderMeta != null) {
627             items = (ret.getClass().isArray()) ? Arrays.asList((Object[]) ret) :
628                  (List)ret;
629             currentState.getDataProvidedMap().put(dataProviderMeta, items);
630             fireDataProvidedListMetadataChanged(dataProviderMeta, items, firstCall);
631          } 
632 
633          if (methodMetadata.getActionMetadata() != null) {
634             afterInvokingMethod(methodMetadata);
635          }
636       }
637 
638       if (log.isDebugEnabled()) {
639          log.debug("CallWhen Condition for method '" + 
640                methodMetadata.getName() + "' evaluated as '" + 
641                satisfied + "'");
642       }
643 
644       return satisfied;
645    }

647    public void resetSelection(DataProviderMetadata meta, List dataProvided)
648          throws Exception {
649       int[] selected;
650 
651       if (meta.isResetSelection()) {
652          meta.resetSelectedFields(getForm());
653          selected = EMPTY_INT_ARRAY;
654       } else {
655          int[] currentSelection = (int[])currentState
656                .getDataProvidedIndexesMap().get(meta);
657          selected = meta.retainSelectedFields(getForm(), dataProvided,
658                currentSelection == null ? EMPTY_INT_ARRAY : currentSelection);
659       }
660 
661       currentState.getDataProvidedIndexesMap().put(meta, selected);
662    }

667    }

722    public void reset(FormState state) throws Exception {
723       if (log.isDebugEnabled()) {
724          log.debug("Reseting form '" + getForm() + "'");
725       }
726 
727       currentState = state = createFormState(state);
728       getContext().getVariables().declareVariable(CURRENT_STATE_KEY, currentState);
729 
730       fireEnabledConditionChanged(state.getEnabledMap());
731       fireVisibleConditionChanged(state.getVisibleMap());
732 
733       for (final Iterator i = state.getDataProvidedMap().entrySet().iterator(); 
734             i.hasNext();) {
735          final Map.Entry entry = (Map.Entry)i.next();
736          fireDataProvidedListMetadataChanged((DataProviderMetadata)entry
737                .getKey(), (List)entry.getValue(), false);
738       }
739 
740       final Map currentValues = PropertyUtils.describe(getForm());
741 
742       for (final Iterator i = formMetadata.getDataProviderIndexes().values()
743             .iterator(); i.hasNext(); ) {
744          DataProviderMetadata dataMeta = (DataProviderMetadata) i.next();
745          String fieldName = dataMeta.getIndexField().getFieldName();
746          Object indexes = currentValues.get(fieldName);
747 
748          if (getFormMetadata().getFieldMetadata(fieldName)
749                .getEqualityComparator().equals(indexes, state.getValuesMap()
750                .get(fieldName))) {
751             continue;
752          }
753 
754          fireDataProvidedIndexesChanged(dataMeta, 
755                dataMeta.getSelectedIndexes(indexes));
756       }
757 
758 
759       for (final Iterator i = currentValues.entrySet().iterator(); 
760             i.hasNext();) {
761          Map.Entry entry = (Map.Entry)i.next();
762          FieldMetadata fieldMeta = formMetadata.getFieldMetadata(entry.getKey()
763                .toString());
764 
765          if (fieldMeta == null || fieldMeta.getEqualityComparator().equals(
766                entry.getValue(), state.getValuesMap().get(entry.getKey()))) {
