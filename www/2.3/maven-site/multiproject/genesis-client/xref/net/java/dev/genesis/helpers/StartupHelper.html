44  public class StartupHelper {
45     private static Log log = LogFactory.getLog(StartupHelper.class);
46     private BeanUtilsBean beanUtilsBean = new BeanUtilsBean(
47           new ConverterRegistry(), new PropertyUtilsBean());
48     private String jxpathContextFactoryClassName =
49           "net.java.dev.genesis.commons.jxpath.JXPathContextFactory";
50     private final Map converters = new HashMap();
51     private final Map formatters = new HashMap();
52     private Locale locale;
53     private String dateFormat;
54     private boolean runNoopCommand = true;
55     private boolean loadValidatorRules = true;

74     public void setDateFormat(String dateFormat) {
75        this.dateFormat = dateFormat;

77        addConverter(Date.class, new DateLocaleConverter(null, locale, dateFormat));
78        addFormatter(Date.class, new FormatAdapter(new SimpleDateFormat(dateFormat),
79              true));
80     }

129    protected void registerConverters() {
130       log.info("Registering converters");
131 
132       for (final Iterator i = converters.entrySet().iterator(); i.hasNext(); ) {
133          Map.Entry entry = (Map.Entry)i.next();
134 
135          ConvertUtils.register((Converter)entry.getValue(), 
136                (Class)entry.getKey());
137       }
138    }

156    protected void registerFormatters() {
157       log.info("Registering formatters");
158 
159       for (final Iterator i = formatters.entrySet().iterator(); i.hasNext(); ) {
160          Map.Entry entry = (Map.Entry)i.next();
161 
162          FormatterRegistry.getInstance().register((Class)entry.getKey(),

167    public boolean isRunNoopCommand() {
168       return runNoopCommand;
169    }
170 
171    public void setRunNoopCommand(boolean runNoopCommand) {
172       this.runNoopCommand = runNoopCommand;
173    }
174 
175    protected void runNoopCommand() {
176       if (!isRunNoopCommand()) {
177          return;
178       }
179 
180       final Thread t = new Thread("NoopCommand-Daemon") {
181          {
182             setDaemon(true);
183          }
184 
185          public void run() {
186             new NoopCommand().remotable();
187             log.info("Noop command executed");
188          }
189       };
190 
191       t.start();
192    }
193 
194    public boolean isLoadValidatorRules() {
195       return loadValidatorRules;
196    }
197 
198    public void setLoadValidatorRules(boolean loadValidatorRules) {
199       this.loadValidatorRules = loadValidatorRules;
200    }
201 
202    protected void loadValidatorRules() {
203       if (!isLoadValidatorRules()) {
204          return;
205       }
206 
207       final Thread t = new Thread("ValidatorRulesLoader-Daemon") {
208          {
209             setDaemon(true);
210          }
211 
212          public void run() {
213             ValidationUtils.getInstance();
214             log.info("Validation rules loaded");
215          }
216       };
217 
218       t.start();
219    }
220 
221    public void initialize() {
222       registerBeanUtilsBean();
223       registerJXPathContextFactory();
224       registerConverters();
225       registerFormatters();
226       runNoopCommand();
227       loadValidatorRules();
228 
229       log.info("Initialization started");
230    }
