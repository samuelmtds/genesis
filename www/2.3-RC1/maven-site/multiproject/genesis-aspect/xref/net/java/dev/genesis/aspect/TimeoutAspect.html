36  public class TimeoutAspect {
37     private static final Log log = LogFactory.getLog(TimeoutAspect.class);
38  
39     private static final class LiveThreadWorkerThreadPair 
40           extends PhantomReference {
41        private final WeakReference reference;
42  
43        public LiveThreadWorkerThreadPair(Thread parentThread, ReferenceQueue queue,
44              WorkerThread workerThread) {
45           super(parentThread, queue);
46           reference = new WeakReference(workerThread);
47        }
48  
49        public WorkerThread getWorkerThread() {
50           return (WorkerThread)reference.get();
51        }
52     }

54     private static final class WorkerThreadDisposer extends Thread {
55        private static WorkerThreadDisposer instance;
56        private final ReferenceQueue queue = new ReferenceQueue();
57        private final Collection references = new LinkedList();
58  
59        public WorkerThreadDisposer() {
60           super("WorkerThreadDisposer-Daemon");
61           setDaemon(true);
62           setPriority(Thread.MIN_PRIORITY);
63           start();
64        }
65  
66        public static WorkerThreadDisposer getInstance() {
67           synchronized (WorkerThreadDisposer.class) {
68              if (instance == null) {
69                 instance = new WorkerThreadDisposer();
70              }
71           }
72  
73           return instance;
74        }
75  
76        public void enqueue(Thread parentThread, WorkerThread workerThread) {
77           LiveThreadWorkerThreadPair pair = new LiveThreadWorkerThreadPair(
78                 parentThread, queue, workerThread);
79  
80           synchronized (references) {
81              references.add(pair);
82           }
83        }
84  
85        public void run() {
86           while (true) {
87              try {
88                 LiveThreadWorkerThreadPair pair = 
89                       (LiveThreadWorkerThreadPair)queue.remove();
90                 WorkerThread worker = pair.getWorkerThread();
91  
92                 if (worker != null) {
93                    worker.interrupt();
94                 }
95                 
96                 pair.clear();
97  
98                 synchronized (references) {
99                    references.remove(pair);
100                }
101             } catch (InterruptedException ie) {
102                log.error("WorkerThreadDisposed interrupted", ie);
103                return;
104             }
105          }
106       }
107    }

109    private static final class WorkerThread extends Thread {
110       private final boolean keepThreadInstance;
111       private Throwable throwable;
112       private Object returnValue;
113       private JoinPoint jp;
114       private Object lock = new Object();
115       private boolean started;
116       private boolean running;
117       private boolean waiting;
118 
119       public WorkerThread(boolean keepThreadInstance) {
120          super("WorkerThread-" + Thread.currentThread().getName());
121          setDaemon(true);
122          this.keepThreadInstance = keepThreadInstance;
123       }
124 
125       public void execute(JoinPoint jp, long timeout) {
126          this.jp = jp;
127          returnValue = null;
128          throwable = null;
129          running = true;
130          waiting = false;
131 
132          if (!started) {
133             started = true;
134             start();
135          } else {
136             synchronized (lock) {
137                lock.notifyAll();
138             }
139          }
140 
141          synchronized (lock) {
142             try {
143                waiting = true;
144                lock.wait(timeout);
145             } catch (InterruptedException ie) {
146                log.info(ie);
147             }
148          }
149       }
150 
151       public void run() {
152          do {
153             try {
154                returnValue = jp.proceed();
155             } catch (final Throwable t) {
156                throwable = t;
157             }
158 
159             running = false;
160             jp = null;
161 
162             while (!waiting) {
163                try {
164                   Thread.sleep(10);
165                } catch (InterruptedException ie) {
166                   if (log.isTraceEnabled()) {
167                      log.trace(ie);
168                   }
169 
170                   return;
171                }
172             }
173 
174             synchronized (lock) {
175                lock.notifyAll();
176 
177                if (!keepThreadInstance) {
178                   return;
179                }
180 
181                try {
182                   lock.wait();
183                } catch (InterruptedException ie) {
184                   if (log.isTraceEnabled()) {
185                      log.trace(ie);
186                   }
187 
188                   return;
189                }
190             }
191          } while (true);
192       }
193 
194       public Object getReturnValue() {
195          return returnValue;
196       }
197 
198       public Throwable getThrowable() {
199          return throwable;
200       }
201 
202       public boolean isRunning() {
203          return running;
204       }
205 
206       public void cleanUp() {
207          returnValue = null;
208          throwable = null;
209       }
210    }

212    private final long timeout;
213    private final boolean keepThreadInstance;
214    private WorkerThread thread;
215    
216    public TimeoutAspect(final CrossCuttingInfo ccInfo) {
217       this.timeout = Long.parseLong(ccInfo.getParameter("timeout"));
218       this.keepThreadInstance = "true".equals(
219             ccInfo.getParameter("keepThreadInstance"));
220    }
221    
222    /***
223     * @Around timeout
224     */
225    public Object timeoutAdvice(final JoinPoint jp) throws Throwable {
226       if (thread == null) {
227          thread = new WorkerThread(keepThreadInstance);
228 
229          if (keepThreadInstance) {
230             if (log.isDebugEnabled()) {
231                log.debug("New thread created: " + thread);
232             }
233 
234             WorkerThreadDisposer.getInstance().enqueue(Thread.currentThread(), 
235                   thread);
236          }
237       } else {
238          if (keepThreadInstance && log.isDebugEnabled()) {
239             log.debug("Reusing thread: " + thread);
240          }
241       }
242 
243       thread.execute(jp, timeout);
244 
245       if (thread.isRunning()) {
246          try {
247             thread.interrupt();
248          } catch (Throwable t) {
249             log.error(t);
250          }
251 
252          thread = null;
253 
254          throw new TimeoutException("Execution took more than " + timeout + 
255                " ms");
256       }
257  
258       final Throwable throwable = thread.getThrowable();
259       final Object returnValue = thread.getReturnValue();
260 
261       thread.cleanUp();
262 
263       if (!keepThreadInstance) {
264          thread = null;
265       }
266 
267       if (throwable != null) {
268          throw throwable;
269       }
270 
271       return returnValue;
272    }
