24  import java.util.Map;

26  import net.java.dev.genesis.annotation.DataProvider;
27  import net.java.dev.genesis.reflection.FieldEntry;
28  import net.java.dev.genesis.ui.metadata.DataProviderMetadata;

35  import org.apache.commons.beanutils.PropertyUtils;
36  import org.codehaus.backport175.reader.Annotation;

38  public class DataProviderAnnotationHandler implements AnnotationHandler {
39     public void processFormAnnotation(final FormMetadata formMetadata,
40           final Annotation annotation) {
41        AnnotationHandlerExceptionFactory.notFormAnnotation(formMetadata, 
42              "DataProvider"); // NOI18N
43     }
44  
45     public void processFieldAnnotation(final FormMetadata formMetadata,
46           final FieldMetadata fieldMetadata, final Annotation annotation) {
47        AnnotationHandlerExceptionFactory.notFieldAnnotation(formMetadata, 
48              fieldMetadata, "DataProvider", true); // NOI18N
49     }
50  
51     public void processMethodAnnotation(final FormMetadata formMetadata,
52           final MethodMetadata methodMetadata, final Annotation annotation) {
53  
54        final DataProvider annon = (DataProvider) annotation;
55        final String widgetName = annon.widgetName();
56        final String objectFieldName = annon.objectField();
57        final String indexFieldName = annon.indexField();
58        final boolean callOnInit = annon.callOnInit();
59        final boolean resetSelection = annon.resetSelection();
60  
61        final DataProviderMetadata dataProviderMetadata = methodMetadata
62              .getDataProviderMetadata();
63  
64        final String methodName =
65              AnnotationHandlerExceptionFactory.getMethodName(formMetadata,
66              methodMetadata);
67  
68        if (GenesisUtils.isBlank(widgetName)
69              && GenesisUtils.isBlank(objectFieldName)
70              && GenesisUtils.isBlank(indexFieldName)) {
71           String errorMessage = Bundle.getMessage(
72                 DataProviderAnnotationHandler.class,
73                 "AT_LEAST_ONE_OF_OPTIONS_MUST_BE_SPECIFIED_FOR_DATAPROVIDER_X", // NOI18N
74                 methodName);
75           throw new IllegalArgumentException(errorMessage);
76        }
77  
78        dataProviderMetadata.setCallOnInit(callOnInit);
79        dataProviderMetadata.setResetSelection(resetSelection);
80  
81        PropertyDescriptor[] descriptors = PropertyUtils
82              .getPropertyDescriptors(formMetadata.getFormClass());
83  
84        Map descriptorsPerPropertyName = new HashMap();
85  
86        for (int i = 0; i < descriptors.length; i++) {
87           descriptorsPerPropertyName.put(descriptors[i].getName(),
88                 descriptors[i]);
89        }
90  
91        if (!GenesisUtils.isBlank(objectFieldName)) {
92           PropertyDescriptor descriptor = (PropertyDescriptor) descriptorsPerPropertyName
93                 .get(objectFieldName);
94           if (descriptor == null || descriptor.getReadMethod() == null ||
95                 descriptor.getWriteMethod() == null) {
96  
97              StringBuffer propertyName = new StringBuffer().append(Character
98                    .toUpperCase(objectFieldName.charAt(0)));
99  
100             if (objectFieldName.length() > 1) {
101                propertyName.append(objectFieldName.substring(1));
102             }
103 
104             String key =
105                   descriptor == null ? "DATAPROVIDER_OBJECTFIELD_REFERS_TO_X_BUT_PUBLIC_PROPERTY_DOES_NOT_EXIST" : // NOI18N
106                   "DATAPROVIDER_OBJECTFIELD_REFERS_TO_X_BUT_IT_IS_NOT_A_READ_WRITE_PROPERTY"; // NOI18N
107             String errorMessage = Bundle.getMessage(
108                   DataProviderAnnotationHandler.class, key, new Object[] {
109                      objectFieldName, propertyName, methodName});
110 
111             throw new IllegalArgumentException(errorMessage);
112          }
113 
114          final Class fieldType = descriptor.getPropertyType();
115 
116          if (fieldType.isPrimitive()
117                || (fieldType.isArray() && fieldType.getComponentType()
118                      .isPrimitive())) {
119             String errorMessage = Bundle.getMessage(
120                   DataProviderAnnotationHandler.class,
121                   "DATAPROVIDER_OBJECTFIELD_REFERS_TO_X_BUT_IT_IS_NOT_A_PRIMITIVE_OR_ARRAY_OF_PRIMITIVES", // NOI18N
122                   objectFieldName, methodName);
123 
124             throw new IllegalArgumentException(errorMessage);
125          }
126 
127          dataProviderMetadata.setObjectField(new FieldEntry(descriptor
128                .getName(), descriptor.getPropertyType()));
129       }
130 
131       if (!GenesisUtils.isBlank(indexFieldName)) {
132          PropertyDescriptor descriptor = (PropertyDescriptor) descriptorsPerPropertyName
133                .get(indexFieldName);
134 
135          if (descriptor == null || descriptor.getReadMethod() == null ||
136                descriptor.getWriteMethod() == null) {
137             final StringBuffer propertyName = new StringBuffer().append(Character
138                   .toUpperCase(indexFieldName.charAt(0)));
139 
140             if (indexFieldName.length() > 1) {
141                propertyName.append(indexFieldName.substring(1));
142             }
143 
144             String key =
145                   descriptor == null ? "DATAPROVIDER_INDEXFIELD_REFERS_TO_X_BUT_SUCH_PUBLIC_PROPERTY_DOES_NOT_EXIST" : // NOI18N
146                   "DATAPROVIDER_INDEXFIELD_REFERS_TO_X_BUT_IT_IS_NOT_A_READ_WRITE_PROPERTY"; // NOI18N
147             String errorMessage = Bundle.getMessage(
148                   DataProviderAnnotationHandler.class, key, new Object[] {
149                      indexFieldName, propertyName, methodName});
150             throw new IllegalArgumentException(errorMessage);
151          }
152 
153          final Class fieldType = descriptor.getPropertyType().isArray() ? descriptor
154                .getPropertyType().getComponentType()
155                : descriptor.getPropertyType();
156 
157          if (!Collection.class.isAssignableFrom(descriptor.getPropertyType())
158                && !Integer.TYPE.isAssignableFrom(fieldType)
159                && !Integer.class.isAssignableFrom(fieldType)) {
160             String errorMessage = Bundle.getMessage(
161                   DataProviderAnnotationHandler.class,
162                   "DATAPROVIDER_INDEXFIELD_REFERS_TO_X_BUT_IT_IS_NOT_AN_INTEGER_INT_ARRAY_COLLECTION", // NOI18N
163                   indexFieldName, methodName);
164             throw new IllegalArgumentException(errorMessage);
165          }
166 
167          dataProviderMetadata.setIndexField(new FieldEntry(
168                descriptor.getName(), descriptor.getPropertyType()));
169       }
170 
171       dataProviderMetadata
172             .setWidgetName(!GenesisUtils.isBlank(widgetName) ? widgetName
173                   : !GenesisUtils.isBlank(objectFieldName) ? objectFieldName
174                         : indexFieldName);
175    }
