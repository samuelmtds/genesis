36  public class JComboBoxComponentBinder extends AbstractComponentBinder {
37     public BoundDataProvider bind(SwingBinder binder, Component component,
38        DataProviderMetadata dataProviderMetadata) {
39        return new JComboBoxBoundMember(binder, (JComboBox) component,
40           dataProviderMetadata);
41     }

43     public class JComboBoxBoundMember extends AbstractBoundMember
44           implements BoundField, BoundDataProvider {
45        private final JComboBox component;
46        private final DataProviderMetadata dataProviderMetadata;
47        private final ActionListener listener;
48  
49        public JComboBoxBoundMember(SwingBinder binder, JComboBox component,
50              DataProviderMetadata dataProviderMetadata) {
51           super(binder, component);
52           this.component = component;
53           this.dataProviderMetadata = dataProviderMetadata;
54  
55           this.component.addActionListener(listener = createActionListener());
56        }
57  
58        protected JComboBox getComponent() {
59           return component;
60        }
61  
62        protected DataProviderMetadata getDataProviderMetadata() {
63           return dataProviderMetadata;
64        }
65  
66        protected ActionListener getListener() {
67           return listener;
68        }
69  
70        protected ActionListener createActionListener() {
71           return new ActionListener() {
72              public void actionPerformed(ActionEvent event) {
73                 getBinder().updateFormSelection(getDataProviderMetadata(), getIndexes());
74              }
75           };
76        }
77  
78        protected int[] getIndexes() {
79           int selectedIndex = component.getSelectedIndex();
80           return getBinder().getIndexesFromUI(
81                 (selectedIndex < 0) ? new int[0] : new int[] { selectedIndex },
82                 isBlank(component));
83        }
84  
85        public void updateIndexes(int[] indexes) {
86           indexes = getBinder()
87                           .getIndexesFromController(indexes, isBlank(component));
88  
89           if (indexes.length > 1) {
90              throw new IllegalArgumentException(Bundle.getMessage(
91                    JComboBoxComponentBinder.class,
92                    "LENGTH_OF_SELECTED_INDEXES_MUST_BE_ONE_OR_LESS")); // NOI18N
93           } else if (indexes.length == 0) {
94              indexes = new int[] { -1 };
95           }
96  
97           if (component.getSelectedIndex() == indexes[0]) {
98              return;
99           }
100 
101          deactivateListeners();
102          try {
103             component.setSelectedIndex(indexes[0]);
104          } finally {
105             reactivateListeners();
106          }
107       }
108 
109       public void updateList(List list) throws Exception {
110          deactivateListeners();
111          try {
112             boolean isBlank = isBlank(component);
113 
114             int selected = component.getSelectedIndex();
115             if (dataProviderMetadata.isResetSelection()) {
116                selected = isBlank ? 0 : -1;
117             } else {
118                selected = selected <= list.size() ? selected : isBlank ? 0 : -1;
119             }
120 
121             createComboBoxModelAdapter().setData(getData(list));
122 
123             component.setSelectedIndex(selected);
124          } finally {
125             reactivateListeners();
126          }
127       }
128 
129       public String getValue() throws Exception {
130          if (dataProviderMetadata.getObjectField() == null) {
131             return null;
132          }
133 
134          if (isBlank(component) && component.getSelectedIndex() == 0) {
135             return null;
136          }
137 
138          return getKey(component.getSelectedItem());
139       }
140 
141       public void setValue(Object value) throws Exception {
142          if (dataProviderMetadata.getObjectField() == null) {
143             return;
144          }
145 
146          boolean isBlank = isBlank(component);
147 
148          int found = -1;
149 
150          if (value != null) {
151             String selectedKey = getKey(value);
152 
153             int size = component.getItemCount();
154 
155             for (int i = isBlank ? 1 : 0; i < size; i++) {
156                Object o = component.getItemAt(i);
157 
158                if (selectedKey.equals(getKey(o))) {
159                   found = i;
160 
161                   break;
162                }
163             }
164          } else if (isBlank) {
165             found++;
166          }
167 
168          deactivateListeners();
169          try {
170             component.setSelectedIndex(found);
171          } finally {
172             reactivateListeners();
173          }
174       }
175 
176       protected Object[] getData(List list) throws Exception {
177          boolean isBlank = isBlank(component);
178          Object[] values = new Object[isBlank ? (list.size() + 1) : list.size()];
179 
180          int i = 0;
181 
182          if (isBlank) {
183             String blankLabel = (String) component
184                   .getClientProperty(SwingBinder.BLANK_LABEL_PROPERTY);
185             values[i] = (blankLabel == null) ? "" : blankLabel; // NOI18N
186             i++;
187          }
188 
189          for (Iterator iter = list.iterator(); iter.hasNext(); i++) {
190             values[i] = iter.next();
191          }
192 
193          return values;
194       }
195 
196       protected ComboBoxModelAdapter createComboBoxModelAdapter() {
197          return new ComboBoxModelAdapter() {
198             public void setData(Object[] data) {
199                DefaultComboBoxModel model = (DefaultComboBoxModel) component
200                      .getModel();
201 
202                model.removeAllElements();
203 
204                for (int i = 0; i < data.length; i++) {
205                   model.addElement(data[i]);
206                }
207             }
208          };
209       }
210 
211       protected String getKey(Object value) throws Exception {
212          return ComponentBinderHelper.getKey(getBinder(), component, getName(), 
213                value);
214       }
215 
216       protected void deactivateListeners() {
217          if (listener != null) {
218             component.removeActionListener(listener);
219          }
220       }
221 
222       protected void reactivateListeners() {
223          if (listener != null) {
224             component.addActionListener(listener);
225          }
226       }
227 
228       public void unbind() {
229          if (listener != null) {
230             component.removeActionListener(listener);
231          }
232       }
233    }
