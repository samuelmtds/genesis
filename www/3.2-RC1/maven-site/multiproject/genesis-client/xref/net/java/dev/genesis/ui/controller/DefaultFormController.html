119    protected FormState createFormState(FormState state) {
120       Map clonedValues = new HashMap(state.getValuesMap().size());
121 
122       for (Iterator iter = state.getValuesMap().entrySet().iterator(); iter
123             .hasNext();) {
124          Map.Entry entry = (Map.Entry)iter.next();
125 
126          FieldMetadata fieldMeta = formMetadata.getFieldMetadata(entry.getKey()
127                .toString());
128 
129          clonedValues.put(entry.getKey(), fieldMeta.getCloner().clone(
130                entry.getValue()));
131       }
132 
133       return new FormStateImpl(state, clonedValues);
134    }

136    public void setup() throws Exception {
137       if (setup) {
138          throw new IllegalStateException(Bundle.getMessage(DefaultFormController.class,
139                "SETUP_METHOD_ALREADY_CALLED")); // NOI18N
140       }
141 
142       setup = true;
143 
144       ScriptContext ctx = createScriptContext();
145       ctx.declare(FORM_METADATA_KEY, formMetadata);
146 
147       currentState = createFormState();
148       ctx.declare(CURRENT_STATE_KEY, currentState);
149 
150       setContext(ctx);
151 
152       if (PropertyUtils.isWriteable(getForm(), "context") // NOI18N
153             && Map.class.isAssignableFrom(PropertyUtils.getPropertyType(getForm(),
154                   "context"))) { // NOI18N
155          PropertyUtils.setProperty(getForm(), "context", ctx.getContextMap()); // NOI18N
156       }
157 
158       evaluate(true);
159    }

189    protected void populate(Map properties, boolean stringMap, Map converters) 
190          throws Exception {
191       createPreviousState();
192 
193       try {
194          if (properties == null) {
195             properties = PropertyUtils.describe(getForm());
196          }
197 
198          updateChangedMap(properties, stringMap, converters);
199 
200          if (currentState.getChangedMap().isEmpty()) {
201             log.debug("Nothing changed."); // NOI18N
202 
203             return;
204          }
205 
206          if (stringMap) {
207             PropertyUtils.copyProperties(getForm(), currentState.getChangedMap());
208          }
209 
210          evaluate(false);
211       } catch (Exception e) {
212          resetPreviousState();
213 
214          throw e;
215       } finally {
216          releasePreviousState();
217       }
218    }

220    protected boolean updateChangedMap(Map newData, boolean stringMap, 
221             Map converters) {
222       boolean changed = false;
223 
224       Object value;
225       FieldMetadata fieldMeta;
226       Map.Entry entry;
227 
228       final Map changedMap = currentState.getChangedMap();
229 
230       changedMap.clear();
231       changedMap.putAll(GenesisUtils.normalizeMap(newData));
232 
233       if (log.isDebugEnabled()) {
234          log.debug("changedMap started as: " + changedMap); // NOI18N
235       }
236 
237       for (final Iterator i = changedMap.entrySet().iterator(); i.hasNext();) {
238          entry = (Map.Entry) i.next();
239          fieldMeta = formMetadata.getFieldMetadata(entry.getKey().toString());
240 
241          if (fieldMeta == null) {
242             continue;
243          }
244 
245          value = (stringMap) ? getConverter(fieldMeta, converters).convert(
246                fieldMeta.getFieldClass(), entry.getValue()) : entry.getValue();
247 
248          if (getPreviousState() != null && fieldMeta.getEqualityComparator().equals(
249                value, getPreviousState().getValuesMap().get(entry.getKey()))
250                && fieldMeta.getEqualityComparator().equals(value,
251                      currentState.getValuesMap().get(entry.getKey()))) {
252             i.remove();
253 
254             if (log.isDebugEnabled()) {
255                log.debug(entry.getKey() + " removed from changedMap for being " // NOI18N
256                      + " equal to previousValue; is " + entry.getValue() // NOI18N
257                      + "; was:" + getPreviousState().getValuesMap().get(entry // NOI18N
258                      .getKey()));
259             }
260 
261             continue;
262          }
263 
264          if (log.isDebugEnabled()) {
265             log.debug("Field '" + entry.getKey() + "' changed to '" // NOI18N
266                   + entry.getValue() + "'; was " + (getPreviousState() == null ? // NOI18N
267                   "undefined" : getPreviousState().getValuesMap().get(entry // NOI18N
268                   .getKey())));
269          }
270 
271          entry.setValue(value);
272          currentState.getValuesMap().put(entry.getKey(), entry.getValue());
273          
274          changed = true;
275       }
276 
277       return changed;
278    }

280    private Converter getConverter(FieldMetadata fieldMetadata, Map converters) {
281       if (converters == null) {
282          return fieldMetadata.getConverter();
283       }
284 
285       final Converter converter = (Converter)converters.get(
286             fieldMetadata.getName());
287 
288       return (converter == null) ? fieldMetadata.getConverter() : converter;
289    }

291    protected void evaluate(boolean firstCall) throws Exception {
292       boolean changed = true;
293 
294       for (int times = 0; changed && times < getMaximumEvaluationTimes(); 
295             times++) {
296          changed = doEvaluate(firstCall);
297          firstCall = false;
298       }
299    }
300 
301    protected boolean doEvaluate(boolean firstCall) throws Exception {
302       boolean changed = false;
303       evaluateNamedConditions();
304 
305       if (evaluateClearOnConditions()) {
306          changed = true;
307          evaluateNamedConditions();
308       }
309 
310       if (evaluateCallWhenConditions(firstCall)) {
311          changed = true;
312          evaluateNamedConditions();
313       }
314 
315       if (evaluateDataProvidedIndexes(firstCall)) {
316          changed = true;
317          evaluateNamedConditions();
318       }
319 
320       final Map newData = PropertyUtils.describe(getForm());
321       changed |= updateChangedMap(newData, false, null);
322 
323       fireValuesChanged(currentState.getChangedMap());
324 
325       changed |= evaluateEnabledWhenConditions();
326       changed |= evaluateVisibleWhenConditions();
327       
328       return changed;
329    }

339    protected boolean evaluateClearOnConditions() throws Exception {
340       final Map fieldMetadatas = formMetadata.getFieldMetadatas();
341       final Map dataProviderMetadatas = formMetadata.getDataProviderMetadatas();
342       final Map toCopy = new HashMap();
343 
344       boolean changed;
345       FieldMetadata fieldMetadata;
346       DataProviderMetadata dataProviderMeta;
347       List items;
348       Map.Entry entry;
349       final Map properties = PropertyUtils.describe(getForm());
350 
351       do {
352          changed = false;
353          properties.putAll(toCopy);
354 
355          for (final Iterator i = fieldMetadatas.entrySet().iterator(); i
356                .hasNext();) {
357             entry = (Map.Entry) i.next();
358             fieldMetadata = (FieldMetadata) entry.getValue();
359 
360             if (fieldMetadata.getClearOnCondition() == null
361                   || fieldMetadata.getEmptyResolver().isEmpty(
362                         properties.get(entry.getKey()))) {
363                i.remove();
364                continue;
365             }
366 
367             if (isConditionSatisfied(fieldMetadata.getClearOnCondition())) {
368                if (log.isDebugEnabled()) {
369                   log.debug("ClearOn Condition for field '" + entry.getKey() // NOI18N
370                         + "' satisfied. Setting '" // NOI18N
371                         + fieldMetadata.getEmptyValue() + "'"); // NOI18N
372                }
373                changed = true;
374                toCopy.put(entry.getKey(), fieldMetadata.getEmptyValue());
375                i.remove();
376             }
377          }
378          
379          for (Iterator i = dataProviderMetadatas.entrySet().iterator(); 
380                i.hasNext();) {
381             entry = (Map.Entry) i.next();
382             dataProviderMeta = (DataProviderMetadata) entry.getValue();
383             
384             if (dataProviderMeta.getClearOnCondition() == null) {
385                i.remove();
386                continue;
387             }
388 
389             if (isConditionSatisfied(dataProviderMeta.getClearOnCondition())) {
390                if (log.isDebugEnabled()) {
391                   log.debug("ClearOn Condition for data provider '" +  // NOI18N
392                         entry.getKey() + "' satisfied. Clearing data " + // NOI18N
393                         "provider list."); // NOI18N
394                }
395 
396                changed = true;
397                updateDataProviderCurrentSelection(dataProviderMeta,
398                      items = EMPTY_LIST, EMPTY_INT_ARRAY);
399                fireDataProvidedListMetadataChanged(dataProviderMeta, items, false);
400                i.remove();
401             }
402          }
403       } while (changed);
404 
405       if (toCopy.isEmpty()) {
406          if (log.isDebugEnabled()) {
407             log.debug("Nothing changed in ClearOn conditions."); // NOI18N
408          }
409 
410          return false;
411       }
412 
413       PropertyUtils.copyProperties(getForm(), toCopy);
414       currentState.getChangedMap().putAll(toCopy);
415 
416       return true;
417    }

419    protected void evaluateNamedConditions() {
420       Map.Entry entry;
421 
422       for (final Iterator i = formMetadata.getNamedConditions().entrySet()
423             .iterator(); i.hasNext();) {
424          entry = (Map.Entry) i.next();
425          
426          evaluateNamedCondition(entry.getKey().toString(), 
427                  (ScriptExpression)entry.getValue());
428       }
429    }

435 
436       if (log.isDebugEnabled()) {
437          log.debug("Named Condition '" + conditionName + "' evaluated as '" + // NOI18N
438                  conditionValue + "'"); // NOI18N
439       }

450    protected boolean evaluateConditions(boolean enabled) {
451       Map.Entry entry;
452 
453       final Map updatedConditions = new HashMap();
454       final Map toEvaluate = new HashMap(formMetadata.getFieldMetadatas());
455       toEvaluate.putAll(formMetadata.getActionMetadatas());
456 
457       for (final Iterator i = toEvaluate.entrySet().iterator(); i.hasNext();) {
458          entry = (Map.Entry) i.next();
459          evaluateCondition((MemberMetadata) entry.getValue(), updatedConditions, enabled);
460       }
461 
462       if (enabled) {
463          fireEnabledConditionChanged(updatedConditions);
464       } else {
465          fireVisibleConditionChanged(updatedConditions);
466       }
467 
468       return !updatedConditions.isEmpty();
469    }

471    protected void evaluateCondition(MemberMetadata memberMetadata, Map updatedConditions, boolean enabled) {
472       if ((enabled && memberMetadata.getEnabledCondition() == null)
473             || (!enabled && memberMetadata.getVisibleCondition() == null)) {
474          return;
475       }
476 
477       final Object newValue = isSatisfied(enabled ? memberMetadata.getEnabledCondition()
478             : memberMetadata.getVisibleCondition());
479       
480       final Map conditionMap = enabled ?
481             currentState.getEnabledMap() : currentState.getVisibleMap();
482 
483       if (!newValue.equals(conditionMap.get(memberMetadata
484             .getName()))) {
485          conditionMap.put(memberMetadata.getName(), newValue);
486          updatedConditions.put(memberMetadata.getName(), newValue);
487       }
488 
489       if (log.isDebugEnabled()) {
490          log.debug((enabled ? "EnabledWhen" : "VisibleWhen") // NOI18N
491                + " Condition for member '" + memberMetadata.getName() // NOI18N
492                + "' evaluated as '" // NOI18N
493                + conditionMap.get(memberMetadata.getName()) + "'"); // NOI18N
494       }
495    }

511    protected boolean evaluateCallWhenConditions(boolean firstCall) throws Exception {
512       boolean changed = false;
513

522    protected boolean evaluateDataProvidedIndexes(boolean firstCall) throws Exception {
523       boolean changed = false;
524 
525       Map currentMap = firstCall ? PropertyUtils.describe(getForm()) : currentState.getChangedMap();
526 
527       for (final Iterator i = formMetadata.getDataProviderIndexes().values()
528             .iterator(); i.hasNext(); ) {
529          changed |= evaluateDataProvidedIndex((DataProviderMetadata) i.next(), currentMap);
530       }

535    protected boolean evaluateDataProvidedIndex(DataProviderMetadata dataMeta,
536          Map currentMap) throws Exception {
537       final Object indexes = currentMap.get(dataMeta.getIndexField()
538             .getFieldName());
539 
540       if (indexes == null) {
541          if (log.isDebugEnabled()) {
542             log.debug("Index field " + dataMeta.getIndexField().getFieldName() // NOI18N
543                   + " haven't been changed."); // NOI18N
544          }
545 
546          return false;
547       }
548 
549       final int[] selectedIndexes = dataMeta.getSelectedIndexes(indexes);
550 
551       final Object objectFieldValue = dataMeta.populateSelectedFields(getForm(),
552             (List)currentState.getDataProvidedMap().get(dataMeta),
553             selectedIndexes);
554       fireDataProvidedIndexesChanged(dataMeta, selectedIndexes);
555 
556       currentState.getDataProvidedIndexesMap().put(dataMeta, selectedIndexes);
557 
558       if (dataMeta.getObjectField() != null) {
559          currentState.getChangedMap().put(
560                dataMeta.getObjectField().getFieldName(), objectFieldValue);
561       }
562 
563       return true;
564    }

566    protected boolean invokeAction(MethodMetadata methodMetadata, 
567          boolean firstCall, boolean conditionally) throws Exception {
568       List items = null;
569       DataProviderMetadata dataProviderMeta = methodMetadata
570             .getDataProviderMetadata();
571 
572       if ((methodMetadata.getCallCondition() == null || firstCall) 
573             && conditionally) {
574          if (dataProviderMeta == null || (conditionally && !firstCall)) {
575             return false;
576          }
577 
578          if (!dataProviderMeta.isCallOnInit()) {
579             updateDataProviderCurrentSelection(dataProviderMeta,
580                   items = EMPTY_LIST, EMPTY_INT_ARRAY);
581             fireDataProvidedListMetadataChanged(dataProviderMeta, items, false);
582 
583             return true;
584          }
585          conditionally = false;
586       }
587 
588       final boolean satisfied = !conditionally || isConditionSatisfied(
589             methodMetadata.getCallCondition());
590 
591       if (satisfied) {
592          if (methodMetadata.getActionMetadata() != null && 
593                !beforeInvokingMethod(methodMetadata)) {
594             return false;
595          }
596 
597          final Object ret = methodMetadata.invoke(getForm());
598 
599          if (dataProviderMeta != null) {
600             if (ret == null) {
601                throw new IllegalStateException(Bundle.getMessage(DefaultFormController.class,
602                      "DATAPROVIDER_X_IN_Y_RETURNED_NULL_AND_SHOULD_RETURN_Z", // NOI18N
603                      new Object[]{methodMetadata.getName(), getForm(),
604                      ReflectionInvoker.getInstance().getMethod(getForm(), 
605                      methodMetadata.getName(), methodMetadata.getMethodEntry()
606                      .getArgsClassesNames()).getReturnType().getName()}));
607             }
608 
609             items = (ret.getClass().isArray()) ? Arrays.asList((Object[]) ret) :
610                  (List)ret;
611             currentState.getDataProvidedMap().put(dataProviderMeta, items);
612             fireDataProvidedListMetadataChanged(dataProviderMeta, items, firstCall);
613          } 
614 
615          if (methodMetadata.getActionMetadata() != null) {
616             afterInvokingMethod(methodMetadata);
617          }
618       }
619 
620       if (log.isDebugEnabled()) {
621          log.debug("CallWhen Condition for method '" + // NOI18N
622                methodMetadata.getName() + "' evaluated as '" + // NOI18N
623                satisfied + "'"); // NOI18N
624       }
625 
626       return satisfied;
627    }

629    public void resetSelection(DataProviderMetadata meta, List dataProvided)
630          throws Exception {
631       int[] selected;
632 
633       if (meta.isResetSelection()) {
634          meta.resetSelectedFields(getForm());
635          selected = EMPTY_INT_ARRAY;
636       } else {
637          int[] currentSelection = (int[])currentState
638                .getDataProvidedIndexesMap().get(meta);
639          selected = meta.retainSelectedFields(getForm(), dataProvided,
640                currentSelection == null ? EMPTY_INT_ARRAY : currentSelection);
641       }
642 
643       currentState.getDataProvidedIndexesMap().put(meta, selected);
644    }

649    }

698    public void reset(FormState state) throws Exception {
699       if (log.isDebugEnabled()) {
700          log.debug("Reseting form '" + getForm() + "'"); // NOI18N
701       }
702 
703       state = createFormState(state);
704       currentState = createFormState(state);
705       getScriptContext().declare(CURRENT_STATE_KEY, currentState);
706 
707       fireEnabledConditionChanged(state.getEnabledMap());
708       fireVisibleConditionChanged(state.getVisibleMap());
709 
710       for (final Iterator i = state.getDataProvidedMap().entrySet().iterator(); 
711             i.hasNext();) {
712          final Map.Entry entry = (Map.Entry)i.next();
713          fireDataProvidedListMetadataChanged((DataProviderMetadata)entry
714                .getKey(), (List)entry.getValue(), false);
715       }
716 
717       final Map currentValues = PropertyUtils.describe(getForm());
718 
719       for (final Iterator i = formMetadata.getDataProviderIndexes().values()
720             .iterator(); i.hasNext(); ) {
721          DataProviderMetadata dataMeta = (DataProviderMetadata) i.next();
722          String fieldName = dataMeta.getIndexField().getFieldName();
723          Object indexes = currentValues.get(fieldName);
724 
725          if (getFormMetadata().getFieldMetadata(fieldName)
726                .getEqualityComparator().equals(indexes, state.getValuesMap()
727                .get(fieldName))) {
728             continue;
729          }
730 
731          fireDataProvidedIndexesChanged(dataMeta, 
732                dataMeta.getSelectedIndexes(indexes));
733       }
734 
735 
736       for (final Iterator i = currentValues.entrySet().iterator(); 
737             i.hasNext();) {
738          Map.Entry entry = (Map.Entry)i.next();
739          FieldMetadata fieldMeta = formMetadata.getFieldMetadata(entry.getKey()
740                .toString());
741 
742          if (fieldMeta == null || fieldMeta.getEqualityComparator().equals(
743                entry.getValue(), state.getValuesMap().get(entry.getKey()))) {
744             i.remove();
745 
746             continue;
747          }
748 
749          currentValues.put(entry.getKey(), state.getValuesMap().get(
750                entry.getKey()));
751       }
752 
753       PropertyUtils.copyProperties(getForm(), currentValues);
754       currentState.getChangedMap().clear();
755       currentState.getChangedMap().putAll(currentValues);
756       currentState.getValuesMap().putAll(currentValues);
757 
758       fireValuesChanged(currentState.getValuesMap());
759    }
