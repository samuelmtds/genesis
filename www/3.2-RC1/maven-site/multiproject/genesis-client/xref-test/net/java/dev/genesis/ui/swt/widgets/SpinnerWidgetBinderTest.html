31  import org.eclipse.swt.SWT;
32  import org.eclipse.swt.widgets.Event;
33  import org.eclipse.swt.widgets.Shell;
34  import org.eclipse.swt.widgets.Spinner;

36  public class SpinnerWidgetBinderTest extends GenesisTestCase {
37     private Shell root;
38     private Spinner spinner;
39     private MockSWTBinder binder;
40     private WidgetBinder widgetBinder;
41     private BoundField boundField;
42     private MockForm form;
43     private FieldMetadata fieldMeta;
44  
45     public SpinnerWidgetBinderTest() {
46        super("Spinner Widget Binder Unit Test");
47     }
48  
49     protected void setUp() throws Exception {
50        spinner = new Spinner(root = new Shell(), SWT.NONE);
51        spinner.setDigits(0);
52        spinner.setMaximum(Integer.MAX_VALUE);
53        spinner.setMinimum(Integer.MIN_VALUE);
54        binder = new MockSWTBinder(root, form = new MockForm(), new Object());
55        widgetBinder = binder.getWidgetBinder(spinner);
56        fieldMeta = form.getFormMetadata().getFieldMetadata("intField");
57     }
58  
59     public void testSetValue() throws Exception {
60        assertTrue(widgetBinder instanceof SpinnerWidgetBinder);
61  
62        assertNull(widgetBinder.bind(binder, spinner, (ActionMetadata) null));
63        assertNull(widgetBinder.bind(binder, spinner,
64              (DataProviderMetadata) null));
65  
66        boundField = widgetBinder.bind(binder, spinner, fieldMeta);
67        assertNotNull(boundField);
68  
69        boundField.setValue(new Integer(3));
70        assertEquals(3, spinner.getSelection());
71  
72        boundField.setValue(new Integer(100));
73        assertEquals(100, spinner.getSelection());
74     }
75  
76     public void testUpdateValue() throws Exception {
77        assertNull(widgetBinder.bind(binder, spinner, (ActionMetadata) null));
78        assertNull(widgetBinder.bind(binder, spinner,
79              (DataProviderMetadata) null));
80        assertNotNull(widgetBinder.bind(binder, spinner, fieldMeta));
81  
82        simulateSelection(3);
83        Object value = binder
84              .get("populateForm(FieldMetadata,Object)");
85        assertNotNull(value);
86        assertEquals(new Integer(3), value);
87  
88        simulateSelection(100);
89        value = binder.get("populateForm(FieldMetadata,Object)");
90        assertNotNull(value);
91        assertEquals(new Integer(100), value);
92  
93        spinner.setDigits(4);
94        simulateSelection(31415);
95        value = binder.get("populateForm(FieldMetadata,Object)");
96        assertNotNull(value);
97        assertEquals(new Float(3.1415), value);
98        
99     }
100    
101    private void simulateSelection(int selection) {
102       spinner.setSelection(selection);
103       Event event = new Event();
104       event.widget = spinner;
105       event.type = SWT.Selection;
106       spinner.notifyListeners(event.type, event);
107    }
