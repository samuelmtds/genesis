19  package net.java.dev.genesis.helpers;

21  import java.lang.reflect.InvocationTargetException;
22  import java.util.Arrays;
23  import java.util.Collection;
24  
25  import net.java.dev.genesis.reflection.ReflectionInvoker;
26  import net.java.dev.genesis.util.Bundle;

29  public final class EnumHelper {
30     private static final EnumHelper instance = new EnumHelper();
31  
32     private Class enumClass;
33  
34     private EnumHelper() {
35        try {
36           enumClass = Class.forName("java.lang.Enum"); // NOI18N
37        } catch (Exception e) {
38           // not JDK 5 or later
39        }
40     }
41  
42     private Object invoke(Object target, String methodName) {
43        try {
44           return ReflectionInvoker.getInstance().invoke(target, methodName);
45        } catch (InvocationTargetException e) {
46           throw new RuntimeException(e.getTargetException());
47        } catch (Exception e) {
48           throw new RuntimeException(e);
49        }
50     }
51  
52     public static EnumHelper getInstance() {
53        return instance;
54     }
55  
56     public Class getEnumClass() {
57        return enumClass;
58     }
59  
60     public boolean supportsEnum() {
61        return enumClass != null;
62     }
63     
64     public boolean isReusableComponentEnum(Class clazz) {
65        return net.java.dev.reusablecomponents.lang.Enum.class.isAssignableFrom(clazz);
66     }
67     
68     public boolean isJava5Enum(Class clazz) {
69        return supportsEnum() && Boolean.TRUE.equals(invoke(clazz, "isEnum")); // NOI18N
70     }
71     
72     public boolean isReusableComponentEnum(Object object) {
73        return object instanceof net.java.dev.reusablecomponents.lang.Enum;
74     }
75     
76     public boolean isJava5Enum(Object object) {
77        return object != null && isJava5Enum(object.getClass());
78     }
79     
80     public boolean isEnum(Object object) {
81        return isReusableComponentEnum(object) || isJava5Enum(object);
82     }
83     
84     public Collection values(Class clazz) {
85        if (isReusableComponentEnum(clazz)) {
86           return net.java.dev.reusablecomponents.lang.Enum.getInstances(clazz);
87        } else if (isJava5Enum(clazz)) {
88           return Arrays.asList((Object[])invoke(clazz, "getEnumConstants")); // NOI18N
89        }
90  
91        throw new IllegalArgumentException(Bundle.getMessage(EnumHelper.class,
92              "X_IS_NOT_AN_ENUM", clazz.getName())); // NOI18N
93     }
94     
95     public Object valueOf(Class clazz, String name) {
96        if (isReusableComponentEnum(clazz)) {
97           return net.java.dev.reusablecomponents.lang.Enum.get(clazz, name);
98        } else if (isJava5Enum(clazz)) {
99           Object[] enums = (Object[]) invoke(clazz, "getEnumConstants"); // NOI18N
100          for (int i = 0; i < enums.length; i++) {
101             String enumName = getName(enums[i]);
102             
103             if (name.equals(enumName)) {
104                return enums[i];
105             }
106          }
107 
108          return null;
109       }
110 
111       throw new IllegalArgumentException(Bundle.getMessage(EnumHelper.class,
112             "X_IS_NOT_AN_ENUM", clazz.getName())); // NOI18N
113    }
114    
115    public Class getDeclaringClass(Object object) {
116       if (isReusableComponentEnum(object)) {
117          return ((net.java.dev.reusablecomponents.lang.Enum)object).getBaseClass();
118       } else if (isJava5Enum(object)) {
119          return (Class) invoke(object, "getDeclaringClass"); // NOI18N
120       }
121 
122       throw new IllegalArgumentException(Bundle.getMessage(EnumHelper.class,
123             "X_IS_NOT_AN_ENUM", object)); // NOI18N
124    }
125    
126    public String getName(Object object) {
127       if (isReusableComponentEnum(object)) {
128          return ((net.java.dev.reusablecomponents.lang.Enum)object).getName();
129       } else if (isJava5Enum(object)) {
130          return object.toString();
131       }
132 
133       throw new IllegalArgumentException(Bundle.getMessage(EnumHelper.class,
134             "X_IS_NOT_AN_ENUM", object)); // NOI18N
135    }
