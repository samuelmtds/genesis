64  public final class ConverterRegistry extends ConvertUtilsBean {
65     private Registry registry;
66  
67     public ConverterRegistry() {
68        getRegistry().register(BigInteger.class, new BigIntegerConverter());
69        registry.register(Boolean.TYPE, new BooleanConverter(Boolean.FALSE));
70        registry.register(Boolean.class, new BooleanConverter(null));
71        registry.register(boolean[].class, new BooleanArrayConverter(
72              new boolean[0]));
73        registry.register(Byte.TYPE, new ByteConverter(new Byte((byte)0)));
74        registry.register(Byte.class, new ByteConverter(null));
75        registry.register(byte[].class, new ByteArrayConverter(new byte[0]));
76        registry.register(Character.TYPE, new CharacterConverter(new Character(
77              ' ')));
78        registry.register(Character.class, new CharacterConverter(null));
79        registry.register(char[].class, new CharacterArrayConverter(new char[0]));
80        registry.register(Class.class, new ClassConverter());
81        registry.register(Double.TYPE, new DoubleConverter(new Double(0.0)));
82        registry.register(Double.class, new DoubleConverter(null));
83        registry
84              .register(double[].class, new DoubleArrayConverter(new double[0]));
85        registry.register(Float.TYPE, new FloatConverter(new Float(0.0f)));
86        registry.register(Float.class, new FloatConverter(null));
87        registry.register(float[].class, new FloatArrayConverter(new float[0]));
88        registry.register(Integer.TYPE, new IntegerConverter(new Integer(0)));
89        registry.register(Integer.class, new IntegerConverter(null));
90        registry.register(int[].class, new IntegerArrayConverter(new int[0]));
91        registry.register(Long.TYPE, new LongConverter(new Long(0L)));
92        registry.register(Long.class, new LongConverter(null));
93        registry.register(long[].class, new LongArrayConverter(new long[0]));
94        registry.register(Short.TYPE, new ShortConverter(new Short((short)0)));
95        registry.register(Short.class, new ShortConverter(null));
96        registry.register(short[].class, new ShortArrayConverter(new short[0]));
97        registry.register(String.class, new StringConverter());
98        registry
99              .register(String[].class, new StringArrayConverter(new String[0]));
100       registry.register(Date.class, new SqlDateConverter());
101       registry.register(Time.class, new SqlTimeConverter());
102       registry.register(Timestamp.class, new SqlTimestampConverter());
103       registry.register(File.class, new FileConverter());
104       registry.register(URL.class, new URLConverter());
105 
106       registry.register(BigDecimal.class, new BigDecimalConverter());
107       registry.register(Enum.class, new EnumConverter());
108       
109       if (EnumHelper.getInstance().supportsEnum()) {
110          registry.register(EnumHelper.getInstance().getEnumClass(), new net.java.dev.genesis.commons.beanutils.converters.EnumConverter());
111       }
112       
113       registry.register(Object.class, new DefaultConverter());
114    }
115 
116    // To avoid NPE when ConvertUtilsBean constructor calls 
117    // register(converter,class). The registry instance would be null at that 
118    // time due to the initialization order of member variables
119    private Registry getRegistry() {
120       if (registry == null) {
121          registry = new Registry();
122       }
123       return registry;
124    }
125 
126    public void register(Converter converter, Class clazz) {
127       getRegistry().register(clazz, converter);
128    }
129 
130    public void deregister() {
131       getRegistry().deregister();
132    }
133 
134    public void deregister(Class clazz) {
135       getRegistry().deregister(clazz);
136    }
137 
138    public Converter lookup(Class clazz) {
139       return (Converter)registry.get(clazz, true);
140    }
