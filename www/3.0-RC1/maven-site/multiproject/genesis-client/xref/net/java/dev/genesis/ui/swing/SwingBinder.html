49  /***
50   * <code>SwingBinder</code> is the default implementation of
51   * <code>AbstractBinder</code> for Swing. This class is
52   * used to connect a Swing <code>Component</code> to a <code>form</code>.
53   * Events generated by the <code>Component</code> are handled by
54   * <code>SwingBinder</code> and passed to the <code>form</code>.
55   * If the <code>form</code> state changes, <code>SwingBinder</code> will
56   * reflect that in <code>Component</code>.
57   */

144    public SwingBinder(Component component,

146       super(component, form, handler, lookupStrategy);
147       this.listener = createFormControllerListener();
148       this.defaultButtonListener = bindDefaultButton && hasDefaultButton() ? createDefautButtonListener()

307    protected void unbindDefaultButton() {
308       if (defaultButtonListener == null) {
309          return;
310       }
311 
312       final JButton defaultButton = ((RootPaneContainer) getRoot())
313             .getRootPane().getDefaultButton();
314 
315       if (defaultButton != null) {
316          defaultButton.removeActionListener(defaultButtonListener);
317       }
318    }
319 
320    protected void bindDefaultButton() {
321       if (defaultButtonListener == null) {
322          return;
323       }
324 
325       final JButton defaultButton = ((RootPaneContainer) getRoot())
326             .getRootPane().getDefaultButton();
327 
328       defaultButton.addActionListener(defaultButtonListener);
329    }

336    protected ActionListener createDefautButtonListener() {
337       return new ActionListener() {
338          public void actionPerformed(ActionEvent event) {
339             Component defaultButton = (Component)event.getSource();
340             Component c = FocusManager.getCurrentManager().getFocusOwner();
341 
342             if (c != null) {
343                c.dispatchEvent(new FocusEvent(defaultButton, 
344                      FocusEvent.FOCUS_LOST));
345             }
346 
347             defaultButton.dispatchEvent(new FocusEvent(defaultButton, 
348                   FocusEvent.FOCUS_GAINED));
349          }
350       };
351    }

357    protected FormControllerListener createFormControllerListener() {
358       return (FormControllerListener) Proxy.newProxyInstance(Thread
359             .currentThread().getContextClassLoader(),
360             new Class[] { FormControllerListener.class },
361             new InvocationHandler() {
362                public Object invoke(Object proxy, final Method method,
363                      final Object[] args) throws Throwable {
364 
365                   if (method.getDeclaringClass() != FormControllerListener.class
366                         || EventQueue.isDispatchThread()) {
367                      if (isHashCode(method)) {
368                         return new Integer(SwingBinder.this.listener.hashCode());
369                      } else if (isEqualMethod(method)) {
370                         return Boolean
371                               .valueOf(SwingBinder.this.listener == args[0]);
372                      }
373 
374                      try {
375                         return method.invoke(SwingBinder.this, args);
376                      } catch (InvocationTargetException ite) {
377                         throw ite.getTargetException();
378                      }
379                   }
380 
381                   final Object[] result = new Object[1];
382                   final Throwable[] throwable = new Throwable[1];
383 
384                   EventQueue.invokeAndWait(new Runnable() {
385                      public void run() {
386                         try {
387                            result[0] = method.invoke(SwingBinder.this, args);
388                         } catch (IllegalArgumentException ex) {
389                            throwable[0] = ex;
390                         } catch (InvocationTargetException ex) {
391                            throwable[0] = ex.getTargetException();
392                         } catch (IllegalAccessException ex) {
393                            throwable[0] = ex;
394                         }
395                      }
396                   });
397 
398                   if (throwable[0] != null) {
399                      throw throwable[0];
400                   }
401 
402                   return result[0];
403                }
404 
405                private boolean isEqualMethod(Method method) {
406                   return "equals".equals(method.getName())
407                         && method.getParameterTypes().length == 1
408                         && method.getParameterTypes()[0] == Object.class;
409                }
410 
411                private boolean isHashCode(Method method) {
412                   return "hashCode".equals(method.getName())
413                         && method.getParameterTypes().length == 0;
414                }
415             });
