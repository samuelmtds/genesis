38  public class TextWidgetBinderTest extends GenesisTestCase {
39     private Shell root;
40     private Text text;
41     private MockSWTBinder binder;
42     private WidgetBinder widgetBinder;
43     private BoundField boundField;
44     private MockForm form;
45     private FieldMetadata fieldMeta;
46  
47     public TextWidgetBinderTest() {
48        super("Text Widget Binder Unit Test");
49     }
50  
51     protected void setUp() throws Exception {
52        text = new Text(root = new Shell(), SWT.NONE);
53        binder = new MockSWTBinder(root, form = new MockForm(), new Object());
54        binder.register("text", text);
55        widgetBinder = binder.getWidgetBinder(text);
56        fieldMeta = form.getFormMetadata().getFieldMetadata("stringField");
57     }
58  
59     public void testSetValue() throws Exception {
60        assertTrue(widgetBinder instanceof TextWidgetBinder);
61        assertNull(widgetBinder
62              .bind(binder, text, (DataProviderMetadata) null));
63        assertNull(widgetBinder.bind(binder, text, (ActionMetadata) null));
64  
65        boundField = widgetBinder.bind(binder, text, fieldMeta);
66        assertNotNull(boundField);
67  
68        boundField.setValue("someValue");
69        assertEquals("someValue", text.getText());
70  
71        boundField.setValue("  ");
72        assertEquals("  ", text.getText());
73  
74        boundField.setValue(null);
75        assertEquals("", text.getText());
76     }
77  
78     public void testUpdateValue() throws Exception {
79        assertNull(widgetBinder.bind(binder, text,
80              (DataProviderMetadata) null));
81        assertNull(widgetBinder.bind(binder, text, (ActionMetadata) null));
82        assertNotNull(widgetBinder.bind(binder, text, fieldMeta));
83  
84        text.setText("someValue");
85        simulateFocusLost(text);
86        assertEquals("someValue", binder.get("populateForm(FieldMetadata,Object)"));
87  
88        text.setText("  ");
89        simulateFocusLost(text);
90        assertEquals("", binder.get("populateForm(FieldMetadata,Object)"));
91  
92        text.setText("");
93        simulateFocusLost(text);
94        assertEquals("", binder.get("populateForm(FieldMetadata,Object)"));
95     }
96  
97     public void testUpdateValueAsYouType() throws Exception {
98        text.setData(AbstractBinder.BINDING_STRATEGY_PROPERTY, 
99              AbstractBinder.BINDING_STRATEGY_AS_YOU_TYPE);
100       assertNull(widgetBinder.bind(binder, text, (DataProviderMetadata) null));
101       assertNotNull(widgetBinder.bind(binder, text, fieldMeta));
102 
103       simulateTyping(text, "someValue");
104       assertEquals("someValue", binder.get("populateForm(FieldMetadata,Object)"));
105 
106       simulateTyping(text, " ");
107       assertEquals("", binder.get("populateForm(FieldMetadata,Object)"));
108    }

133    protected void simulateFocusLost(Widget component) {
134       Event event = new Event();
135       event.widget = component;
136       event.button = 1;
137       event.type = SWT.FocusOut;
138       component.notifyListeners(event.type, event);
139    }
140    
141    protected void simulateTyping(Text component, String text) {
142       component.setText("");
143 
144       final char[] chars = text.toCharArray();
145 
146       for (int i = 0; i < chars.length; i++) {
147          component.insert(String.valueOf(chars[i]));
148       }
149    }
