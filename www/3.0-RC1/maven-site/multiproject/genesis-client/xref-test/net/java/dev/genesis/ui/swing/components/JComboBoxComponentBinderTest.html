257

259       assertNotNull(boundField = (BoundField) componentBinder.bind(binder,

263 
264       Object value = model.getElementAt(3);
265       boundField.setValue(value);
266       assertEquals(value, combo.getSelectedItem());
267       assertEquals(3, combo.getSelectedIndex());
268 
269       value = model.getElementAt(0);
270       boundField.setValue(value);

272       assertEquals(0, combo.getSelectedIndex());
